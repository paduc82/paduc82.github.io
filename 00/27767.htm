ossiemac say:

I assume that you want to create the combobox on a worksheet. If so, there are 3 examples below using different methods of positioning, sizeing and creating the list.

 

 

Sub Macro1()

    Dim ws As Worksheet
    Dim ctlCombo As OLEObject
   
    Set ws = Worksheets("Sheet1")
   
    Set ctlCombo = ws.OLEObjects.Add _
                (ClassType:="Forms.ComboBox.1", _
                Link:=False, _
                DisplayAsIcon:=False)
               
    With ctlCombo
        .Name = "cboMyName1"
        .Left = 140
        .Top = 30
        .Width = 150
        .Height = 25
        .ListFillRange = "Sheet1!A1:A10"    'Values in cells A1:A10 become the list
    End With

End Sub

 

 

Sub Macro2()

    Dim ws As Worksheet
    Dim ctlCombo As OLEObject
    Dim i As Long
   
    Set ws = Worksheets("Sheet1")
   
    Set ctlCombo = ws.OLEObjects.Add _
                (ClassType:="Forms.ComboBox.1", _
                Link:=False, _
                DisplayAsIcon:=False)
               
    With ctlCombo
        .Name = "cboMyName2"
        .Left = ws.Range("D10").Left    'Position relative to a specific cell
        .Top = ws.Range("D10").Top      'Position relative to a specific cell
        .Width = 150
        .Height = 25
    End With
   
    'Following adds 1 to 10 to the ComboBox List.
    With ctlCombo.Object
        For i = 1 To 10
            .AddItem i
        Next i
    End With


End Sub

 

 

Sub Macro3()

    Dim ws As Worksheet
    Dim ctlCombo As OLEObject
    Dim i As Long
    Dim arr(1 To 10)
   
    Set ws = Worksheets("Sheet1")
   
    Set ctlCombo = ws.OLEObjects.Add _
                (ClassType:="Forms.ComboBox.1", _
                Link:=False, _
                DisplayAsIcon:=False)
               
    With ctlCombo
        .Name = "cboMyName3"
        .Left = ws.Range("D20").Left    'Position relative to a specific cell
        .Top = ws.Range("D20").Top      'Position relative to a specific cell
        .Width = ws.Columns("D:F").Width  'Set width to total width of columns D:F
        .Height = ws.Rows("20:21").Height        'Set height to total height of rows 20:21
    End With
   
    For i = 1 To 10
        arr(i) = Chr(i + 64)    'Chr(65) is character A. Creates list from A to J
    Next i
       
    ctlCombo.Object.List = arr()
   
    'Alternative code to previous line of code using Array function for populating List.
    'ctlCombo.Object.List = Array("January", "February", "March", "April", _
                    "May", "June", "July", "August", "September", _
                    "October", "November", "December")

End Sub
