<Configuration>

        </Product>
    </Add>
    <Updates Enabled="TRUE" AutoUpgrade="FALSE"/>
    <Display Level="Full" AcceptEULA="TRUE"/>
    <Logging Level="Standard" Path="%temp%"/>
</Configuration>
