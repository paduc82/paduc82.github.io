Hi,

Try this code.

Sub somesub()
Dim WS1 As Worksheet, WS2 As Worksheet
Dim Lastrow As Long
Set WS1 = Sheets("Data")
Set WS2 = Sheets("Result")
Lastrow = WS1.Cells(Rows.Count, "A").End(xlUp).Row
WS1.Range("A1:E" & Lastrow).SpecialCells(xlCellTypeVisible).Copy WS2.Range("A1")
WS2.Columns("C:D").Delete
End Sub
