Function GetIE(sAddress As String) As Object
' Find an IE window with a matching (partial) URL.  Assumes no frames.
'
    Dim objShell As Object, objShellWindows As Object, obj As Object
    Dim retVal As Object, sURL As String
   
    Set retVal = Nothing
    Set objShell = CreateObject("Shell.Application")
    Set objShellWindows = objShell.Windows

        sURL = ""

        End If
    Next obj
   
    Set GetIE = retVal
End Function

To grab the IE instance with a certain title, you can use the following:

 

Option Explicit
 

Public Declare Function SetForegroundWindow Lib "user32" (ByVal hwnd As Long) As Long
 

Private Declare Function ShowWindow Lib "user32" (ByVal hwnd As Long, ByVal nCmdShow As Long) As Long
 
Private Const SW_SHOWNORMAL = 1
Private Const SW_MAXIMIZE = 3
Private Const SW_SHOW = 5
Private Const SW_MINIMIZE = 6
Private Const SW_RESTORE = 9

 
Private Declare Function IsIconic Lib "user32.dll" (ByVal hwnd As Long) As Long
 

Sub LoadPageInIE()
' Add references in the VBE via Tools > References to:
' - Microsoft Internet Controls
' - Microsoft HTML Object Library
'
    Dim IE As InternetExplorer
    Dim strWinTitle As String
   
    strWinTitle = "How to use IE Statusbar text in vba. - Microsoft Community"
   
    On Error GoTo ExitPoint
   
     'Find IE instance with desired Window Title:
    Set IE = GetIEWinTitle(strWinTitle)
   
    If IE Is Nothing Then
        Stop 'What to do if desired window is not found?
'        Exit Sub
    End If
   
    If CBool(IsIconic(IE.hwnd)) Then    'if window is minimized...
        ShowWindow IE.hwnd, SW_RESTORE  'restore the window
    End If
       
    SetForegroundWindow IE.hwnd 'activeate the IE window
   
ExitPoint:
    Set IE = Nothing
End Sub
 
Function GetIEWinTitle(strWinTitleIn As String) As Object
' Find an IE window with a matching title.  Assumes no frames.
'
    Dim objShell As Object, objShellWindows As Object, obj As Object
    Dim retVal As Object, strWinTitle As String
   
    Set retVal = Nothing
    Set objShell = CreateObject("Shell.Application")
    Set objShellWindows = objShell.Windows
    
    For Each obj In objShellWindows
        strWinTitle = ""
        On Error Resume Next
        strWinTitle = obj.Document.Title
        On Error GoTo 0
       
        If strWinTitle = strWinTitleIn Then
            Set retVal = obj
            Exit For
        End If
    Next obj
   
    Set GetIEWinTitle = retVal
End Function

myall_blues say: How are you adding the list? You can create a list in a column of Excel cells (this can go on another sheet if you don't want it visible, and then point to that list as the source for the validation items.
For example, to apply data validation to cell A1 based on a list in column A, sheet 2:
Click in A1
Data->Data validation
On the settings tabs, in Allow select List
In the source box enter or select the range where the validation items are eg: Sheet2!$A$1:$A$100
Click OK

charles kenyon say:

This is easiest if you start (under the Design Tab) with the Black and White Numbered Style set. It has a complete set of numbered headings with the numbering format you want. You can change the actual formatting by modifying the styles. Otherwise, as Suzanne suggested, follow the directions at How to create numbered headings or outline numbering in Ribbon Versions of Word by Shauna Kelly.

I was surprised to note that I could save this Style Set to a template that Word 2010 and 2007 would accept. Apparently all ribbon versions of Word share one folder of custom Style Sets. If you change themes, the style set still works.
