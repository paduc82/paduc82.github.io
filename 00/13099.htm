Function GenArray(ParamArray inputs() As Variant) As Variant
    Dim strList As String
    Dim i As Long, j As Long
   
    If (1 + UBound(inputs) - LBound(inputs)) Mod 2 <> 0 Then
        GenArray = Array("Un-even number of arguments parsed!")
        Exit Function
    End If
   
    strList = ""
    For i = LBound(inputs()) To UBound(inputs()) Step 2
        For j = 1 To inputs(i + 1)
