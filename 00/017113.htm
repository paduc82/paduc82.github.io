Private Function FindAll(ByVal Where As Range, ByVal What, _
    Optional ByVal After As Variant, _
    Optional ByVal LookIn As XlFindLookIn = xlValues, _
    Optional ByVal LookAt As XlLookAt = xlWhole, _
    Optional ByVal SearchOrder As XlSearchOrder = xlByRows, _
    Optional ByVal SearchDirection As XlSearchDirection = xlNext, _
    Optional ByVal MatchCase As Boolean = False, _
    Optional ByVal SearchFormat As Boolean = False) As Range
  'Find all occurrences of What in Where (Windows version)
  Dim FirstAddress As String
  Dim c As Range
  'From FastUnion:
  Dim Stack As New Collection
  Dim Temp() As Range, Item
  Dim i As Long, j As Long

  If Where Is Nothing Then Exit Function
  If SearchDirection = xlNext And IsMissing(After) Then
    'Set After to the last cell in Where to return the first cell in Where in front if _
      it match What
    Set c = Where.Areas(Where.Areas.Count)
    'BUG in XL2010: Cells.Count produces a RTE 6 if C is the whole sheet
    'Set After = C.Cells(C.Cells.Count)
    Set After = c.Cells(c.Rows.Count * CDec(c.Columns.Count))
  End If

  Set c = Where.Find(What, After, LookIn, LookAt, SearchOrder, _
    SearchDirection, MatchCase, SearchFormat:=SearchFormat)
  If c Is Nothing Then Exit Function

  FirstAddress = c.Address
  Do
    Stack.Add c
    If SearchFormat Then
      'If you call this function from an UDF and _
        you find only the first cell use this instead
      Set c = Where.Find(What, c, LookIn, LookAt, SearchOrder, _
        SearchDirection, MatchCase, SearchFormat:=SearchFormat)
    Else
      If SearchDirection = xlNext Then
        Set c = Where.FindNext(c)
      Else
        Set c = Where.FindPrevious(c)
      End If
    End If
    'Can happen if we have merged cells
    If c Is Nothing Then Exit Do
  Loop Until FirstAddress = c.Address

  'FastUnion algorithm ¬© Andreas Killer, 2011:
  'Get all cells as fragments
  ReDim Temp(0 To Stack.Count - 1)
  i = 0
  For Each Item In Stack
    Set Temp(i) = Item
    i = i + 1
  Next
  'Combine each fragment with the next one
  j = 1
  Do
    For i = 0 To UBound(Temp) - j Step j * 2
      Set Temp(i) = Union(Temp(i), Temp(i + j))
    Next
    j = j * 2
  Loop Until j > UBound(Temp)
  'At this point we have all cells in the first fragment
  Set FindAll = Temp(0)
End Function

Private Function Area2Array(ByVal r As Range, _
    Optional ByVal SortRC As Boolean = True) As Variant
  'Returns a 2 dimensional 1 based array, includes only the areas of R
  'If SortRC Then Range("D3:E3,D1:E1,A3:B3,A1:B1")
  '  returns row 1 at the top and column A on the left (as cells are visible in the sheet)
  'Else
  '  returns row 3 at the top and column D on the left (as cells are stored in the range)
  Dim Area As Range
  Dim Cs, Rs, Data, Temp
  Dim i As Long, j As Long, A As Long, B As Long

  'Collect all row and column numbers from the areas
  Set Cs = CreateObject("Scripting.Dictionary")
  Set Rs = CreateObject("Scripting.Dictionary")
  For Each Area In r.Areas
    j = Area.Row - 1
    For i = 1 To Area.Rows.Count
      If Not Rs.Exists(j + i) Then Rs.Add j + i, Rs.Count
    Next
    j = Area.Column - 1
    For i = 1 To Area.Columns.Count
      If Not Cs.Exists(j + i) Then Cs.Add j + i, Cs.Count
    Next
  Next

  'Sort the row and column numbers
  If SortRC Then
    Data = Rs.Keys
    GoSub Sort
    Rs.RemoveAll
    For i = LBound(Data) To UBound(Data)
      Rs.Add Data(i), i
    Next
    Data = Cs.Keys
    GoSub Sort
    Cs.RemoveAll
    For i = LBound(Data) To UBound(Data)
      Cs.Add Data(i), i
    Next
  End If

  'Create space for the output
  ReDim Data(1 To Rs.Count, 1 To Cs.Count)
  For Each Area In r.Areas
    'Read in all values
    Temp = Area
    If Not IsArray(Temp) Then
      ReDim Temp(1 To 1, 1 To 1)
      Temp(1, 1) = Area
    End If
    'Get the position inside data (0 based)
    A = Rs(Area.Row)
    B = Cs(Area.Column)
    'Copy the values
    For i = 1 To UBound(Temp)
      For j = 1 To UBound(Temp, 2)
        Data(i + A, j + B) = Temp(i, j)
      Next
    Next
  Next
  Area2Array = Data
  Exit Function

Sort:
  'Insertionsort ascending
  For i = LBound(Data) + 1 To UBound(Data)
    Temp = Data(i)
    For j = i - 1 To LBound(Data) Step -1
      If Data(j) <= Temp Then Exit For
      Data(j + 1) = Data(j)
    Next
    Data(j + 1) = Temp
  Next
  Return
End Function

ravidavi say:

The data counter at the bottom-right of Nokia Maps only tells you data usage directly by the Maps app when in "online" mode. This includes map downloads, traffic updates, and online searches.  It does NOT include AGPS usage.

 

As you seem to know, you can use GPS free of charge if you disable AGPS.  As someone has said above, you can enable AGPS when you're having trouble getting a regular GPS fix.

 

Now as far as AGPS data usage goes, here's what I've noticed.  When you first start up the GPS module (ie via running Nokia Maps), AGPS will send 2kb and receive 9kb of data to get the initial fix.  After that the AGPS network connection is shut down and GPS works on its own (since it now knows the locations of satellies).  Future AGPS uses after this initial fix has been obtained use LESS data than the initial "cold" fix.

 

In an area of very bad reception, your phone may have to use AGPS more often.  It will also have to re-use AGPS if you shut down Maps for a while, or if you're in an area of bad reception where the GPS needs more frequent help.

 

So you can see why neither Nokia nor your provider was willing to give you a firm answer on costs.  It's because how often your phone uses AGPS depends GREATLY on where you are.  If you're in a good area, you may only need the initial AGPS use for a total of 11kb ... but if you travel to a bad reception area, you may use up another 50 kb.

 

You said that your provider only charges $0.005/kb?  So you'd have to use 200kb of AGPS data to have Singapore $1 in charges.  I'm not sure whether that is a "lot" or "little", but you can use this information how you see fit.

 

Good luck!

Ravi
