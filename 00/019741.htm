sample

vba macro

Sub RemoveDuplicates_Join()
'June 16, 2015
Dim r1 As Long, r2 As Long, i As Long
r1 = Cells(Rows.Count, 1).End(xlUp).Row
Range("A1:B" & r1).Copy Range("E1")
Range("E1:F" & r1).RemoveDuplicates Columns:=Array(1, 2), Header:=xlNo
r2 = Cells(Rows.Count, "E").End(xlUp).Row
For i = 1 To r2
s = ""
For Each r In Range("A1:A" & r1)
If r.Value = Cells(i, "E").Value And r.Offset(, 1).Value = Cells(i, "F").Value Then s = s & " " & r.Offset(, 2).Value
Next
Cells(i, "G").Value = s
Next
End Sub
