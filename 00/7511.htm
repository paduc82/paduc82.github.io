Hi

 

This code should go in the relevant Sheet module

 

Private Sub Worksheet_Change(ByVal Target As Range)
Dim cl As Range
For Each cl In Target
   If cl.Column = 6 Or cl.Column = 7 Or cl.Column = 8 Then
      Select Case True
         Case cl = "-"
            cl.Interior.ColorIndex = 3
         Case IsDate(cl.Text)
            cl.Interior.ColorIndex = 5
         Case cl = ""
            cl.Interior.ColorIndex = 0
      End Select
   Else
      cl.Interior.ColorIndex = 0
   End If
Next cl
End Sub
