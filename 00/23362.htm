    jgkzdl say:

    Nick,

    blnResult = False is not really needed because blnResult is always false until it is set True, but here used for completeness.

    Youre basefield is Sales and the pivotfield made by youre code has the caption "Sum of Sales".

    Then every time 'Set pf = pt.PivotFields("Sales")' makes a new Pivotfield and that is still Hidden, so pf.Orientation is zero (xlHidden).

    (and Hans said it already:  pf.Orientation = xlDataField = True has to be pf.Orientation = xlDataField for testing pf.Orientation or (pf.Orientation = xlDataField) = True ).

    Perhaps this is the better way to test if a pivotfield is present, you don't need to change the default name:

    Sub Import()
        Dim ws As Worksheet, pvtCache As PivotCache
        Dim wsp As Worksheet
        Dim pf As PivotField, pt As PivotTable
        Dim strName As String

        Set pt = wsp.PivotTables("PivotTable")

            Set pf = pt.PivotFields(strName)
            With pf
                .Orientation = xlDataField
                .Function = xlSum
            End With
            pt.PivotCache.Refresh
        End If

    Function FieldExists(pt As PivotTable, strName As String) As Boolean
        Dim fld As PivotField
        Dim blnResult As Boolean
        blnResult = False
        For Each fld In pt.VisibleFields
            If fld.SourceName = strName Then
                blnResult = True
                Exit For
            End If
        Next
        FieldExists = blnResult
    End Function

    nvx say: So, after not being able to find any free service or tool that would do what I needed I resorted to a workaround of sorts:
    1. subscribe to the Google calendar in Outlook.com
    2. add the Outlook.com account into desktop Outlook
    3. write a little VBA macro that takes all events having their end date between (today - 1 month) and (today + 6 months) from the remote Google calendar (or any other available calendar for that matter) and synces them into the default Outlook.com calendar (everything else is removed from the default calendar)
    4. auto-run the macro via a timer every 3 hours
    5. set-up an Exchange ActiveSync account on the phone (that is, link the phone to the Outlook.com account)
    Since my computer is on 24/7 due to scientific computations being performed continuously, virtually nothing has changed for me. I only needed to set up Outlook to auto-start with the PC to make sure that I won't forget this when restarting e.g. because of updates.

    Points to keep in mind:
    • The data source is not "live" -- it only gets updated every so often. Also, the sync is not performed when changes are made to the source data, but at a pre-set interval. Anyone requiring 100% up-to-date data will find this workaround unusable. (As far as I know, it is not possible to take a source iCal feed and import it into an Outlook calendar via VBA without any additional libraries being loaded. If this could be done, then one could always have 100% up-to-date source data.)
    • The synchronization is one-way only (source calendar(s) -> default Outlook.com calendar). A more complex synchronization algorithm would be necessary for a two-way sync.
    • Reminders are not synchronized due to a bug in Outlook.com (no reminders are visible in the remote calendar even though in the source iCal feed they are present).
    • Outlook must run for the synchronization to take place.
    • People with a cell phone data plan may notice increased data usage should the VBA synchronization algorithm be rather rough (e.g. delete all from the default calendar, then mirror every relevant event from the source calendars). But then again, with a more sophisticated algorithm there should be no significant difference when compared to typical synchronization scenarios.
    In spite of all the limitations, this suits my needs just fine.

    P.S.: I am hesitant to mark this as a solution, because it does not solve the problem per se. As mentioned before, it is more of a workaround...

    --------------------------------------------------------------------------
    Code to be put into ThisOutlookSession
    ==========================================================================

    Private Sub Application_Startup()
        Call activateTimer(5)
    End Sub


    Private Sub Application_Quit()
        If timerID <> 0 Then Call deactivateTimer
        ' Due to timers, we cannot sync on quit -- Outlook would hang!
        'Call syncCalendars
    End Sub

    --------------------------------------------------------------------------
    Code to be put into a Module
    ==========================================================================

    Declare PtrSafe Function SetTimer Lib "user32" (ByVal hwnd As LongPtr, ByVal nIDEvent As LongPtr, ByVal uElapse As LongPtr, _
            ByVal lpTimerfunc As LongPtr) As LongPtr
    Declare PtrSafe Function KillTimer Lib "user32" (ByVal hwnd As LongPtr, ByVal nIDEvent As LongPtr) As LongPtr
    Declare PtrSafe Function InternetGetConnectedState Lib "wininet.dll" (ByRef dwflags As LongPtr, _
            ByVal dwReserved As LongPtr) As LongPtr
    Public timerID As LongPtr


    Public Sub activateTimer(ByVal nMinutes As LongPtr)
        ' Is the timer running? If so, we must deactivate it first.
        If timerID <> 0 Then
            deactivateTimer
        End If
        ' We need to convert the delay to milliseconds
        nMinutes = nMinutes * 1000 * 60
        timerID = SetTimer(0, 0, nMinutes, AddressOf triggerTimer)
        If timerID = 0 Then
            MsgBox "The auto-sync timer failed to activate!"
        End If
    End Sub


    Public Sub deactivateTimer()
        Dim lSuccess As LongPtr
        lSuccess = KillTimer(0, timerID)
        If lSuccess = 0 Then
            MsgBox "The auto-sync timer could not be deactivated!"
        Else
            timerID = 0
        End If
    End Sub


    Private Sub triggerTimer(ByVal hwnd As LongPtr, ByVal uMsg As LongPtr, ByVal idevent As LongPtr, ByVal Systime As LongPtr)
        If idevent = timerID Then
            If InternetGetConnectedState(0&, 0&) Then
                activateTimer 180
                syncCalendars
            Else
                activateTimer 5
            End If
        End If
    End Sub


    Public Sub syncCalendars()
        Dim daysBefore As Integer
        Dim daysAfter As Integer
        Dim waitTime As String
        Dim nS As Outlook.NameSpace
        Set nS = Application.GetNamespace("MAPI")
        ' Default folder = target calendar!
        Dim tgtCal As Folder
        Set tgtCal = nS.GetDefaultFolder(olFolderCalendar)
        ' How many days before today should we synchronize?
        daysBefore = 30
        ' How many days after today should we synchronize?
        daysAfter = 180
        ' How long should we wait after each sub-operation so that the changes get propagated to
        ' the server? (prevents creation of duplicates of events; minimum seems to be ~30 secs)
        waitTime = "00:01:00"
        ' First we must delete all events in the default calendar
        Set itemsTgt = tgtCal.Items
        For i = itemsTgt.Count To 1 Step -1
            itemsTgt(i).Delete
        Next i
        ' WAIT FOR THE CHANGES TO PROPAGATE TO THE SERVER
        wait waitTime
        ' Copy events from the Google calendar added to the desktop Outlook as a remote ICS feed
        ' (that is, NOT via the subscribsion in Outlook.com, because this way it is updated much faster!)
        copyEvents Session.GetDefaultFolder(olFolderCalendar).Parent.Parent.Folders("Internet Calendars").Folders("Google ICS"), _
                tgtCal, daysBefore, daysAfter, ""
        ' WAIT FOR THE CHANGES TO PROPAGATE TO THE SERVER
        wait waitTime
        ' Clean-up
        Set nS = Nothing
        Set tgtCal = Nothing
        Set itemsTgt = Nothing
        ' Let's add a note to the log file
        addNoteToLog
    End Sub


    Private Sub wait(ByVal timeStr As String)
        Dim currTime As Date
        currTime = Now
        Do Until currTime + TimeValue(timeStr) <= Now
        Loop
    End Sub


    Private Sub copyEvents(ByRef srcCal As Folder, ByRef tgtCal As Folder, ByVal daysBefore As Integer, _
            ByVal daysAfter As Integer, ByVal prefix As String)
        ' Correction of possible invalid inputs
        If daysBefore < 0 Then
            daysBefore = 0
        End If
        If daysAfter < 0 Then
            daysAfter = 30
        End If
        ' Now let's copy the events
        Set itemsSrc = srcCal.Items
        For i = itemsSrc.Count To 1 Step -1
            If (itemsSrc(i).End >= (Now - daysBefore)) And (itemsSrc(i).End <= (Now + daysAfter)) Then
                itemsSrc(i).CopyTo tgtCal, OlAppointmentCopyOptions.olCreateAppointment
                ' This is necessary both to ensure that events are actually uploaded to the remote calendar
                ' and to add a prefix if required
                Dim appt As AppointmentItem
                Set appt = tgtCal.Items.GetLast
                With appt
                    .Subject = prefix & appt.Subject
                    .Save
                End With
                Set appt = Nothing
            End If
        Next i
        ' Clean-up
        Set itemsSrc = Nothing
    End Sub


    Private Sub addNoteToLog()
        Dim fso, stream
        Set fso = CreateObject("Scripting.FileSystemObject")
        ' We need to append text, hence the parameter '8' below...
        Set stream = fso.OpenTextFile("D:\Documents\CalendarSyncLog.log", 8, True)
        stream.WriteLine "Last sync attempt: " & Now
        stream.Close
        Set stream = Nothing
        Set fso = Nothing
    End Sub

    --------------------------------------------------------------------------

    Declare Function SetTimer Lib "user32" (ByVal hwnd As Long, ByVal nIDEvent As Long, ByVal uElapse As Long, _
            ByVal lpTimerfunc As Long) As Long
    Declare Function KillTimer Lib "user32" (ByVal hwnd As Long, ByVal nIDEvent As Long) As Long
    Public timerID As Long

    would be enough to make it work in 32-bit Outlook -- you will need to do some debugging in that case.
