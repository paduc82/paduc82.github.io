You can try it on a copy to see if it does what you need. Obviously you should specify a different output folder.

Sub ExportSlidesAsPresentations()
   Dim oPres As Presentation
   Dim oExportedPres As Presentation
   Dim oSld As Slide
   Dim sSlideOutputFolder As String
   Dim sFilename As String
   'alter this to suit
   sSlideOutputFolder = "C:\Users\Optiplex\Desktop\Slides\"
   Set oPres = ActivePresentation
   For Each oSld In oPres.Slides
      sFilename = (sSlideOutputFolder) & "Slide" & _
                  Format(oSld.SlideIndex, "000") & ".ppt"
      oSld.Export sFilename, "PPT"
      Set oExportedPres = Presentations.Open(FileName:=sFilename, WithWindow:=False)
      Call oExportedPres.SaveAs(FileName:=sFilename & "x", _
      FileFormat:=ppSaveAsOpenXMLPresentation, EmbedTrueTypeFonts:=False)
      Kill sFilename
      oExportedPres.Close
      Set oExportedPres = Nothing
   Next oSld
   Set oPres = Nothing
End Sub
