Hi RD4AG,

When you create a task list in a site in SharePoint Online and open it in Project 2016, the custom fields may not be synced. To make it synced, you may manually add the new field such as Text 1, Text 2 and Text 3 in Project 2016 and then rename them as crop, farm description and place(for example).

Please follow the steps below to make custom fields synced between SharePoint Online and Project 2016.
1. Create a task list and use the option Open with Project to export it in Project 2016.
2. After exporting it, you will see that the custom fields can’t be successfully synced.
3. Navigate to Add New Column and select Text 1.
4. Right-click Text 1 and select Field Settings.
5. Type crop in Title.

6. Do the same steps to add farm description and place.
7. Populate the values in the custom columns.
8. Go to File>Info>Map Fields>select Text 1, Text 2 and Text 3 in Project Field and check the Sync checkboxes.

Private Sub Worksheet_Change(ByVal Target As Range)
On Error Resume Next
Const ShapeCell As String = "S"  ' <<< Target Shape Column
Application.ScreenUpdating = False
If Not Intersect(Target, Range("H:H")) Is Nothing Then '<<< codes in column H
Application.DisplayAlerts = False

Select Case Target.Value
Case Is = "111"
'you can place in range AA1:AD2 the GroupShape that refers in code 111
Set rng = ActiveSheet.Range("AA1:AD2")

Case Is = "200"
'..also in range AA3:AD4 the GroupShape that refers in code 200
Set rng = ActiveSheet.Range("AA3:AD4")

Case Is = "501"
'also.. in AA5:AD6....in code 500
Set rng = ActiveSheet.Range("AA5:AD6")

Case Is = "900"
'also.. in AA7:AD8....in code 900
Set rng = ActiveSheet.Range("AA7:AA8")
Case Else
MsgBox "wrong code.."
Exit Sub
End Select

For Each s In ActiveSheet.Shapes
If Not Intersect(s.TopLeftCell, ActiveSheet.Cells(Target.Row, ShapeCell)) Is Nothing Then s.Delete
Next s

For Each s In ActiveSheet.Shapes
If Not Intersect(s.TopLeftCell, rng) Is Nothing Then
s.Select
Selection.ShapeRange.Ungroup
End If
Next s


'start

'GROUP METHOD BY TOM URTIS.
'http://www.mrexcel.com/forum/excel-questions/279115-visual-basic-applications-how-group-selected-shapes.html
Dim shp As Shape, i%, varShape() As Variant 'TOM
'Application.ScreenUpdating = False
i = 0 'TOM
For Each shp In ActiveSheet.Shapes 'TOM
If Not Intersect(shp.TopLeftCell, rng) Is Nothing Then
shp.Select
Selection.Text = Cells(Target.Row, 9 + i).Value
'If Left(shp.Name, 3) = "pre" Then 'TOM
ReDim Preserve varShape(i) 'TOM
varShape(i) = shp.Name ' TOM
i = i + 1 'TOM
End If 'TOM
Next shp
ActiveSheet.Shapes.Range(varShape).Select 'TOM
Selection.ShapeRange.Group

'end


For Each s In ActiveSheet.Shapes
If Not Intersect(s.TopLeftCell, rng) Is Nothing Then
s.Copy
With ActiveSheet
.Cells(Target.Row, ShapeCell).Select
.Paste
End With
End If
Next s
For Each s In ActiveSheet.Shapes
If Not Intersect(s.TopLeftCell, Cells(Target.Row, ShapeCell)) Is Nothing Then
Set rr = Cells(Target.Row, ShapeCell)
With s
    .Height = rr.Height
    .Width = rr.Width
    .Top = rr.Top
    .Left = rr.Left
End With
End If
Next s
Cells(Target.Row, ShapeCell).Select
End If
Application.DisplayAlerts = True
Application.ScreenUpdating = True
On Error GoTo 0
End Sub
