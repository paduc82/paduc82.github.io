Something on the order of:

Private Sub chkMyCheckbox_AfterUpdate()
Dim iAns As Integer
On Error GoTo Proc_Error
If Me!chkMyCheckbox = True Then
   iAns = vbYes
   If Not IsNull(Me!CheckAmt) Then
      iAns = MsgBox("Check amount already filled in! Should I stomp on it?", vbYesNo)

   If iAns = vbYes Then

Proc_Exit:

Proc_Error:

Resume Proc_Exit
End Sub
