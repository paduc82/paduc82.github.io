Hi,

try the 'shape property', in order to change fill color

sample

add an icon in range c1:d2

Sub Add_OLE_Object_Icon()
Dim ws As Worksheet
Set ws = ActiveSheet
Dim obj As OLEObject
Dim sPath As String

sPath = "c:\users\username\desktop\test.xlsx"

Application.ScreenUpdating = False
For Each p In ws.Shapes '<< delete old shape from cell C1/topleftcell property
If Not Intersect(p.TopLeftCell, ws.Range("c1:d2")) Is Nothing Then p.Delete
Next
Set obj = ws.OLEObjects.Add(Filename:=sPath, Link:=False, DisplayAsIcon:=True)
With obj
.Top = Range("c1").Top
.Left = Range("c1").Left

.Width = Range("c1:d2").Width
.Height = Range("c1:d2").Height

End With
ws.Shapes(obj.Name).Fill.ForeColor.RGB = vbRed
Application.ScreenUpdating = True
End Sub
