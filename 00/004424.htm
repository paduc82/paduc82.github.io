Sub Hide_I_Rows()
  Dim RngCol As Range, i As Range
  Dim FirstAddress As String
  Set RngCol = Range("I1", Range("I" & Rows.Count).End(xlUp))
  'Search for the first value
  Set i = RngCol.Find("Complete", LookIn:=xlValues, LookAt:=xlWhole)
  'Found?
  If i Is Nothing Then Exit Sub
  'Remember this position
  FirstAddress = i.Address
  Do
    i.EntireRow.Hidden = True
    'Find next value
    Set i = RngCol.FindNext(i)
  Loop Until i.Address = FirstAddress
End Sub
