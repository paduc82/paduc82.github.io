Need help in creating Excel macro to change the font color based on the following:

Green: Direct link from same worksheet (=A25)

Black: Formulas (example: =VLOOKUP; =sum(B1:B25))

andreas killer say:

Sub Test()
  Dim All As Range, R As Range
  Dim Links(0 To 2) As Range
  Dim i As Integer
  With ActiveSheet
    'Blue: Hard-coded values (text or numbers)
    Set All = SpecialCells(.UsedRange, xlCellTypeConstants, xlNumbers Or xlTextValues)
    If Not All Is Nothing Then All.Font.ColorIndex = 5
    Set All = SpecialCells(.UsedRange, xlCellTypeFormulas)
    If Not All Is Nothing Then
      For Each R In All
        If InStr(R.Formula, "!") > 0 Then
          'Purple: Link from another worksheet in the same file (example: =Sheet2!D20)
          i = 0
        ElseIf InStr(R.Formula, "(") > 0 Then
          'Black: Formulas (example: =VLOOKUP; =sum(B1:B25))
          i = 1
        Else
          'Green: Direct link from same worksheet (=A25)
          i = 2
        End If
        If Links(i) Is Nothing Then
          Set Links(i) = R
        Else
          Set Links(i) = Union(Links(i), R)
        End If
      Next
      For i = 0 To 2
        If Not Links(i) Is Nothing Then
          Select Case i
            Case 0
              Links(i).Font.ColorIndex = 7
            Case 1
              Links(i).Font.ColorIndex = xlColorIndexAutomatic
            Case 2
              Links(i).Font.ColorIndex = 10
          End Select
        End If
      Next
    End If
  End With
End Sub

Private Function SpecialCells(ByVal R As Range, ByVal Typ As XlCellType, _
    Optional ByVal Value As XlSpecialCellsValue = &H17) As Range
  'Avoid the SpecialCells-BUG to return all cells from the current region
  On Error Resume Next
  Select Case Typ
    Case xlCellTypeConstants, xlCellTypeFormulas
      Set SpecialCells = Intersect(R, R.SpecialCells(Typ, Value))
    Case Else
      Set SpecialCells = Intersect(R, R.SpecialCells(Typ))
  End Select
End Function
