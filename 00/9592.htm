julie fka profjulie say:

If you hard reset your phone and then set it up to use a different Microsoft account, you will not be able to restore any backed up data that is stored in the previous account to the phone.  You won't lose that data, the data will still be stored in the old Microsoft account, but your phone will have no way to get to it.

Because the billing history for any apps or games you purchased is stored in the old Microsoft account, you will have to repurchase those apps and games if you want to use them on the phone after you've set it up under the new Microsoft account.

You can, however, set up a secondary account on the phone to sync any contacts and calendar items that are stored in the old Microsoft account, and you can share the files that are stored in the old Microsoft account's OneDrive with the new Microsoft account and access them from the phone.

All you need to do is change the login email address! You DO NOT need to use a different Microsoft Account.

From a computer Go log into your Microsoft account

Create an alias on the account (this will be a outlook.com email address)

Change the primary alias (which is now the yahoo email address) to the new alias

Test to make sure you can log into your Microsoft account with the new email address (password will stay the same unless you change it).

Option Explicit

Sub Test()
  Dim Arr
  Dim R As Range
  Dim C As Collection
  'read the values form column A
  Set R = Range("A1", Range("A" & Rows.Count).End(xlUp))
  Arr = R
  With WorksheetFunction
    'Build one dimensional array
    Arr = .Transpose(Arr)
    'Build all combinations
    Set C = CombinationsL(Arr)
    'Check each
    For Each Arr In C
      'Match?
      If .Sum(Arr) = 10 Then
        'Show the values (Search each in column A and show the address of the cell)
        Debug.Print Join(Arr)
      End If
    Next
  End With
End Sub

Function CombinationsL(Arr) As Collection
  'Return all possible combinations of count elements in Arr as arrays in a collection
  Dim Index() As Long
  Dim Result()
  Dim i As Long, j As Integer, k As Integer, Count As Long
  'Create a collection
  Set CombinationsL = New Collection
  'Return empty collection if Count exceed the bound's of Arr
  If UBound(Arr) < LBound(Arr) Then Exit Function
  Count = UBound(Arr) - LBound(Arr) + 1
  'Setup space for index and result array
  ReDim Index(1 To Count) As Long
  For k = 1 To Count
    ReDim Result(1 To k)
    'Setup index for first combination
    For i = 1 To k
      Index(i) = LBound(Arr) + i - 1
    Next
    Do
      'Build combination
      For i = 1 To k
        Result(i) = Arr(Index(i))
      Next
      'Save it
      CombinationsL.Add Result
      'Get next index
      i = k
      j = 0
      Do
        'Max. position for this index reached?
        If Index(i) = UBound(Arr) - k + i Then
          j = j + 1
          'Move index before up one step in next loop
          i = i - 1
          If i < 1 Then GoTo NextCount
        Else
          'Increment position
          Index(i) = Index(i) + 1
          'Setup next indices
          For j = 1 To j
            i = i + 1
            Index(i) = Index(i - 1) + 1
          Next
          Exit Do
        End If
      Loop
    Loop
NextCount:
  Next
End Function
