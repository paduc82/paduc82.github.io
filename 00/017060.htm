As a worksheet formula I would use this in case there are no value <n

and the vb code could become this which simply tests if we've read anything into the array

 

Sub Mode_If()
 Dim lastrow As Long, x As Long
 Dim GotValue As Boolean, FullRange As Range
 lastrow = Cells(Cells.Rows.Count, "A").End(xlUp).Row
 Set FullRange = Range("A1:A" & lastrow)
 Dim myarray(10000)
 
 For x = 1 To lastrow
     If Cells(x, 1) < Range("B1").Value Then
         myarray(x) = Cells(x, 1).Value
         GotValue = True
     End If
 Next

 
 If GotValue Then
    newvalue = WorksheetFunction.Mode(myarray())
 Else
     newvalue = WorksheetFunction.Mode(FullRange)
End If
 
End Sub
