You can use the following macro and adjust width/height/left/top by trial and error to appropriate values for you.

Sub Macro4()
'
' Macro4 Macro
'
' Keyboard Shortcut: Ctrl+q
'
For Each objCht In ActiveSheet.ChartObjects
With objCht.Chart
                .PlotArea.Width = 125

                .PlotArea.Height = 125
                .PlotArea.Left = 30
                .PlotArea.Top = 100
End With
Next
End Sub
