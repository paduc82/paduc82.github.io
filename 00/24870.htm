ken sheridan say: I wrote a little function for this sort of thing some years ago:

Public Function GetListItems(strItemList As String, _
                    strDelimiter As String, _
                    intStart As Integer, _
                    intNumber As Integer) As String

    Dim intDelimLength As Integer
    Dim intSubStart As Integer
    Dim intSubEnd As Integer
    Dim n As Integer
    
    strItemList = strItemList & strDelimiter
    intDelimLength = Len(strDelimiter)
    intSubStart = 1
    
    ' get start of substring
    If intStart > 1 Then
        For n = 1 To intStart - 1
            intSubStart = InStr(intSubStart, strItemList, strDelimiter) + Len(strDelimiter)
            If intSubStart > Len(strItemList) Then Exit For
        Next n
    End If
    
    intSubEnd = intSubStart
    
    ' get position immediately after end of substring
    For n = 1 To intNumber
        If InStr(intSubEnd + 1, strItemList, strDelimiter) > 0 Then
            intSubEnd = InStr(intSubEnd + 1, strItemList, strDelimiter)
        End If
    Next n
    
    ' return value is substring starting at intSubStart
    ' and of length intSubEnd - intSubStart
    GetListItems = Mid(strItemList, intSubStart, intSubEnd - intSubStart)

End Function
