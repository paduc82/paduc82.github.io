Sub AdvFilter()
    Dim wsSource As Worksheet
    Dim wsDestin As Worksheet
    Dim rngSource As Range
    Dim rngDestin As Range
    Dim rngCriteria As Range
    Dim lngLastRow As Long
    Dim lngLastCol As Long
    Dim i As Integer
   
    Set wsSource = Worksheets("Sheet1")
    Set wsDestin = Worksheets("Sheet2")
   
    'Following section assigns the data in Sheet1 to a range variable
    'It Excludes the Criteria
    'Note: Must be at least one blank column between data and criteria
    With wsSource
        lngLastRow = .Cells(.Rows.Count, "A").End(xlUp).Row
        lngLastCol = .Cells(1, "A").End(xlToRight).Column
        Set rngSource = .Range(.Cells(1, "A"), .Cells(lngLastRow, lngLastCol))
    End With
   
    'Following initializes the first output to the first cell in the Destination worksheet
    With wsDestin
        Set rngDestin = .Cells(1, "A")
    End With
   
    For i = 1 To 3
       
        'Following assigns criteria range to a range variable
        With wsSource
            Set rngCriteria = .Range(.Cells(1, 22 + i), .Cells(2, 22 + i))
        End With
       
        rngSource.AdvancedFilter Action:=xlFilterCopy, _
                CriteriaRange:=rngCriteria, _
                CopyToRange:=rngDestin, _
                Unique:=False
       
        'Following finds the last used row in the destination sheet and offsets down
        'by 4 to provide a space between each output.
        With wsDestin
            'Increment or decrement the 4 in Offset to alter number of lines between each output.
            Set rngDestin = .Cells(.Rows.Count, "A").End(xlUp).Offset(4, 0)
        End With
       
    Next i

End Sub

There is no advanced option in the envelope dialog in Word 2003, so I must presume that it related to the Outlook version which provides the address book function to the envelope tool.

It would not be too difficult to create a macro that would grab the address from the letterhead and save it as an Outlook contact. Something along the lines of the following.

Sub AddOutlookCont()
Dim ol As New Outlook.Application
Dim ci As ContactItem
Dim straddress As String
Dim strName As String
Dim strFullName As String
Dim strBusiness As String
Dim iSplit As Integer
Dim iResult As Integer

straddress = Selection.Range
iResult = MsgBox("Is the address correct?" & _
vbCr & vbCr & straddress, vbYesNo, "Address")
If iResult = 7 Then GoTo UserCancelled:

strFullName = InputBox("Enter contact's full name if known" _
& vbCr & "in the format 'Mr. John Smith", "Contact name")

On Error GoTo UserCancelled:
iSplit = InStr(straddress, Chr(13))
strName = Left(straddress, iSplit - 1)
strBusiness = Right(straddress, (Len(straddress) - Len(strName)))

Set ci = ol.CreateItem(olContactItem)
ci.MailingAddress = strBusiness
ci.CompanyName = strName
If strFullName <> "" Then ci.FullName = strFullName
ci.FileAs = strName
ci.Categories = InputBox("Category?", "Categories")
ci.Save
Set ol = Nothing
Exit Sub
UserCancelled:
MsgBox "User Cancelled or address not selected"
Set ol = Nothing
End Sub
