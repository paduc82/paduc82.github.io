Function IsDeveloper() As Boolean
On Error Resume Next
   
    Dim UserName As String
    IsDeveloper = False
    UserName = VBA.Environ("Username")

'************ Code Start **********
' This code was originally written by Dev Ashish.

Global Const SW_HIDE = 0
Global Const SW_SHOWNORMAL = 1
Global Const SW_SHOWMINIMIZED = 2
Global Const SW_SHOWMAXIMIZED = 3
Private Declare Function apiShowWindow Lib "user32" _
    Alias "ShowWindow" (ByVal hwnd As Long, _
          ByVal nCmdShow As Long) As Long
Function fSetAccessWindow(nCmdShow As Long)
'Usage Examples
'Maximize window:
'       ?fSetAccessWindow(SW_SHOWMAXIMIZED)
'Minimize window:
'       ?fSetAccessWindow(SW_SHOWMINIMIZED)
'Hide window:
'       ?fSetAccessWindow(SW_HIDE)
'Normal window:
'       ?fSetAccessWindow(SW_SHOWNORMAL)
'
Dim loX  As Long
Dim loForm As Form
    On Error Resume Next
    Set loForm = Screen.ActiveForm
    If Err <> 0 Then 'no Activeform
      If nCmdShow = SW_HIDE Then
        MsgBox "Cannot hide Access unless " _
      Else
        loX = apiShowWindow(hWndAccessApp, nCmdShow)
        Err.Clear
      End If
    Else
        If nCmdShow = SW_SHOWMINIMIZED And loForm.Modal = True Then
            MsgBox "Cannot minimize Access with " _
        ElseIf nCmdShow = SW_HIDE And loForm.PopUp <> True Then
            MsgBox "Cannot hide Access with " _
        Else
            loX = apiShowWindow(hWndAccessApp, nCmdShow)
        End If
    End If
    fSetAccessWindow = (loX <> 0)
End Function
'************ Code End **********

No, you misunderstood. You should have a table like this:

tblAcctOwners

AcctOwnerID (PK Autonumber)

OwnerFirst

OwnerLast

other information about the owner

Then your data table should be:

tblKPI

KPIID (PK Autonumber)

AcctOwnerID (FK)

KPIMonth

FiscalYear

then another table

tblKPIScores

KPIScoreID (PK Autonumber)

KPIID (FK)

ScoreDescription

Score

So instead of having fields for each score type you have a table with a record for each score type. This will give you a normalized structure.

To prevent a KPI record that is a duplicate you would create a unique multi-field index on the combination of AcctOwnerID, KPIMonth and FiscalYear.
