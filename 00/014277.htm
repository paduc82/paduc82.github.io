if %errorlevel% == 0 echo Service started successfully.

 

net start “Microsoft Office Service”

if %errorlevel% == 2 echo Could not start service.

if %errorlevel% == 0 echo Service started successfully.

hansv mvp say:

There is an inconsistency in your description: you mention UC, UP, AV, EC as strings to look for, but in the example you mention AL. So check the following macro carefully and modify it to match the correct requirements.

Sub DeleteSpecific()
    Dim r As Long   ' Row
    Dim m As Long   ' Last row
    Dim p As String ' Product
    Dim s As String ' Seller
    Application.ScreenUpdating = False
    m = Range("D" & Rows.Count).End(xlUp).Row
    For r = m To 2 Step -1
        p = Range("D" & r).Value
        Select Case True
            Case p Like "*UC"
                Range("D" & r).EntireRow.Delete
            Case p Like "*UC[#]*"
                Range("D" & r).EntireRow.Delete
            Case p Like "*UP"
                Range("D" & r).EntireRow.Delete
            Case p Like "*UP[#]*"
                Range("D" & r).EntireRow.Delete
            Case p Like "*AV"
                s = Range("B" & r).Value
                Select Case s
                    Case "Nike", "Puma", "Zones"
                        ' Skip
                    Case Else
                        Range("D" & r).EntireRow.Delete
                End Select
            Case p Like "*AV[#]*"
                s = Range("B" & r).Value
                Select Case s
                    Case "Nike", "Puma", "Zones"
                        ' Skip
                    Case Else
                        Range("D" & r).EntireRow.Delete
                End Select
            Case p Like "*EC"
                Range("D" & r).EntireRow.Delete
            Case p Like "*EC[#]*"
                Range("D" & r).EntireRow.Delete
        End Select
    Next r
    Application.ScreenUpdating = True
End Sub

Sub DeleteRowWithConditions()
    Dim lastrow As Long
    Dim i As Long
    lastrow = Range("A" & Rows.Count).End(xlUp).Row
    For i = 2 To lastrow
        If Range("D" & i).Value Like "*UC" Or Range("D" & i).Value Like "*UC[#]*" Or Range("D" & i).Value Like "*UP" Or Range("D" & i).Value Like "*UP[#]*" Or Range("D" & i).Value Like "*EC" Or Range("D" & i).Value Like "*EC[#]*" Then
            Range("D" & i).EntireRow.Delete shift:=xlUp
            lastrow = lastrow - 1
            i = i - 1
        ElseIf (Range("D" & i).Value Like "*AV" Or Range("D" & i).Value Like "*AV[#]*") And (Range("B" & i).Value <> "Puma" And Range("B" & i).Value <> "Zones" And Range("B" & i).Value <> "Nike") Then
            Range("D" & i).EntireRow.Delete shift:=xlUp
            lastrow = lastrow - 1
            i = i - 1
        End If
    Next i
End Sub
