Sub UpdateMasterList()
 
Dim MasterList As Range
Dim UpdateList As Range
Dim DataPoint As Range
Dim Trg As Range
Dim Dp As Variant
Dim Cnt As Long
 
On Error GoTo Stp
''' This line will allow you to select the list with new data
Set Trg = Application.InputBox(Prompt:="Please Select One Cell in Update List", Title:="Update List", Type:=8)
Application.ScreenUpdating = False
Col = Trg.Column
 
Sheets("Update").Activate
Set UpdateList = Sheets("Update").Range(Cells(2, Col), Cells(Rows.Count, Col).End(xlUp))
Cnt = 0
 
''' Looping through all data and check if new data point exists to update Master List
For Each cell In UpdateList
        Dp = cell.Value
        Sheets("MasterList").Activate
        Set MasterList = Sheets("MasterList").Range(Cells(2, "A"), Cells(Rows.Count, "A").End(xlUp))
        Set DataPoint = MasterList.Find(What:=Dp, LookAt:=xlWhole)
        
        If DataPoint Is Nothing Then
                Sheets("MasterList").Activate
                Sheets("MasterList").Cells(Rows.Count, "A").End(xlUp).Offset(1, 0).Value = Dp
                Cnt = Cnt + 1  ''' The counter
        Else
                GoTo Nxt
        End If
 
Nxt:
Next cell
 
'''These lines will sort the updated Master List

Set MasterList = Sheets("MasterList").Range(Cells(2, "A"), Cells(Rows.Count, "A").End(xlUp))
 
 With ThisWorkbook.Worksheets("MasterList").Sort
        .SetRange MasterList
        .Header = xlNo
        .MatchCase = False
        .Orientation = xlTopToBottom
        .SortMethod = xlPinYin
        .Apply
    End With
    Range("A1").Select

Application.ScreenUpdating = True
MsgBox "Master List have been Updated with" & vbNewLine & Chr(9) & Cnt & " New Data Points"
 
Stp:
End Sub

Hi,

 

Yes the data layout was fooling my formula. Try these each array entered and dragged down do 8 cells

 

This ARRAY formula returns the 8 smallest values in the last 20 rows.

 

=SMALL((OFFSET($AC$1,MATCH(1E+100,AC:AC),):OFFSET($AC$1,MATCH(1E+100,AC:AC)-20,)),ROW(A1))

 

This one returns the smallest 8 items from the last 20 rows that contain data and I think this is the one you want.

 

=(SMALL((OFFSET($AC$1,MATCH(1E+100,AC:AC),):OFFSET($AC$1,LARGE(INDEX(ISNUMBER(AC:AC)*ROW(AC:AC),0),20)-1,)),ROW(A1)))
