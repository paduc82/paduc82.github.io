kergga say:

This code may help give you a starting point, although you'll need to edit it to get some of your desired functionality. I used this to scrape emails for one of my projects where I had to grab a comma-delimited report from email text.

 

This code brings all emails from a selected folder into a new Excel workbook, but it shouldn't be hard to use an existing workbook and find the last used row (that is code that is pretty common in forum answers)

 

 

Sub ParseEmailFolderToExcel()

    Set objApp = Application
    Set olns = Outlook.GetNamespace("MAPI")
    Set myInbox = olns.PickFolder
    'Set myItems = myinbox.Items
    Dim XLApp As Excel.Application
    Dim wkb As Excel.Workbook
    Dim wks As Excel.Worksheet
   
    Dim ExcelWasNotRunning As Boolean
   
    On Error Resume Next
    Set XLApp = GetObject(, "Excel.Application")
   
    If Err Then
       ExcelWasNotRunning = True
       Set XLApp = New Excel.Application
       XLApp.Visible = True
    End If

    Set wkb = XLApp.Workbooks.Add
    Set wks = wkb.Sheets(1)
    With wks
        StartCount = 0
        strEmailContents = ""
        For Each outlookmessage In myInbox.Items
            StartCount = StartCount + 1
            .Range("A" & StartCount).Value = outlookmessage.ReceivedTime
            .Range("B" & StartCount).Value = outlookmessage.SenderName
            .Range("C" & StartCount).Value = outlookmessage.Subject
            .Range("D" & StartCount).Value = outlookmessage.Subject
           
            UseCol = 4 
            
            FullMsg = outlookmessage.Body
            AllLines = Split(FullMsg, vbCrLf)
           
            For FullLine = LBound(AllLines) To UBound(AllLines)
                On Error Resume Next
                eachVal = Split(AllLines(FullLine), ",")
                For EachDataPoint = LBound(eachVal) To UBound(eachVal)
                    UseCol = UseCol + 1
                    '.cells(row,column)
                    .Cells(StartCount, UseCol).Value = eachVal(EachDataPoint)
                Next
            Next
            On Error GoTo 0
        Next
    End With
   
    Set myOlApp = Nothing
    Set olns = Nothing
    Set myInbox = Nothing
    Set myItems = Nothing

End Sub

Option Explicit


Private Declare Function BringWindowToTop Lib "user32" (ByVal HWnd As Long) As Long
Private Declare Function SetFocus Lib "user32" (ByVal HWnd As Long) As Long


Sub LogIt()
' Set reference to "Microsoft Excel xx.0 Object Library"
' via Outlook VBE > Tools > References...
'
    Dim XL As Excel.Application
    Dim xlWB As Excel.Workbook
    Dim xlWS As Excel.Worksheet
    Dim xlRng As Excel.Range
   
    Dim strPath As String, strFileName As String, strText As String
    Dim bWeStartedExcel As Boolean
    Dim objMsg As Object, objMsgText As Object, objSel As Object
    Dim mbxResp As VbMsgBoxResult
    Dim Res As Variant
   
    strPath = "C:/temp/"            'Include trailing "/"
    strFileName = "My Log.xlsx"     'Inlcude extension
   
     'Check a message is open and active:
    If TypeName(Application.ActiveWindow) <> "Inspector" Then 'an item is not open and active
        MsgBox "Please open a message and select text for exporting to Excel.", vbCritical + vbOKOnly, "Decision Logger"
        GoTo ExitPoint
    Else
        Set objMsg = Application.ActiveInspector.CurrentItem
    End If
   
     'Get currently selected text:
    Set objMsgText = Application.ActiveInspector.WordEditor
    On Error GoTo 0
    If objMsgText Is Nothing Then
        MsgBox "Please select text to export to Excel", vbCritical + vbOKOnly, "Decision Logger"
        GoTo ExitPoint
    End If
   
    On Error GoTo ExitPoint
   
    Set objSel = objMsgText.Windows(1).Selection
    strText = objSel.Text
   
     'Strip any trailing carriage returns:
    Do While Right(strText, 1) = vbCr And Len(strText) > 0
        strText = Left(strText, Len(strText) - 1)
    Loop
    If Len(strText) = 0 Then
        MsgBox "Please select text to export to Excel", vbCritical + vbOKOnly, "Decision Logger"
        GoTo ExitPoint
    End If
   
     'Look for existing instance of Excel, create new one if required:
    On Error Resume Next
    Set XL = GetObject(, "Excel.Application")
    On Error GoTo 0
    If XL Is Nothing Then
        Set XL = CreateObject("Excel.Application")
        bWeStartedExcel = True
    End If
   
     'Open the Log file:
    XL.DisplayAlerts = False
    Set xlWB = XL.Workbooks.Open(strPath & strFileName) 'will reopen without saving if already open!
    Set xlWS = xlWB.Worksheets(1)
    Set xlRng = xlWS.Cells(xlWS.Rows.Count, "A").End(xlUp).Offset(1, 0)
   
     'Insert the data:
    xlRng.Value = objMsg.ReceivedTime   'date/time message was received into col A.
    xlRng.Offset(0, 1).Value = strText  'selected text into col B.
'    xlRng.Offset(0, 2).Value = Now()    'time of exporting into col C.
   
     'Confirmation message:
    mbxResp = MsgBox("The following text:" & vbCr & vbCr _
         & "'" & strText & "'" & vbCr _
         & vbCr _
         & "has been added to row " & xlRng.Row & " in '" & strPath & strFileName & "'" & vbCr _
         & vbCr _
         & "Should the Excel log file be saved and closed?" & vbCr _
         & vbCr _
         & " - click 'Yes' to save and close the Excel log file." & vbCr _
         & " - click 'No' to display the unsaved Excel log file." & vbCr _
         & " - click 'Cancel' close the Excel log file without saving." _
         , vbYesNoCancel + vbQuestion, "Decision Logger")
        
    If mbxResp = vbYes Then
        xlWB.Close SaveChanges:=True
        If bWeStartedExcel Then XL.Quit
    ElseIf mbxResp = vbNo Then
        Res = BringWindowToTop(HWnd:=XL.HWnd)  'See http://www.cpearson.com/excel/ActivateExcelMain.aspx
        If Res <> 0 Then SetFocus HWnd:=XL.HWnd
        XL.Visible = True
        XL.DisplayAlerts = True
        xlRng.Offset(0, 1).Select
    Else 'mbxResp = Cancel
        xlWB.Close SaveChanges:=False
        If bWeStartedExcel Then XL.Quit
    End If
   
ExitPoint:
    Set xlRng = Nothing
    Set xlWS = Nothing
    Set xlWB = Nothing
    Set XL = Nothing
End Sub

Hi,

Glad it works.

The lines of code you need to look at are:

 

    Set xlRng = xlWS.Cells(xlWS.Rows.Count, "A").End(xlUp).Offset(1, 0)

   

     'Insert the data:

    xlRng.Value = objMsg.ReceivedTime   'date/time message was received into col A.

    xlRng.Offset(0, 1).Value = strText  'selected text into col B.

 

In that first row above, we create a range object for the next empty cell in column A.  You could change that to be column C to better suit your set-up.

 

Then in the last two lines of code, we populate first that cell (rng) with the ReceivedTime of the e-mail, so if you changed the first line to C, then this will go into column C.

 

The last of these three lines adds the text (decision) and from your reply, that still needs to go into col B.  Here we are using the Offset function to access the cell adjacent to the rng cell.  So if rng is now col C, you would need to change this to:

 

    Set xlRng = xlWS.Cells(xlWS.Rows.Count, "C").End(xlUp).Offset(1, 0)

   

     'Insert the data:

    xlRng.Value = objMsg.ReceivedTime   'date/time message was received into col C.

    xlRng.Offset(0, -1).Value = strText  'selected text into col B.

 

(i.e. -1 instead of 1 for the column offset).

Cheers
Rich

 

PS.  Note that further down in the code, the line:

 

    xlRng.Offset(0, 1).Select

 

selects the latest Decision cell if you click No in the confirmation msgbox.  You should probably change that to

    xlRng.Offset(0, -1).Select

as well.

Hi Daniel,

According to the information you provided, I assume that you are sharing a list with a new external user who hasn't access the site/list before. Based on my further research and test, it shows that this is a by-design behavior in SharePoint Online, even all external sharing settings are set to enabled.

This issue occurs because, when you first share a resource with a new external user, you must share either a site or a document. The initial invitation is required for the user to access the site, and this invitation is supported only when you try to share a document or site.

For a list and a library, you can't directly share it with a new external user as it is not supported. The suggested way is do this to:

1. Share an item in the list to the new external user you want to invite.

2. Let the external user access the file once.

3. Share the list with this external user.

Hi Mike,

 

Kindly keep in mind that Microsoft Word may not automatically capitalize the first letter of the first sentence of the paragraph. This occur when the first sentence of a paragraph follows any of the following:

 

  • A form field

 

  • A hyperlink

 

  • A number (or letter) that is followed by punctuation (for example, a period). This occurs whether you type the number or use the Numbering feature in Microsoft Word.

Take note that this will occur even when the Capitalize first letter of sentences option is selected.
