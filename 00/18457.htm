I assume that
sheet1 is Valdese name in row1 in cell A1

headings are in row2 in Sh1, Sh2,Sh3,....
and data starts from row3 up to .....

Now,

into the Master Sheet.(MASTER SUMMARY)

if so, try this code

'16 Dec 2011
Const ShName As String = "MASTER SUMMARY"
Dim myWS As Worksheet
Dim r As Long, N As Long
Set myWS = Sheets(ShName)
Application.ScreenUpdating = False
myWS.[A1]="MILL"
myWS.[B1]="#  of PO's Received"
myWS.[C1]="Total PO Amount"
myWS.[D1]="Total Amount Received"
myWS.[E1]="Average % Receipt"
myWS.[F1]="Total Difference"
N = 2
For Each WS In Worksheets
If WS.Name <> myWS.Name Then
r = WS.Cells(Rows.Count, "A").End(xlUp).Row
WS.[A1].Copy
myWS.Cells(N, 1).PasteSpecial xlValues
WS.Range(WS.Cells(r, 1), WS.Cells(r, 5)).Copy
myWS.Cells(N, 2).PasteSpecial xlValues
N = N + 1
End If
Next WS
Application.CutCopyMode = False
myWS.Columns("A:F").AutoFit
Application.ScreenUpdating = True
