Public zOtherMacros As String
Public Const zRange = "Developer_Table"
Public Const New_ToolBar_Name = "Critex_Macro_Addins"
Public Const zWorkSheet = "Developer"

Sub Auto_Open()
'zOtherMacros holds the name of the file containing the code being run - used to ensure that the correct piece of code is being accessed
    zOtherMacros = ThisWorkbook.Name
' Run the code to add items to the Excel ribbon
    Call AddMenuBar
End Sub

Sub AddMenuBar()
    Dim nStart As Integer, nEnd As Integer, n As Integer, nCount As Integer
    Dim lExist As Boolean, l As Boolean
    Dim myBar As CommandBar
    Dim myControl As CommandBarControl
   
' Adds the menu bar with icons contained in the master file
' zWorkSheet is the name of the sheet containing the list of icons

' nStart is the first row of the range containing the list of icons to be added
    nStart = Workbooks(zOtherMacros).Worksheets(zWorkSheet).Range(zRange).Row
' nEnd is the last row in the range containing the list of icons to be added
    nEnd = nStart + Workbooks(zOtherMacros).Worksheets(zWorkSheet).Range(zRange).Rows.Count - 1

' lExist determines whether the Command Bar exists
    lExist = True
' Set error trapping to ignore attempts to delete a menu bar that doesn't exist
    On Error GoTo NotExist
' Determine whether the menu bar already exists
    l = Application.CommandBars(New_ToolBar_Name).Enabled
' Reset error trapping to Excel defaults
    On Error GoTo 0

' If the menu bar does not exist
    If lExist = False Then
' Add new menu bar
' Set variable myBar to hold the details of the menu bar added
        Set myBar = CommandBars _
            .Add(Name:=New_ToolBar_Name, Position:=msoBarTop)
' Set the new menu bar to be visible
        myBar.Visible = True
    End If
   
' For each icon listed in the range add the relevant icon and macro details
' nCount is a count variable to loop through the range of icons
    For nCount = nStart To nEnd
' Delete the button if it already exists
        Set myControl = Application.CommandBars(New_ToolBar_Name).FindControl(, , Workbooks(zOtherMacros).Worksheets(zWorkSheet).Cells(nCount, 1).Value)
' lExist determines whether the current item already exists
        lExist = True
' Set error trapping to ignore attempts to select the current control being added/amended
        On Error GoTo NotExist
' Select the icon ID of the control
        n = myControl.FaceId
' Reset error trapping to Excel defaults
        On Error GoTo 0
' Select the new menu bar and add the next icon
        With Application.CommandBars(New_ToolBar_Name)
            If lExist = False Then
' Add a button to the start (left end) of the menu bar
                With .Controls.Add(Type:=msoControlButton, Temporary:=True, Before:=1)
' Make the button type an icon button
                    .Style = msoButtonIcon
' Add the macro details to be run when the icon is clicked
                    .OnAction = Workbooks(zOtherMacros).Worksheets(zWorkSheet).Cells(nCount, 3).Value
' Add the tool tip text
                    .TooltipText = Workbooks(zOtherMacros).Worksheets(zWorkSheet).Cells(nCount, 5).Value
' Add the reference tag
                    .Tag = Workbooks(zOtherMacros).Worksheets(zWorkSheet).Cells(nCount, 1).Value
' Amend the image
                    .FaceId = Workbooks(zOtherMacros).Worksheets(zWorkSheet).Cells(nCount, 6).Value
                End With
            Else
                With myControl
' Make the button type an icon button
                    .Style = msoButtonIcon
' Add the macro details to be run when the icon is clicked
                    .OnAction = Workbooks(zOtherMacros).Worksheets(zWorkSheet).Cells(nCount, 3).Value
' Add the tool tip text
                    .TooltipText = Workbooks(zOtherMacros).Worksheets(zWorkSheet).Cells(nCount, 5).Value
' Amend the image
                    .FaceId = Workbooks(zOtherMacros).Worksheets(zWorkSheet).Cells(nCount, 6).Value
                End With
            End If
        End With
    Next

    Exit Sub
   
' Resets lExist to identify whether the command bar or control can be access - i.e. if it exists
NotExist:
lExist = False
Resume Next

End Sub
