norman david jones say:

Hi Tenison,

  • Select the sheet of interest
  • Right-click the sheet tab
  • Select the View Code option from the resultant context menu
  • Paste the following event code

'=========>>
Option Explicit
Const myPhrase As String = "Date Required"
Dim bDateRequired As Boolean
Dim rDate As Range

'--------->>
Private Sub Worksheet_Change(ByVal Target As Range)
    Dim Rng As Range, Rng2 As Range

    Set Rng = Target.Cells(1)

    With Rng
        If UCase(.Value) = UCase(myPhrase) Then
            On Error GoTo XIT
            With .Offset(0, 1)
                bDateRequired = Not IsDate(.Value)
                If Not bDateRequired Then
                    Set rDate = Nothing
                Else
                    Application.EnableEvents = False
                    Set rDate = .Item(1)
                    .Select

                End If
            End With
        ElseIf Not rDate Is Nothing Then
            If .Address = rDate.Address Then
                If IsDate(.Value) Then
                    Set rDate = Nothing
                End If
            End If
        ElseIf .Column > 1 Then
            If .Offset(0, -1).Value = myPhrase And Not IsDate(.Value) Then
                .Select
                Set rDate = .Item(1)
                Call MsgBox(Prompt:="A date entry is required in cell " _
                                  & .Address(0, 0), _
                            Buttons:=vbInformation, _
                            Title:="Date Entry")
            End If
        End If
End With
XIT:
Application.EnableEvents = True
End Sub

'--------->>
Private Sub Worksheet_SelectionChange(ByVal Target As Range)
    Dim Rng As Range

    Set Rng = Target

    If Not rDate Is Nothing Then
        With rDate
            On Error GoTo XIT
            Application.EnableEvents = False
            .Select
            Call MsgBox(Prompt:="A date entry is required in cell " _
                              & .Address(0, 0), _
                        Buttons:=vbInformation, _
                        Title:="Date Entry")
        End With
    End If
XIT:
    Application.EnableEvents = True
End Sub
'<<=========

Here is the code to protect all sheets

In the code below we are:

  • Turning off screen updating to stop flickering as we protect and unprotect each sheet.
  • Using the For Each loop
  • Protecting each worksheet with the password â€œOnlineâ€
  • Screen updating is turned back on (even though this is not necessary as it will happen by default)

Sub Protect_All()
    Dim ws As Worksheet
    'turn screen updating off
    Application.ScreenUpdating = False
    'loop through each sheet and protect with a password
    For Each ws In ActiveWorkbook.Worksheets
        ws.Protect Password:="Online", DrawingObjects:=True, Contents:=True, _
        Scenarios:=True, AllowFormattingCells:=True, AllowFormattingColumns:=True, _
        AllowFormattingRows:=True, AllowSorting:=True, UserInterfaceOnly:=True
    Next ws
    'turn screen updating on
    Application.ScreenUpdating = True
End Sub

Sub Unprotect_All()
    Dim ws As Worksheet
    'remove error handling
    On Error Resume Next
    'turn screen updating off
    Application.ScreenUpdating = False
    'loop through sheets and unprotect with a password
    For Each ws In ActiveWorkbook.Worksheets
        ws.Unprotect Password:="Online"
    Next ws
    'turn screen updating on
    Application.ScreenUpdating = True
    'reset error handler
    On Error GoTo 0
End Sub

I understand that before, when creating a team in Microsoft Teams, the group and group calendar will be automatically added to the OWA and Outlook client. However, the feature is changed now. When creating a team in Microsoft Teams, an Office 365 group will be created. But it will be hidden from the Global Address List. 3. Go to OWA or Outlook client, join the group manually and the group calendar will display now.
