Function MultiLookup(val As Range, rng As Range) As String
Dim c As Range
For Each c In rng
    If c.Value = val.Value Then
        MultiLookup = MultiLookup + c.Offset(, 1) + ", "
        End If
Next
If MultiLookup = "" Then
    MultiLookup = "No Match"
Else
MultiLookup = Left(MultiLookup, Len(MultiLookup) - 2)
End If
End Function
