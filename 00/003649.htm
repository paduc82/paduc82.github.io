graham mayor say:
The best approach is to set a range where you want the footer text to appear, which you cannot do with the macro recorder. Bearing in mind what Charles said about the different footer ranges you could use something like the following which I have annotated to show how it works.

Sub AddFNametoFooter()
Dim oRng As Range
Dim iRM As Long
ActiveDocument.Save 'save the document so that it has a filename
With Selection.Sections(1) 'address the section containing the cursor
    With .PageSetup 'determine where the right margin is located
        iRM = .PageWidth - .RightMargin - .LeftMargin
        .DifferentFirstPageHeaderFooter = False
    End With
    'Set a range to reference the required footer - here the primary footer of the current section
    Set oRng = .Footers(wdHeaderFooterPrimary).Range
    'Set the font size of the footer
    oRng.Font.Size = 10
    'add a tab
    oRng.Text = vbTab
    'Collapse the range to the end
    oRng.Collapse wdCollapseEnd
    'Add a filename field at the end of the footer
    ActiveDocument.Fields.Add oRng, wdFieldFileName, "\p", False
    'Clear the tab stops and add a new right aligned tab at the right margin
    With oRng.Paragraphs(1).TabStops
        .ClearAll
        .Add Position:=iRM, Alignment:=wdAlignTabRight
    End With
    'Update the fields
    .Footers(wdHeaderFooterPrimary).Range.Fields.Update
End With
End Sub
