Private Declare Function GetSystemMenu _
    Lib "user32" ( _
    ByVal hWnd As Long, _
    ByVal bRevert As Long) _

    Lib "user32" ( _

Private Sub UserForm_Initialize()
Dim lFrmHdl As Long, iCount As Integer
With Me
    .Width = Application.Width
    .Height = Application.Height
End With
lFrmHdl = FindWindowA(vbNullString, Me.Caption)
If lFrmHdl <> 0 Then
    For iCount = 0 To 1
        RemoveMenu GetSystemMenu(lFrmHdl, False), 0, MF_BYPOSITION
    Next iCount
End If

End Sub

doug robbins - mvp office apps & services (word) say:

To have

inserted as a single paragraph at the position of the cursor, with present tab stops being the default of 0.5", use

Dim txt_test As String
txt_test = "First Name:" & vbTab & "Amr" _
& Chr(11) & "Last Name:" & vbTab & "El-Bendary" _
& Chr(11) & "Live:" & vbTab & "XXXXXX" _
& Chr(11) & "XXXXXXX:" & vbTab & "XXXXXX"
Selection.Text = txt_test
Selection.Paragraphs(1).TabStops.Add InchesToPoints(1)

If you want individual paragraphs, then use:

Dim txt_test As String
txt_test = "First Name:" & vbTab & "Amr" _
& Chr(13) & Chr(10) & "Last Name:" & vbTab & "El-Bendary" _
& Chr(13) & Chr(10) & "Live:" & vbTab & "XXXXXX" _
& Chr(13) & Chr(10) & "XXXXXXX:" & vbTab & "XXXXXX"
Selection.Paragraphs(1).TabStops.Add InchesToPoints(1)
Selection.Text = txt_test

Note however that the Chr(10) is redundant and can be deleted and for consistency with the use of vbTab, use vbCr in place of Chr(13) & Chr(10)
