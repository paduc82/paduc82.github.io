marshall barton say:

No, they will not automatically sync.  Think about what you would have to do if the subform and the popup needed to display different records.

The simplest/best way to do what you want is to use the OpenForm method/action's WhereCondition argument to filter the popup to the desired note according to the current record in the subform.  Find the line of code that is used to open the popup form and add the argument to the OpenForm line:

   DoCmd.OpenForm "popup form name",WhereCondition:= "[notes key field] = " & Me.[foreign key field for the project detail note field]

If you are using a macro action instead of VBA code in the Click event, set the WhereCondition argument to the same expression.

    Chris,

    I wonder if you might be better served with a macro.

    Assuming your cells with the whole or partial sentences are in column
    A, the following macro will process the column and put the results
    into Column B.

    It uses your rule of "A new sentence always starts with a capital
    letter" and will put each full sentence into a cell in column B.

    It will put the first full sentence into B1; the second into B2; and
    so forth.  This can be changed, and the results could even be placed
    on a different worksheet.

    It does not matter how many cells make up the sentence.  It can be 1,
    3, 10 or 100.

    To enter this Macro (Sub), <alt-F11> opens the Visual Basic
    Editor. Ensure your project is highlighted in the Project
    Explorer window. Then, from the top menu, select
    Insert/Module and paste the code below into the window that
    opens. To use this Macro (Sub), <alt-F8> opens the macro dialog
    box. Select the macro by name, and <RUN>.

    ===============================
    Option Explicit
    Sub CombineSentences()
        Dim rSrc As Range, rDest As Range
        Dim i As Long, j As Long
        Dim s As String

    Set rDest = Range("B1")
        rDest.EntireColumn.ClearContents
    Set rSrc = Range("A1", Cells(Cells.Rows.Count, "A").End(xlUp))

    i = 1
    j = 1
    Do Until i > rSrc.Count
        s = rSrc(i).Text
        Do Until UCase(Left(rSrc(i + 1).Text, 1)) _
                = Left(rSrc(i + 1).Text, 1)
            s = s & " " & rSrc(i + 1).Text
            i = i + 1
        Loop
        rDest(j).Value = WorksheetFunction.Trim(s)
        If Len(s) Then j = j + 1
        s = ""
        i = i + 1
    Loop
    End Sub

    Chris,

    As written, the macro also assumes that if there is a blank line
    (empty cell), then the preceding cell is the end of the sentence, even
    if the cell after the empty cell starts with a lower case character.

    If that is not the case, one way of handling this would be to first
    delete all the blank cells.

    This modification will handle that if it is an issue:

    ==========================
    Option Explicit
    Sub CombineSentences()
        Dim rSrc As Range, rDest As Range
        Dim i As Long, j As Long
        Dim s As String

    Set rSrc = Range("A1", Cells(Cells.Rows.Count, "A").End(xlUp))

        For i = rSrc.Count To 1 Step -1
            If rSrc(i).Value = "" Then _
                rSrc(i).Delete (xlShiftUp)
        Next i

    Set rSrc = Range("A1", Cells(Cells.Rows.Count, "A").End(xlUp))
    Set rDest = Range("B1")
        rDest.EntireColumn.ClearContents

    i = 1
    j = 1
    Do Until i > rSrc.Count
        s = rSrc(i).Text
        Do Until UCase(Left(rSrc(i + 1).Text, 1)) _
                = Left(rSrc(i + 1).Text, 1)
            s = s & " " & rSrc(i + 1).Text
            i = i + 1
        Loop
        rDest(j).Value = WorksheetFunction.Trim(s)
        If Len(s) Then j = j + 1
        s = ""
        i = i + 1
    Loop
    End Sub
