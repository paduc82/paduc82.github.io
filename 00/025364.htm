lars-Åke aspelin say:

Assuming that you don't have anything in the column to the right of your list, try the following macro:

Sub remove_duplicates()
  firstrow = 1
  lastrow = Cells(1, "A").End(xlDown).Row
  For i = firstrow To lastrow
    Cells(i, "C") = Cells(i, "A") & Cells(i, "B")
  Next i
  For i = firstrow To lastrow
    For j = i + 1 To lastrow
      If Cells(j, "C") = Cells(i, "C") Then
         Cells(j, "C").EntireRow.Delete
         j = j - 1
         lastrow = lastrow - 1
      End If
    Next j
  Next i
  Cells(1, 3).EntireColumn.Delete
End Sub
