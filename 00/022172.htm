Private Function RemovePrefix(strPhone As String) As String

strPhone = Trim(strPhone)
RemovePrefix = strPhone
If strPhone = "" Then Exit Function
Dim prefix As String
prefix = Left(strPhone, 1)

Do While (prefix = "+" Or prefix = "1")
    strPhone = Mid(strPhone, 2)
    prefix = Left(strPhone, 1)
Loop

RemovePrefix = strPhone

End Function

Private Function RemoveChars(strPhone As String) As String

strPhone = Trim(strPhone)
RemoveChars = strPhone
If strPhone = "" Then Exit Function
strPhone = Replace(strPhone, "(", "")
strPhone = Replace(strPhone, ")", "")
strPhone = Replace(strPhone, ".", "-")
strPhone = Replace(strPhone, " ", "-")
RemoveChars = strPhone

End Function

Sub HighlightWholeWordsOnly()
    'this macro sets the format of sections of strings in cells
    'that are whole words only - eg, "Note" but not "Denotes
    Dim rngC As Range
    Dim rngF As Range
    Dim intStart As Integer
    Dim intLen As Integer
    Dim strFind As String
    Dim strFirstAddress As String
    Dim boolHighLight As Boolean

    strFind = InputBox("Enter the word to make Red", , "note")
    intLen = Len(strFind)

    'find the instances

    With ActiveSheet.Cells

        Set rngC = .Find(strFind, LookIn:=xlValues, lookAt:=xlPart)

        If Not rngC Is Nothing Then
            Set rngF = rngC
            strFirstAddress = rngC.Address
        Else:
            MsgBox "Not Found"
            End
        End If

        Set rngC = .FindNext(rngC)
        If Not rngC Is Nothing And rngC.Address <> strFirstAddress Then
            Do
                Set rngF = Union(rngF, rngC)
                Set rngC = .FindNext(rngC)
            Loop While Not rngC Is Nothing And rngC.Address <> strFirstAddress
        End If
    End With

    'Change word to Red
    For Each rngC In rngF
        intStart = InStr(1, UCase(rngC.Value), UCase(strFind))
        While intStart > 0
            boolHighLight = False
            If InStr(1, " ,.:;'""", Mid(rngC.Value, intLen + intStart, 1)) > 0 Then
                boolHighLight = True
            End If
            If intStart <> 1 Then
                If Mid(rngC.Value, intStart - 1, 1) <> " " Then
                    boolHighLight = False
                End If
            End If
            If boolHighLight Then
                With rngC.Characters(Start:=intStart, Length:=intLen).Font
                    .ColorIndex = 3
                End With
            End If
            intStart = InStr(intStart + 1, UCase(rngC.Value), UCase(strFind))
        Wend
    Next rngC

End Sub

steve rindsberg say:

>> Is there something I'm missing?

I don't think so.  It looks as though creating a grayscale PDF isn't in PowerPoint's bag of tricks.

>> PPS  Does anyone know why the macro recorder was nixed in this ppt?  It was tremendously useful for scouting VBA command

The application and its features had gotten so far ahead of the recorder's ability to record them that it no longer was able to create even marginally usable macros. Some of us felt (as I expect you do) that it would have been worth leaving it in place, even in an incomplete state, because it made it easier to work out how to do at least SOME things.  But I imagine the tech support for it got to be overwhelming, so they chose to remove the recorder. 

One thing that can help:  In the VB editor, press F2 and learn your way around the object browser that appears.  It at least gives you a list of commands and properties you can use, even if it doesn't offer much help in how to use them. 

I had a look at the properties supported by the PDF export feature, both in the Object Browser and in the user interface;  I can't find any mention of grayscale or b/w options in either place, so I'm betting it's not supported.

Since it looks like you have Acrobat, here's something to try:

    With ActivePresentation.PrintOptions
        .PrintColorType = ppPrintBlackAndWhite
        .ActivePrinter = "Adobe PDF"
        ' set other options here
    End With
    
    ActivePresentation.PrintOut PrintToFile:="C:\Temp\Test.PS"

You may need to mess about with the Distiller settings to get this to work as expected.  Once you've created the PS file, you can doubleclick it to launch Distiller and have it converted to a same-named PDF.  It's been a while since I messed with this but ISTR that you can run Distiller in a watched folder mode that automatically converts any PS files that appear in the watched folder.  If you can work that out, it'll save some steps.
