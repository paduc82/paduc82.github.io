jllatham say:

I think your FOR..NEXT and IF...END IF statements were a bit confused even though they weren't causing a debug or even a direct runtime error.   I've rewritten the whole process as I think you meant it to be and it has worked for me.  Here's my revision of it.

 

Sub Adding_Row_Formulas()
  Dim newName As String
  Dim ws As Worksheet
  For Each ws In ActiveWorkbook.Worksheets
    If ws.Name <> "Data" Then
      ws.Activate
      newName = ws.Name

      Range("A1").Select
      ws.ListObjects.Add(xlSrcRange, _
       Range("A1").CurrentRegion, , xlYes).Name = newName
      Range(newName & "[#All]").Select
      ActiveSheet.ListObjects(newName).TableStyle = _
       "TableStyleLight13"
      ActiveWindow.DisplayGridlines = False
     
      Columns("W:W").EntireColumn.AutoFit
      Columns("T:T").EntireColumn.AutoFit
     
      Rows("1:2").Select
      Selection.Insert Shift:=xlDown, _
       CopyOrigin:=xlFormatFromLeftOrAbove
      Rows("2:2").Select
      Selection.RowHeight = 50
      Rows("2:2").Select
      With Selection
        .HorizontalAlignment = xlGeneral
        .VerticalAlignment = xlCenter
        .WrapText = False
        .Orientation = 0
        .AddIndent = False
        .IndentLevel = 0
        .ShrinkToFit = False
        .ReadingOrder = xlContext
        .MergeCells = False
      End With
      Range("A2").Select
      ActiveCell.FormulaR1C1 = "Saving"
      Range("B1").Select
      ActiveCell.FormulaR1C1 = "EUR"
      Range("C1").Select
      ActiveCell.FormulaR1C1 = "GBP"
      Range("D1").Select
      ActiveCell.FormulaR1C1 = "CHF"
      Range("E1").Select
      ActiveCell.FormulaR1C1 = "*  "
      Range("F1").Select
      ActiveCell.FormulaR1C1 = "Number of bookings"
      Range("B2").Select
      ActiveCell.FormulaR1C1 = "=SUMIFS(" & newName & "[DIFF]," & _
       newName & "[CUR],R[-1]C)"
      Range("C2").Select
      ActiveCell.FormulaR1C1 = "=SUMIFS(" & newName & "[DIFF]," & _
       newName & "[CUR],R[-1]C)"
      Range("D2").Select
      ActiveCell.FormulaR1C1 = "=SUMIFS(" & newName & "[DIFF]," & _
       newName & "[CUR],R[-1]C)"
      Range("E2").Select
      ActiveCell.FormulaR1C1 = "=SUMIFS(" & newName & "[DIFF]," & _
       newName & "[CUR],R[-1]C)"
      Range("F2").Select
      ActiveCell.FormulaR1C1 = "=COUNTA(" & newName & "[PIC])"
      Range("F2").Select
      On Error Resume Next
    End If
  Next
  Sheets("Data").Select
End Sub

michael bauer (vboffice.net) say:

Scott, I think you just need to give it some time. This VB6 example uses a form with a timer control on it. With my hosted Exchange it takes about three intervalls, or 9 seconds, to get new items printed in the Timer event:

 

Option Explicit

 

Private OL As Outlook.Application

Private Ns As Outlook.NameSpace

Private Inbox As Outlook.Items

 

Private Sub Form_Load()

  Set OL = New Outlook.Application

  Set Ns = OL.GetNamespace("mapi")

  Set Inbox = Ns.GetDefaultFolder(olFolderInbox).Items

  Timer.Interval = 3000

  Timer.Enabled = True

End Sub

 

Private Sub Form_QueryUnload(Cancel As Integer, UnloadMode As Integer)

  Timer.Enabled = False

  Set Inbox = Nothing

  Set Ns = Nothing

  Set OL = Nothing

End Sub

 

Private Sub Timer_Timer()

  On Error Resume Next

  Dim Item As Object

  Dim i&

  Static Counter&

 

  Counter = Counter + 1

  Debug.Print Counter & ". Intervall"

  If Inbox.Count Then

    For Each Item In Inbox

      i = i + 1

      Debug.Print i & " - " & Item.Subject

    Next

    Debug.Print "---"

  End If

  Err.Clear

End Sub
