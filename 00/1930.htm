Private Sub ComboBox1_Change()
  'Fill the Listbox and restore selected item
  Dim j As Integer, Key As String
  ListBox1.Clear
  If ComboBox1.ListIndex < 0 Then Exit Sub
  For j = 1 To UBound(Data, 2)
    ListBox1.AddItem Data(ComboBox1.ListIndex + 1, j)
  Next
  '
  ' ######## here is the interesting part: #########
  '
  'Get current selected item
  Key = ComboBox1.List(ComboBox1.ListIndex)
  If SaveListIndex.Exists(Key) Then
    'Restore selected item in ListBox
    ListBox1.ListIndex = SaveListIndex.Item(Key)
  End If
End Sub

Private Sub CommandButton1_Click()
  'Prev item
  If ComboBox1.ListIndex <= 0 Then
    ComboBox1.ListIndex = ComboBox1.ListCount - 1
  Else
    ComboBox1.ListIndex = ComboBox1.ListIndex - 1
  End If
End Sub

Private Sub CommandButton2_Click()
  'Next item
  If ComboBox1.ListIndex = ComboBox1.ListCount - 1 Then
    ComboBox1.ListIndex = 0
  Else
    ComboBox1.ListIndex = ComboBox1.ListIndex + 1
  End If
End Sub

Private Sub ListBox1_Click()
  'Save selected item
  Dim Key As String
  '
  ' ######## here is the interesting part: #########
  '
  Key = ComboBox1.List(ComboBox1.ListIndex)
  If SaveListIndex.Exists(Key) Then
    'Update
    SaveListIndex.Item(Key) = ListBox1.ListIndex
  Else
    'Add it
    SaveListIndex.Add Key, ListBox1.ListIndex
  End If
End Sub

  ComboBox1.AddItem "Whatever"

  ReDim Data(1 To ComboBox1.ListCount, 1 To MaxListBox)

      Data(i, j) = ComboBox1.List(i - 1) & j
    Next
  Next

hongcc1 say:

If you know Macro, you can consider converting the manual task to Macro.

Referring to this link: http://www.msofficeforums.com/word/6888-how-have-two-formats-caption-style-label.html

The macro is as follow:

Sub CaptionBold()
Application.ScreenUpdating = False
With ActiveDocument.Range
  With .Find
    .ClearFormatting
    .Text = ""
    .Style = "Caption"
    .Replacement.Text = ""
    .Forward = True
    .Wrap = wdFindStop
    .Format = True
    .MatchCase = False
    .MatchWholeWord = False
    .MatchWildcards = False
    .MatchSoundsLike = False
    .MatchAllWordForms = False
    .Execute
  End With
  Do While .Find.Found
    With .Paragraphs.Last.Range.Duplicate
      .End = .Start + Len(Split(.Text, " ")(0)) + 1
      .MoveEndUntil " ", wdForward
      .Style = "Strong"
    End With
    .Collapse wdCollapseEnd
    .Find.Execute
  Loop
End With
Application.ScreenUpdating = True
End Sub

Run the Macro to update the character style to Strong for all the captions.
