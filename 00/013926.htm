Private Sub Command1103_Click()
    On Error GoTo Error_Handler

    Dim strName               As String

                               "  Click ok for the report & Pottsville search for a search. ", _

                DoCmd.OpenForm "FRM_SearchMultiHarrisburg"
                Interaction.Beep
        End Select
    End If

Error_Handler_Exit:
    On Error Resume Next
    Exit Sub

Error_Handler:
    MsgBox "The following error has occured" & vbCrLf & vbCrLf & _
           "Error Number: " & Err.Number & vbCrLf & _

    Resume Error_Handler_Exit
