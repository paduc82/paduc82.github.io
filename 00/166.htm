Sub GiveMeHyp()
    Dim theBase as Long
    Dim theSide as Long
    Dim theHyp as Long

    theBase = InputBox("What is the base?")
    theSide = InputBox("What is the side?")
    theHyp = Sqrt((theBase*theBase)+(theSide*theSide))

    MsgBox "The hypotensue is " & theHyp
End Sub
