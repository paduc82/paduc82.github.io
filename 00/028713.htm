Function GetFilename(odoc As Document, strStyle As String) As String
Dim oPara As Paragraph
Dim oRng As Range
Dim i As Long
    For Each oPara In odoc.Paragraphs
        Set oRng = oPara.Range
        oRng.End = oRng.End - 1
        If oRng.Style = strStyle Then
            GetFilename = oRng.Text
            Exit For
        End If
    Next oPara
End Function
