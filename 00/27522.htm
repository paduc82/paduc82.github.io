tom ogilvy say:

LuckyJackPWinner,

If you mean you want to generate combinations of 5 that hold a particular number (ex: 23) and you want to show that number in a specific position, then this will do that:

(so for each combination generated, if the combination contains the number specified, it forces that number to appear in the specified condition).

Sub combinations5LocationandValue()
Dim Last As Long
Dim b As Boolean
Application.ScreenUpdating = False
Application.Calculation = xlCalculationManual
Dim g, h, i, j, k, l, ii
Dim v(1 To 5) As Long, s As String
Dim iloc As Long, ival As Long
iloc = 2
ival = 23
Last = Cells(Rows.Count, "A").End(xlUp).Row
For g = 1 To Last - 4
    For h = g + 1 To Last - 3
        For i = h + 1 To Last - 2
            For j = i + 1 To Last - 1
              For k = j + 1 To Last
                s = ""
                v(1) = Cells(g, 1)
                v(2) = Cells(h, 1)
                v(3) = Cells(i, 1)
                v(4) = Cells(j, 1)
                v(5) = Cells(k, 1)
                b = False
                For ii = 1 To 5
                  If v(ii) = ival Then
                   b = True
                   If i <> iloc Then
                    v(ii) = v(iloc)
                    v(2) = ival
                   End If
                   Exit For
                  End If
                Next ii
                If b Then
                  l = l + 1
                  For ii = 1 To 5
                    s = s & v(ii) & "-"
                  Next ii
                  Cells(l, 2) = Left(s, Len(s) - 1)
                End If
             Next k, j, i, h, g

Application.ScreenUpdating = True
Application.Calculation = xlCalculationAutomatic
 End Sub

If you mean you want to only show the the items where the number appears based on the code in a specific position that would be:

Sub combinations5specificposition()
Dim Last As Long
Application.ScreenUpdating = False
Application.Calculation = xlCalculationManual
Dim g, h, i, j, k, l
Dim b As Boolean, iloc As Long, ival As Long
iloc = 2
ival = 20
Last = Cells(Rows.Count, "A").End(xlUp).Row
For g = 1 To Last - 4
    For h = g + 1 To Last - 3
        For i = h + 1 To Last - 2
            For j = i + 1 To Last - 1
                For k = j + 1 To Last
                  b = False
                  Select Case iloc
                     Case 1: If Cells(g, 1) = ival Then b = True
                     Case 2: If Cells(h, 1) = ival Then b = True
                     Case 3: If Cells(i, 1) = ival Then b = True
                     Case 4: If Cells(j, 1) = ival Then b = True
                     Case 5: If Cells(k, 1) = ival Then b = True
                  End Select
                  If b Then
                Cells(l + 1, 2) = Cells(g, 1) & "-" & Cells(h, 1) & "-" _
                & Cells(i, 1) & "-" & Cells(j, 1) & "-" & Cells(k, 1)
                l = l + 1
                  End If
                Next k, j, i, h, g

Application.ScreenUpdating = True
Application.Calculation = xlCalculationAutomatic
 End Sub
