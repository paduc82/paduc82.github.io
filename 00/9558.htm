Sub GetRid()
    Dim lnLastRow As Long, lnFirstRow As Long, i As Long
    Dim rngToSearch As Range, rngFound As Range
    Dim ws1 As Worksheet, ws2 As Worksheet
   
    Set ws1 = Worksheets("Sheet1")
    Set ws2 = Worksheets("Sheet2")
   
    lnFirstRow = 3
    lnLastRow = ws1.Range("C" & Rows.Count).End(xlUp).Row
   
    Set rngToSearch = ws2.Range("C:C")
   
    For i = lnLastRow To lnFirstRow Step -1
        Set rngFound = Nothing
        Set rngFound = rngToSearch.Find( _
            What:=ws1.Range("C" & i).Value, _
            After:=ws2.Cells(ws2.Rows.Count, "C"), _
            LookIn:=xlValues, _
            LookAt:=xlWhole, _
            SearchOrder:=xlByRows, _
            SearchDirection:=xlNext, _
