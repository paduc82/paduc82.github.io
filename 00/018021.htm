jllatham say:

How about we use some substitutes to change the "-" to 3 different characters and work from there.  That seems an "easy" way to do it. 

The basis for our longer final formulas would be this:

=SUBSTITUTE(SUBSTITUTE(SUBSTITUTE(A1,"-","!",1),"-","#",1),"-","$",1)

which will change the 1st "-" to a "!", the second one to a "#" and the 3rd one to a $ symbol.  If it turns out you have any of those particular symbols in some of the cells, you could pick any 3 that don't appear in any of your data cells.

Now with that said, to get the 13,548LB out of it you would have this in B1

=TRIM(MID(SUBSTITUTE(SUBSTITUTE(SUBSTITUTE(A1,"-","!",1),"-","#",1),"-","$",1),FIND("!",SUBSTITUTE(SUBSTITUTE(SUBSTITUTE(A1,"-","!",1),"-","#",1),"-","$",1))+1,FIND("#",SUBSTITUTE(SUBSTITUTE(SUBSTITUTE(A1,"-","!",1),"-","#",1),"-","$",1))-FIND("!",SUBSTITUTE(SUBSTITUTE(SUBSTITUTE(A1,"-","!",1),"-","#",1),"-","$",1))-1))

and this in C1

=TRIM(MID(SUBSTITUTE(SUBSTITUTE(SUBSTITUTE(A1,"-","!",1),"-","#",1),"-","$",1),FIND("#",SUBSTITUTE(SUBSTITUTE(SUBSTITUTE(A1,"-","!",1),"-","#",1),"-","$",1))+1,FIND("$",SUBSTITUTE(SUBSTITUTE(SUBSTITUTE(A1,"-","!",1),"-","#",1),"-","$",1))-FIND("#",SUBSTITUTE(SUBSTITUTE(SUBSTITUTE(A1,"-","!",1),"-","#",1),"-","$",1))-1))

    Sub ExceptionAndSummaryReport()
      Dim Data, Report, Goals, Samples, Temp
      Dim i As Long, j As Long, k As Long

      Dim R As Range, IDs As Range, Dates As Range
      Dim MinDate As Date, MaxDate As Date, ThisDate As Date
      Dim MinId As Long, MaxId As Long, ThisId As Long
      Dim ResVal As Double

      'Step 0: Create the exception grid

      With Sheets("Input Cl2 Data")
        'Find the last used cell in column A
        Set R = .Range("A" & Rows.Count).End(xlUp)
        'Span the range to A3:C[LastRow]
        Set R = .Range("A3", .Range("C" & R.Row))
        'Find the min./max. dates and ID's
        MinDate = WorksheetFunction.Min(Intersect(R, .Columns("B")))
        MaxDate = WorksheetFunction.Max(Intersect(R, .Columns("B")))
        MinId = WorksheetFunction.Min(Intersect(R, .Columns("A")))
        MaxId = WorksheetFunction.Max(Intersect(R, .Columns("A")))
        'Read all data into an array
        Temp = R
        'Create space for the exception grid
        ReDim Data(MinDate - 1 To MaxDate, MinId - 1 To MaxId)
        'Fill the dates in the first column
        For i = MinDate To MaxDate
          Data(i, LBound(Data, 2)) = i
        Next
        'Fill the ID's in the first row
        For i = MinId To MaxId
          Data(LBound(Data), i) = i
        Next
        'Fill the grid
        For i = 1 To UBound(Temp)
          ThisId = Temp(i, 1)
          ThisDate = Temp(i, 2)
          ResVal = Temp(i, 3)
          If ResVal < 0.5 Then
            Data(ThisDate, ThisId) = 1
          ElseIf Data(ThisDate, ThisId) <> 1 Then
            Data(ThisDate, ThisId) = 0
          End If
        Next
      End With
      With Sheets("Exception Grid")
        'Erase all existing data
        .Cells.ClearContents
        'Write to the sheet
        .Range("B2").Resize(UBound(Data) - LBound(Data) + 1, _
          UBound(Data, 2) - LBound(Data, 2) + 1) = Data
      End With

      'Note: It is not necessary to write the data into a sheet!
      '  We can convert the array to 1 based if we transpose it twice.
      '  You can use this lines instead:
      'With WorksheetFunction
      '  Data = .Transpose(.Transpose(Data))
      'End With

      'Step 1: Create the exception report

      With Sheets("Exception Grid")
        'Read in all data into an array
        Data = .Range("B2").CurrentRegion
      End With
      'Make a copy of the array
      Report = Data
      k = 1
      'Visit each row, except the header
      For i = 2 To UBound(Report)
        'Does the month change in the first column?
        If Month(Report(i, 1)) <> Month(Report(i - 1, 1)) Then
          'Next result row
          k = k + 1
          'Copy the current row as result row (include the month)
          For j = 1 To UBound(Report, 2)
            Report(k, j) = Report(i, j)
          Next
        Else
          'Add the current row to the result row (exclude the month)
          For j = 2 To UBound(Report, 2)
            Report(k, j) = Report(k, j) + Report(i, j)
          Next
        End If
      Next
      'Change the date in the first column to a month name
      For i = 2 To k
        Report(i, 1) = MonthName(Month(Report(i, 1)), True)
      Next
      With Sheets("Exception Report")
        'Erase all existing data
        .Cells.ClearContents
        'Write the array to the sheet, transpose columns to rows
        .Range("B2").Resize(UBound(Report, 2), k) = WorksheetFunction.Transpose(Report)
      End With

      'Step 2: Calculate Number of Routine Test Results Not meeting Goals

      ReDim Goals(1 To k, 1 To 1)
      For i = 2 To k
        For j = 2 To UBound(Data, 2)
          Goals(i, 1) = Goals(i, 1) + Report(i, j)
        Next
      Next
      For i = 2 To k
        'Annual
        Goals(1, 1) = Goals(1, 1) + Goals(i, 1)
      Next
