Sub Sort_Sections_ByDate()
'Oct  18, 2016

Const N As Integer = 3 '<< headers in row 3

Dim v As Variant
v = Array("A - F", "G - M", "N - Z") '<< sheet names
Dim x As Long, r As Long, c As Long, j As Long
Application.ScreenUpdating = False
For x = 0 To UBound(v)
With Sheets(v(x))
c = .Cells(N, Columns.Count).End(xlToLeft).Column
For j = 1 To c Step 4
r = .Cells(Rows.Count, j).End(xlUp).Row

If r > N Then

.Range(.Cells(N, j), .Cells(r, j + 3)).Sort Key1:=.Cells(N, j), Order1:=xlAscending, Header:=xlYes, _
OrderCustom:=1, MatchCase:=False, Orientation:=xlTopToBottom

End If

Next

End With

Next
Application.ScreenUpdating = True
End Sub
