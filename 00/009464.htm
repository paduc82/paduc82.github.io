hansv mvp say:

Try this:

 

Sub FindNextOccurrence()
    Dim rst As DAO.Recordset
    Dim strChr As String
    Dim strFind As String
    strChr = "A"
    ' Set up the where-condition
    strFind = "Cust_Name Like " & Chr(34) & strChr & "*" & Chr(34)
    ' Create a copy of the form's recordset
    Set rst = Me.RecordsetClone
    ' Try to find the next match
    rst.FindNext strFind
    If rst.NoMatch Then
        ' If not found, try to find the first match
        rst.FindFirst strFind
        If rst.NoMatch Then
            ' Not found - get out
            Beep
            Exit Sub
        End If
    End If
    ' Move to the record
    Me.Bookmark = rst.Bookmark
End Sub
