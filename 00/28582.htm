Sub PushValues()
   Dim r As Range

   Set r = Sheets("TableShooters").[A3]
   With Sheets("RoundsPoints")
      While Trim(r) <> ""
         If r = .[A2] Then
            r.EntireRow.Cells(, "B") = .[B2]
            r.EntireRow.Cells(, "C") = .[E2]
            r.EntireRow.Cells(, "D") = .[H2]
         End If
         Set r = r.Offset(1)
      Wend
   End With

   'refresh AutoFilter
   If r.Worksheet.FilterMode = True Then

      Application.EnableEvents = False
      Application.ScreenUpdating = False

      With ThisWorkbook.CustomViews.Add("MyView", , True)
         r.Worksheet.AutoFilterMode = False
         .Show
         .Delete
      End With

      Application.ScreenUpdating = True
      Application.EnableEvents = True
   End If
End Sub

gleam2 say:

To understand what is happening you can click the cursor somewhere in the code and press F8 to step through the code.

To understand what a command does, put the cursor inside the word and press F1.

Here is some alternative code. Not better just different. It also clears the values on the sheet once the data has been transferred. If the "Add shots" = 0 it queries whether you really want to transfer the data.

Option Explicit
Sub MoveValues()
'
Dim Name As String
Dim MyRow As Long, NoRows As Long, i1 As Long
Dim ans As Integer
Dim Fnd As Boolean
'
    If ActiveSheet.Name <> "RoundsPoints" Then
        MsgBox "Must start on sheet RoundsPoints"
        Exit Sub
    End If
    Name = Range("A2").Value
    Sheets("TableShooters").UsedRange ' reset value in case rows have been deleted
    NoRows = Sheets("TableShooters").UsedRange.Rows.Count ' depends on cells from row 1 being used
' look for name on summary
    i1 = 3
    Fnd = False
    While i1 <= NoRows And Not Fnd
        If Name = Sheets("TableShooters").Range("A" & i1) Then
            Fnd = True
            If Range("D2") = 0 Then
                ans = MsgBox("For " & Name & " Add shots =0. Do you really want to transfer this data?", vbYesNo)
                If ans <> vbYes Then Exit Sub
            End If
            Sheets("TableShooters").Range("B" & i1) = Range("B2")
            Sheets("TableShooters").Range("C" & i1) = Range("E2")
            Range("D2").ClearContents
            Sheets("TableShooters").Range("D" & i1) = Range("H2")
            Range("G2").ClearContents
        End If
        i1 = i1 + 1
    Wend
    If Not Fnd Then
        MsgBox "Name " & Name & " not found."
    Else
        MsgBox Name & " transfered."
    End If
   
End Sub
