Since you didn't tell us what version of Excel you are using, I'll provide 2 sets of instructions to deal with the printed version of the worksheet:

 

For Excel 2007 and/or 2010 - you will have several "Tabs" across the top of the sheet instead of the older Excel 2003 and earlier menu items.

Go to the [Page Layout] tab and in the {Page Setup} group, click on the little pull-down arrow at the lower right corner of that group.  The Page Setup dialog will open;

To fit to page, go to the [Page] tab in the dialog and click on the "Fit to:" button and choose 1 wide by 1 tall to fit all of it on a single sheet of paper.  If you have a really long (lots of rows) worksheet, you can choose 1 for wide and delete the entry in the "tall" setting to get it all to fit in 1 sheet's width, but go on to other sheets to printout all used rows.

For Excel 2003 and earlier, use the menu:  File --> Page Setup which will bring up the Page Setup dialog and you can then follow the instructions above to complete the tasks.

If the Word documents have not yet been created/distributed, try the following. I haven't explored the new content controls, but since the form fields are still there, I suspect this will do what you need.

  • Show the developer ribbon in Word
  • click to enable design mode
  • Under controls, select more [drop down]/legacy/text form field (stay away from the Active X controls)

here is some code to get you started- this is in Excel VBA, used to open and grab data from all files in a target directory. Note that I last used this in Excel 2010, so YMMV (especially if you are using 64-bit Excel 2016). You will see that in addition to just a few open text fields, I was also collecting data from a bunch of radiobuttons (this was a survey, so I had a bunch of questions, and within each question I had multiple levels of response- so I used shorthand in naming the buttons in Word so that they would be easy to identify).

Sub LoopThroughFiles()
    Dim PasteRow As Long
    PasteRow = 0
    Dim StrFile As String
    oPath = GetPathToUse 'NC
    StrFile = Dir(oPath)
    'Dim ctrl As MSForms.Control
   
    Dim wdISh As Object 'Word.InlineShape
    Dim ctlObj As Object 'This is not a MSForms.Control !
    Dim ctlOB As Object 'MSForms.OptionButton
    Const wdInlineShapeOLEControlObject = 5
   
    'StrFile = Dir(BrowseFolder("Please select a folder") & "\*")
    Do While Len(StrFile) > 0
        'MsgBox StrFile
        DocExtLong = Right(StrFile, 4)
        If InStr(DocExtLong, "doc") > 0 Then 'word documents only
       
            PasteRow = PasteRow + 1
            'Sheet1.Range("A" & PasteRow).Value = StrFile 'file name capture
       
            Set myDoc = Documents.Open(Filename:=oPath & StrFile, Visible:=True)
   
   
   
            With myDoc
                'On Error Resume Next
               
                For Each ffld In ActiveDocument.FormFields
                'Debug.Print ffld.Result, ffld.Name
                    If ffld.Name = "TxtName" Then 'this was to get the respondent's name
                        'Put Name above both sets of data
                        Sheet2.Range(ColLetter(PasteRow) & 1).Value = ffld.Result
                        Sheet1.Range(ColLetter(PasteRow) & 1).Value = ffld.Result
                    Else
                        'Just adding text comments now

                        'I had one text field per survey question, so I incorporated

                        'the question number into the field name, like Text1, Text2, etc. 

                        'so the next line strips off the word 'Text', then adds 1 because

                        'I had header rows, so "Text1" needed to be pasted into row 2

                        ffldID = Val(Right(ffld.Name, Len(ffld.Name) - 4)) + 1 'gives us the UseRow

                        'PasteRow is bad naming on my part- each subsequent word file

                        'goes into the next *column*, so the value is incremented and converted

                        'to the alpha col letter. I originally started this project expecting to have

                        'the survey results in rows instead of columns, and never went back to rename

                        Sheet1.Range(ColLetter(PasteRow) & ffldID).Value = ffld.Result
                       
                    End If
               
                Next
               
                'do all the optionbuttons 'this was the majority of my fields
               
                For Each wdISh In .InlineShapes
                     'Is it a OLE control?
                     If wdISh.Type = wdInlineShapeOLEControlObject Then
                       'Get the object inside
                       Set ctlObj = wdISh.OLEFormat.Object
                       'What kind of?
                       Select Case TypeName(ctlObj)
                         Case "OptionButton"
                           'Show it the caption and value
                           Set ctlOB = ctlObj
                           'Debug.Print ctlOB.Caption, ctlOB.Value, ctlOB.Name, ctlOB.GroupName
                            If ctlOB.Value = True Then
                                If Right(ctlOB.GroupName, 6) = "Source" Then 'a question about who was responding
                                    'initial group identifier
                                    xVal = Right(ctlOB.Name, 4)
                                    If Left(xVal, 1) = "_" Then xVal = Right(xVal, 3)
                                    Sheet2.Range(ColLetter(PasteRow) & "2").Value = xVal
                                Else
                                    'all other optionbuttons
                                    xRow = Val(Right(ctlOB.GroupName, 2)) + 2
                                    xVal = Mid(ctlOB.Name, 5, 2)
                                    'i = i + 1
                                    Sheet2.Range(ColLetter(PasteRow) & xRow).Value = xVal
                                End If
                           End If
                         Case Else 'not an optionbutton

                           'I had the form protected with a password so I never had a problem,

                           'this code was just in case I came across an unexpected form field I'd

                           'at least be able to detect there was something weird going on

                           Debug.Print "POSSIBLE PROBLEM: " & TypeName(ctlObj), ctlOB.Caption, ctlOB.Value, ctlOB.Name, ctlOB.GroupName 
                       End Select
                    
                     Else 'what else is it?
                        Debug.Print wdISh.Type
                    
                     End If
                   Next
           
                '.Saved = True
                .Close (Word.WdSaveOptions.wdDoNotSaveChanges)

            End With
   
    End If
        StrFile = Dir
    Loop
End Sub


Private Function GetPathToUse() As Variant
Dim fDialog As FileDialog
  Set fDialog = Application.FileDialog(msoFileDialogFolderPicker)
  With fDialog
    .Title = "Select Folder containing the completed form documents to and click OK"
    .AllowMultiSelect = False
    .InitialView = msoFileDialogViewList
    If .Show <> -1 Then
      GetPathToUse = ""
      Set fDialog = Nothing
      Exit Function
    End If
    GetPathToUse = fDialog.SelectedItems.Item(1)
    If Right(GetPathToUse, 1) <> "\" Then GetPathToUse = GetPathToUse + "\"
  End With
lbl_Exit:
  Exit Function
End Function


Function ColLetter(ColNumber As Long) As String
    On Error Resume Next
    ColLetter = Application.Substitute(Application.ConvertFormula("R1C" & ColNumber, xlR1C1, xlA1, 4), "1", "")
End Function
