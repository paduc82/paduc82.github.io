bernie deitrick say:

Code like this will replace any string starting with the word "damage" with whatever the user enters into the inputbox

Sub Textbox1_Change()

    If LCase(Me.TextBox1.Value) = "damage" Then Call AddNote_Command(Me.TextBox1)

End Sub

Sub AddNote_Command(Cntrl As Control)
    Cntrl.Value = InputBox("Please provide a reason as to why this item is damaged.", "Add Note")
    If Cntrl.Value = "" Then Exit Sub
    'do something else
End Sub
