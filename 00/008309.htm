I've found a better technic to handle the tree and to get rid of all those keys, with a slight changes.

Replace the Drop-Routine with this one:

Private Sub TreeView1_OLEDragDrop(Data As MSComctlLib.DataObject, _
    Effect As Long, Button As Integer, Shift As Integer, _
    x As Single, Y As Single)
  'Drop to last highlighted item
  Dim Item As String
  'Get item to drop
  Item = Data.GetData(1)
  'Do nothing if there is nothing
  If Len(Item) = 0 Then Exit Sub
  'Add a node
  With TreeView1
    If DragNode Is Nothing Then
      'Add to root
      Set DragNode = .Nodes.Add(Text:=Item)
    Else
      'Add to existing node
      Set DragNode = .Nodes.Add(DragNode, tvwChild, Text:=Item)
    End If
    'Expand the node
    DragNode.Expanded = True
  End With
End Sub

And the save/load routines with the code below. Now you are able to load several different trees to one together. Routines to save/load a tree into/from a sheet are added too.

Sub SaveTree(ByVal FileName As String, ByVal Treeview As MSComctlLib.Treeview)
  'Save all nodes from tree in a file
  Dim fso As Object    'FileSystemObject
  Dim T As Object    'TextStream
  'Stack of nodes for non recursiv visiting
  Dim Stack() As Node
  Dim StackPtr As Long
  Dim Root As Node
  'Try open the file
  On Error Resume Next
  Set fso = CreateObject("Scripting.FileSystemObject")
  Set T = fso.CreateTextFile(FileName, True)
  If T Is Nothing Then Exit Sub
  'Start by the root node
  On Error GoTo ExitPoint
  Set Root = Treeview.Nodes(1)
  ReDim Stack(1 To 64) As Node
  Do While Not Root Is Nothing
    T.WriteLine Root.Text
    T.WriteLine GetNodesDepth(Root)
    'Are there subnodes?
    If Root.Children > 0 Then
      'Yes, save this node
      StackPtr = StackPtr + 1
      If StackPtr > UBound(Stack) Then ReDim Preserve Stack(1 To UBound(Stack) * 2) As Node
      Set Stack(StackPtr) = Root
      'Go on with the child node
      Set Root = Root.Child
    Else
      'Go on with the next node
      Set Root = Root.Next
      'Get the next node from stack if there isn't a node
      Do While Root Is Nothing And StackPtr > 0
        Set Root = Stack(StackPtr).Next
        StackPtr = StackPtr - 1
      Loop
    End If
  Loop
ExitPoint:
  T.Close
End Sub

Sub LoadTree(ByVal FileName As String, ByVal Treeview As MSComctlLib.Treeview)
  'Load all nodes from a file into a tree, remove existing nodes in tree
  Dim fso As Object    'FileSystemObject
  Dim T As Object    'TextStream
  Dim Contents As String, Level As String
  Dim Stack() As Node

  'Try open the file
  On Error Resume Next
  Set fso = CreateObject("Scripting.FileSystemObject")
  Set T = fso.OpenTextFile(FileName, Create:=False)
  If T Is Nothing Then Exit Sub

  'Read the file line by line
  On Error GoTo ExitPoint
  ReDim Stack(1 To 64) As Node
  Do While Not T.AtEndOfStream
    'Get the contents of the node
    Contents = T.ReadLine
    Level = T.ReadLine
    If Level > UBound(Stack) Then
      ReDim Preserve Stack(1 To UBound(Stack) * 2) As Node
    End If
    If Level = 1 Then
      'This is a root node
      Set Stack(Level) = Treeview.Nodes.Add(Text:=Contents)
    Else
      'This is a child node
      Set Stack(Level) = Treeview.Nodes.Add( _
        Relative:=Stack(Level - 1), RelationShip:=tvwChild, Text:=Contents)
    End If
  Loop
ExitPoint:
  T.Close
End Sub

Sub SaveTreeSheet(ByVal WS As Worksheet, ByVal Treeview As MSComctlLib.Treeview)
  'Save all nodes from tree into a sheet
  Dim Stack() As Node
  Dim StackPtr As Long
  Dim Root As Node
  Dim Line As Long
  'Start by the root node
  On Error GoTo ExitPoint
  Set Root = Treeview.Nodes(1)
  ReDim Stack(1 To 64) As Node
  'Prepare sheet
  WS.Columns(1).Resize(, 2).ClearContents
  WS.Columns(1).NumberFormat = "@"
  WS.Columns(2).NumberFormat = "0"
  Do While Not Root Is Nothing
    Line = Line + 1
    WS.Cells(Line, 1) = Root.Text
    WS.Cells(Line, 2) = GetNodesDepth(Root)
    'Are there subnodes?
    If Root.Children > 0 Then
      'Yes, save this node
      StackPtr = StackPtr + 1
      If StackPtr > UBound(Stack) Then ReDim Preserve Stack(1 To UBound(Stack) * 2) As Node
      Set Stack(StackPtr) = Root
      'Go on with the child node
      Set Root = Root.Child
    Else
      'Go on with the next node
      Set Root = Root.Next
      'Get the next node from stack if there isn't a node
      Do While Root Is Nothing And StackPtr > 0
        Set Root = Stack(StackPtr).Next
        StackPtr = StackPtr - 1
      Loop
    End If
  Loop
ExitPoint:
End Sub

Sub LoadTreeSheet(ByVal WS As Worksheet, ByVal Treeview As MSComctlLib.Treeview)
  'Load all nodes from a sheet into a tree, remove existing nodes in tree
  Dim Contents As String, Level As String
  Dim Stack() As Node
  Dim Line As Long

  'Read the sheet line by line
  On Error GoTo ExitPoint
  ReDim Stack(1 To 64) As Node
  Line = 1
  Do While Not IsEmpty(Cells(Line, 1))
    'Get the contents of the node
    Contents = Cells(Line, 1)
    Level = Cells(Line, 2)
    Line = Line + 1
    If Level > UBound(Stack) Then
      ReDim Preserve Stack(1 To UBound(Stack) * 2) As Node
    End If
    If Level = 1 Then
      'This is a root node
      Set Stack(Level) = Treeview.Nodes.Add(Text:=Contents)
    Else
      'This is a child node
      Set Stack(Level) = Treeview.Nodes.Add( _
        Relative:=Stack(Level - 1), RelationShip:=tvwChild, Text:=Contents)
    End If
  Loop
ExitPoint:
End Sub

Please be informed that if you chose to rename your account to @Outlook.com, a prompt would have asked you Where do you want to save your emails?, you will have the option to create a new folder for your old account or save the emails in the Inbox of your renamed account.
