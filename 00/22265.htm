Adding to TasosK's fine answer: if for some reason you want to sort only one column when there is a change:

Private Sub Worksheet_Change(ByVal Target As Range)
 Dim rA As Long, rB As Long
 rA = Cells(Rows.Count, "A").End(xlUp).Row
 rB = Cells(Rows.Count, "B").End(xlUp).Row
 If Not Intersect(Target, Range("A1:A" & rA, "B1:B" & rB)) Is Nothing Then
   If Target.Column = 1 Then
    Range("A1:A" & rA).Sort Key1:=Range("A1"), Order1:=xlAscending, Header:=xlYes
   Else
    Range("B1:B" & rB).Sort Key1:=Range("B1"), Order1:=xlAscending, Header:=xlYes
   End If
 End If
 End Sub
