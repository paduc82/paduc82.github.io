Sub macro_01()

myPath = ThisWorkbook.Path & "\"

.SaveAs Filename:=myPath & WB.Sheets(i).Name & ".xlsx"
Application.DisplayAlerts = True
.Close False

End With
Next i
On Error GoTo 0
Application.ScreenUpdating = True
End Sub

Target.SpecialCells(xlCellTypeAllValidation) causes an error on a protected sheet, so you have to unprotect the sheet in the code, then protect it again/

Here is a modified version:

Private Sub Worksheet_Change(ByVal Target As Range)
    'Code by Sumit Bansal from https://trumpexcel.com
    ' To allow multiple selections in a Drop Down List in Excel (without repetition)
    Dim Oldvalue As String
    Dim Newvalue As String
    Application.EnableEvents = True
    On Error GoTo ExitSub
    If Target.Address = "$I$13" Then
        Me.Unprotect 'Password:="secret"
        If Target.SpecialCells(xlCellTypeAllValidation) Is Nothing Then
            GoTo ExitSub
        ElseIf Target.Value = "" Then
            GoTo ExitSub
        Else
            Application.EnableEvents = False
            Newvalue = Target.Value
            Application.Undo
            Oldvalue = Target.Value
            If Oldvalue = "" Then
                Target.Value = Newvalue
            ElseIf InStr(1, Oldvalue, Newvalue) = 0 Then
                Target.Value = Oldvalue & ", " & Newvalue
            Else
                Target.Value = Oldvalue
            End If
        End If
        Me.Protect 'Password:="secret"
    End If
ExitSub:
    Application.EnableEvents = True
End Sub
