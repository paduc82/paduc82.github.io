andreas killer say:

Does anyone know what the limit is for number of rows that can be accessed via .ListObject.ListRows?

65535 rows, but that seems to be a limit inside the ListRows object only.

When you use the DataBodyRange property you can access all cells. See code below.

Andreas.

Option Explicit

Sub Setup()
  Dim Data(0 To 250000, 1 To 1)
  Dim i As Long
  Data(0, 1) = "Nr"
  For i = 1 To UBound(Data)
    Data(i, 1) = i
  Next
  Range("A1").Resize(UBound(Data) + 1, 1).Value = Data
  ActiveSheet.ListObjects.Add xlSrcRange, Range("A1").CurrentRegion, , xlYes
End Sub

Sub Test()
  Dim LO As ListObject
  Dim R As Range
  Dim LR As ListRow
 
  Set LO = ActiveSheet.ListObjects(1)
 
  'Works:
  Debug.Print LO.ListRows.Count
 
  'RTE 7: Not enough memory
  For Each LR In LO.ListRows
    Debug.Print LR.Range.Row
  Next
 
  'Works:
  For Each R In LO.DataBodyRange.Rows
    Debug.Print R.Row
  Next
 
  'Resize the table
  LO.Resize Range("A1").Resize(65535, 1)
  Debug.Print LO.ListRows.Count
 
  'Works now:
  For Each LR In LO.ListRows
    Debug.Print LR.Range.Row
  Next
End Sub
