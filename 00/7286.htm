Alternate Workaround

This will allow you to use CreateObject(‚ÄúScriptlet.TypeLib‚Äù) however it will lessen the security protection added by the Office July 2017 security update.

You can set a registry key that will allow Scriptlet.TypeLib to instantiate inside of Office applications. When this registry key is set, Office will not block any use of this COM control.

  • Open Registry Editor.

  • Go to: Computer\HKEY_LOCAL_MACHINE\SOFTWARE\Microsoft\Office\16.0\Common\COM Compatibility\{06290BD5-48AA-11D2-8432-006008C3FBFC}

HKEY_LOCAL_MACHINE

    COM Compatibility

  7. In the Value data box, type 1 to enable the registry entry, and then click OK.

  9. Exit Registry Editor, and then restart any open Office application.

Sub CopyPicsFromAllExcelSheetsToPPT()
' Add a reference to the Microsoft PowerPoint Library via
' VBE > Tools > References
'
    Dim pp As PowerPoint.Application
    Dim ppSld As PowerPoint.Slide
    Dim ppPres As PowerPoint.Presentation
    Dim ppShpRng As PowerPoint.ShapeRange
   
    Dim ws As Worksheet
    Dim arrPic() As String
    Dim strActFileName As String
    Dim x As Long
    Dim xlShp As Shape
   
    arrPic = getPicNames
   
    strActFileName = Application.GetOpenFilename _
          ("Microsoft PowerPoint-Files (*.ppt), *.ppt")
          'WHAT ABOUT PPTX FILES????
         
     'Look for existing instance, create new one if required:
    On Error Resume Next
    Set pp = GetObject(, "PowerPoint.Application")
    On Error GoTo 0
    If pp Is Nothing Then
        Set pp = CreateObject("Powerpoint.Application")
    End If
       
    If strActFileName = False Then
        pp.Activate
        Set ppPres = pp.Presentations.Add
    Else
        pp.Activate
        Set ppPres = pp.Presentations.Open(strActFileName)
    End If
    
     'Loop through each worksheet:
    For Each ws In ActiveWorkbook.Worksheets
        If ws.Visible Then
             'Loop through each picture in the worksheet and paste them into the PowerPoint
            x = 0
            Do While x < 30
                On Error Resume Next
                Set xlShp = ws.Shapes(arrPic(x))
                On Error GoTo 0
                If Not xlShp Is Nothing Then
                     'Add a new slide where we will paste the picture
                    Set ppSld = ppPres.Slides.Add(1 + x, PowerPoint.ppLayoutTitleOnly)
                   
                    xlShp.CopyPicture xlScreen, xlPicture
               
                    Set ppShpRng = ppSld.Shapes.PasteSpecial(DataType:=ppPasteMetafilePicture)
                                       
                     'Adjust the positioning of the Picture on Powerpoint Slide
                    ppShpRng.Left = 25
                   
                    'ppSld.Shapes(2).Width = 200
                    'ppSld.Shapes(2).Left = 505
                   
                Else
                    Debug.Print "Shape '" & arrPic(x) & "' could not be found!"
                    Stop 'for debug only
                    GoTo NextWS
                End If
           
                Set xlShp = Nothing
                x = x + 1
            Loop
        End If
NextWS:
    Next ws
End Sub

Function getPicNames()
' Array of names
'
    Dim arr(29) As String
    Dim i As Long
   
    For i = 0 To 29
        arr(i) = "Picture " & CStr(i + 1)
    Next i
   
    getPicNames = arr
End Function
