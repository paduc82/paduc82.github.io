jay freedman say:

Let's assume that the string you're searching for may occur in text outside any table, within a table, or both. Let's say you want to change the font color of the string to red, only where it occurs in a table. Then you need code like this:

Sub FindInTable()
    Dim X As String
    X = InputBox("Enter the string to find:")
    If Len(X) > 0 Then
        Selection.HomeKey wdStory, False
        With Selection.Find
            .Text = X
            .Wrap = wdFindStop
            While .Execute
                If Selection.Information(wdWithInTable) Then
                    Selection.Font.ColorIndex = wdRed
                End If
            Wend
        End With
    End If
End Sub

You can do the same kind of thing with a Range object instead of the Selection, which would avoid moving the cursor. Because moving the cursor could cause Word to redraw the screen many times, which is a comparatively slow operation, using a Range can make such a macro run much faster. Other than needing to declare the variable As Range and replacing Selection with that variable's name, the only other change is to replace the Selection.HomeKey statement with a statement that assigns the starting range, something like

   Set rng = ActiveDocument.Range

EDIT: adding the code for people who avoid downloading. Just make sure you have a sheet named "centralize" to gather info from all others...


Sub A_YES_Copier()
Dim ASht As Object, srcI As Long, destI As Long, srcLastI As Long, CntrlSht As Object
Dim srcJ As Long, srcLastJ As Long, YesFL As Boolean

Application.Calculation = xlCalculationManual
Application.ScreenUpdating = False

Set CntrlSht = Sheets("centralize")
destI = 4 'Row to start report
With CntrlSht
.Cells(destI, 2).Resize(Cells(Rows.Count, 2).End(xlUp).Row + 1 - destI, Cells(4, Columns.Count).End(xlToLeft).Column - 1).ClearContents
End With

For Each ASht In Sheets

If ASht.Name <> CntrlSht.Name Then

srcLastI = ASht.Cells(Rows.Count, 2).End(xlUp).Row
For srcI = 4 To srcLastI
YesFL = False: srcLastJ = ASht.Cells(srcI, Columns.Count).End(xlToLeft).Column
For srcJ = 2 To srcLastJ
If InStr(LCase(ASht.Cells(srcI, srcJ)), "yes") > 0 Then YesFL = True: Exit For
Next
If YesFL Then
Range(ASht.Cells(srcI, 2), ASht.Cells(srcI, srcLastJ)).Copy
CntrlSht.Cells(destI, 2).PasteSpecial Paste:=xlPasteValues
destI = destI + 1
End If
Next

End If

Next

Application.CutCopyMode = True
Application.Calculation = xlCalculationAutomatic
Application.ScreenUpdating = True

End Sub
