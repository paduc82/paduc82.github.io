Use a macro containing the following code:

Dim i As Long
Dim rngFruit As Range
With Selection.Tables(1)
    For i = 2 To .Rows.count
        Set rngFruit = .Cell(i, 2).Range
        With rngFruit
            .End = .End - 1

            End Select
        End With
    Next i
End With
