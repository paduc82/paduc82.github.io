steve rindsberg say:

I don't know that you can set this as a default for new shapes. 

But a simple but of VBA will fix all the shapes in the presentation:

Sub LockEmDanO()
' Sets all shapes to Locked Aspect Ratio

Dim oSl As Slide
Dim oSh As Shape

For Each oSl In ActivePresentation.Slides
    For Each oSh In oSl.Shapes
        oSh.LockAspectRatio = True
    Next
Next
End Sub

Or this, to set all of the currently selected shapes:

Sub LockEmDanO()
' Sets all selected shapes to Locked Aspect Ratio
Dim oSh As Shape

With ActiveWindow.Selection
For Each oSh In .ShapeRange
    oSh.LockAspectRatio = True
Next

End Sub
