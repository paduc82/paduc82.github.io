graham mayor say:
It shouldn't do that - try the following version instead

Sub ExtractEMailAddresses()
Dim oDoc As Document
Dim oNewDoc As Document
Dim oRng As Range
Dim strAddress As String
Dim hLink As Hyperlink
Set oDoc = ActiveDocument
Set oNewDoc = Documents.Add
Options.AutoFormatReplaceHyperlinks = True
oDoc.Range.AutoFormat
For Each hLink In oDoc.Hyperlinks
    If InStr(1, hLink.Address, "@") Then
        strAddress = Replace(hLink.Address, "mailto:", "")
        oNewDoc.Range.InsertAfter strAddress & vbCr
    End If
Next hLink
oNewDoc.Paragraphs.Last.Range.Delete
oDoc.Close wdDoNotSaveChanges
oNewDoc.Range.Sort
oNewDoc.Range.Find.Execute FindText:="(*^13)@", _
                           MatchWildcards:=True, _
                           ReplaceWith:="\1", _
                           Replace:=wdReplaceAll
End Sub
