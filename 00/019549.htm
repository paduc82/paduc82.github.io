You can assign the ranges directly to range variables without creating the string. However, you are still limited on the size of the string for assigning to individual range variables but you can break them up and assign to separate ranges and then use Union to join the ranges into a single range variable.

In the following example I simply used find and replace to create the second range with column G and the third range with column J for testing purposes but I have effectively trebled the size of the final range. (I don't know if there is a limit to how many you can have but I guess there is.)

Also note that I have used the space and underscore at the end of each line which is a line break in an otherwise single line of code. When this is done in the middle of strings enclosed in double quotes it is necessary to close off the double quote, add an ampersand and then open the double quotes on the next line so that the separate strings are concatenated.

Sub test()
   
    Dim inputWks
    Dim myRng1 As Range
    Dim myRng2 As Range
    Dim myRng3 As Range
   
    Dim myRng As Range
   
    Set inputWks = Worksheets("Sheet1")
   
    With inputWks
   
        Set myRng1 = .Range("D5,D7,d9,d11,d12,d17,d21,d23,d25,d27," & _
                   "d29,d31,d35,d37,d39,d41,d43,d45,d49,d51,d53," & _
                   "d55,d57,d59,d62,d64,d67,d69,d71,d73,d77,d79,d81," & _
                   "d83,d87,d89,d91,d93,d95,d97,d99,d101,d103,d105," & _
                   "d109,d111,d113,d115,d117,d121,d123,d125,d127,d129," & _
                   "d131,d133,d135,d137,d139")
       
        Set myRng2 = .Range("G5,G7,G9,G11,G12,G17,G21,G23,G25,G27," & _
                   "G29,G31,G35,G37,G39,G41,G43,G45,G49,G51,G53," & _
                   "G55,G57,G59,G62,G64,G67,G69,G71,G73,G77,G79,G81," & _
                   "G83,G87,G89,G91,G93,G95,G97,G99,G101,G103,G105," & _
                   "G109,G111,G113,G115,G117,G121,G123,G125,G127,G129," & _
                   "G131,G133,G135,G137,G139")
                  
       
        Set myRng3 = .Range("J5,J7,J9,J11,J12,J17,J21,J23,J25,J27," & _
                   "J29,J31,J35,J37,J39,J41,J43,J45,J49,J51,J53," & _
                   "J55,J57,J59,J62,J64,J67,J69,J71,J73,J77,J79,J81," & _

                   "J109,J111,J113,J115,J117,J121,J123,J125,J127,J129," & _
                   "J131,J133,J135,J137,J139")
       
       
        Set myRng = Union(myRng1, myRng2, myRng3)
       
        If Application.CountA(myRng) <> myRng.Cells.Count Then
            MsgBox "Please fill in all the cells!"
            Exit Sub
        End If
    End With

End Sub
