shyam pillai say:

I had this written a while back so figured I'd post it since it is a complete example.


Option Explicit

Private Type POINT
    x As Single
    y As Single
End Type


Const PI = 3.14159265358979


Public Function DegreeToRadians(ByVal Angle As Single) As Single
DegreeToRadians = Angle / 180 * PI
End Function


Public Function RotatePoint(ByRef pPoint As POINT, ByRef pOrigin As POINT, ByVal Degrees As Long) As POINT
RotatePoint.x = Cos(DegreeToRadians(Degrees)) * (pPoint.x - pOrigin.x) - Sin(DegreeToRadians(Degrees)) * (pPoint.y - pOrigin.y) + pOrigin.x
RotatePoint.y = Sin(DegreeToRadians(Degrees)) * (pPoint.x - pOrigin.x) + Cos(DegreeToRadians(Degrees)) * (pPoint.y - pOrigin.y) + pOrigin.y
End Function


Function Max(a As Single, b As Single) As Single
If a > b Then
    Max = a
Else
    Max = b
End If
End Function


Function Min(a As Single, b As Single) As Single
If a > b Then
    Min = b
Else
    Min = a
End If
End Function


Sub DrawRotatedBounds()
Dim P1 As POINT
Dim P2 As POINT
Dim P3 As POINT
Dim P4 As POINT


Dim Origin As POINT


Dim R1 As POINT
Dim R2 As POINT
Dim R3 As POINT
Dim R4 As POINT


Dim MinBounds As POINT
Dim MaxBounds As POINT


Dim shp As Shape
Dim sld As Slide


Set sld = ActiveWindow.Selection.SlideRange(1)

Set shp = sld.Shapes.AddShape(msoShapeRectangle, 150, 50, 100, 180)


MinBounds.x = shp.Left
MinBounds.y = shp.Top


MaxBounds.x = shp.Left + shp.Width
MaxBounds.y = shp.Top + shp.Height


'Setting up the non-rotated vertices
P1.x = shp.Left
P1.y = shp.Top


P2.x = shp.Left + shp.Width
P2.y = shp.Top


P3.x = shp.Left + shp.Width
P3.y = shp.Top + shp.Height


P4.x = shp.Left
P4.y = shp.Top + shp.Height


'Shape rotates about the center
Origin.x = shp.Left + (shp.Width / 2)
Origin.y = shp.Top + (shp.Height / 2)


'Rotate the shape
shp.Rotation = 30


' Get rotated points
R1 = RotatePoint(P1, Origin, shp.Rotation)
R2 = RotatePoint(P2, Origin, shp.Rotation)
R3 = RotatePoint(P3, Origin, shp.Rotation)
R4 = RotatePoint(P4, Origin, shp.Rotation)


'Find the min and max bounding points
MinBounds.x = Min(R1.x, MinBounds.x)
MinBounds.y = Min(R1.y, MinBounds.y)

MinBounds.x = Min(R2.x, MinBounds.x)
MinBounds.y = Min(R2.y, MinBounds.y)

MinBounds.x = Min(R3.x, MinBounds.x)
MinBounds.y = Min(R3.y, MinBounds.y)

MinBounds.x = Min(R4.x, MinBounds.x)
MinBounds.y = Min(R4.y, MinBounds.y)


MaxBounds.x = Max(R1.x, MaxBounds.x)
MaxBounds.y = Max(R1.y, MaxBounds.y)

MaxBounds.x = Max(R2.x, MaxBounds.x)
MaxBounds.y = Max(R2.y, MaxBounds.y)

MaxBounds.x = Max(R3.x, MaxBounds.x)
MaxBounds.y = Max(R3.y, MaxBounds.y)

MaxBounds.x = Max(R4.x, MaxBounds.x)
MaxBounds.y = Max(R4.y, MaxBounds.y)


Dim oShpRotatedBounds As Shape

Set oShpRotatedBounds = sld.Shapes.AddShape(msoShapeRectangle, _
                        MinBounds.x, _
                        MinBounds.y, _
                        MaxBounds.x - MinBounds.x, _
                        MaxBounds.y - MinBounds.y)

oShpRotatedBounds.Fill.Visible = False

End Sub

Sub BBox()
Dim oshp As Shape
Dim lngRot As Long
Dim BB_W As Single
Dim BB_H As Single
Dim BB_L As Single
Dim BB_T As Single
Dim oshpC(1 To 2) As Single
Const deg2Rad As Single = 3.14159 / 180
Set oshp = ActiveWindow.Selection.ShapeRange(1)
oshpC(1) = oshp.Left + oshp.Width / 2
oshpC(2) = oshp.Top + oshp.Height / 2
lngRot = oshp.Rotation
BB_W = oshp.Height * Abs(Sin(lngRot * deg2Rad)) + oshp.Width * Abs(Cos(lngRot * deg2Rad))
BB_H = oshp.Width * Abs(Sin(lngRot * deg2Rad)) + oshp.Height * Abs(Cos(lngRot * deg2Rad))
BB_L = oshpC(1) - BB_W / 2
BB_T = oshpC(2) - BB_H / 2
MsgBox "Bounding Box: " & vbCrLf & vbCrLf & _
"Left: " & vbTab & BB_L & vbCrLf & _
"Top: " & vbTab & BB_T & vbCrLf & _
"Width: " & vbTab & BB_W & vbCrLf & _
"Height: " & vbTab & BB_H
End Sub
