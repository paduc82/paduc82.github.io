jllatham say:

Try this on a COPY of your workbook - it would definitely be destructive if I haven't gotten it all correct.  Test it and check it very, very closely, especially the formulas created, to make sure that nothing got hopelessly messed up before using it in the real world.

 

If it should happen to generate a runtime error, please report the error number and text shown to you, and then click the [Debug] button and copy down the line of code that's highlighted (a screen capture of it would even be better).  It does compile without error, but that doesn't mean I haven't slipped up somewhere.

 

To adapt it to other workbooks, just change the appropriate values that are Const values in this code.  With luck on your side, none of the columns will need to be changed, and only the names of the two sheets involved will have to be.  Some of the columns are referenced twice, but I thought it safer just in case things aren't always the same in other workbooks.

 

Here's the code I came up with:

 

Sub ProcessTwoSheets()
  'change these Const values to set up to match with the two sheets
  'in a different workbook
'Info for the First sheet (was 27361 in original)
  Const firstSheetName = "27361" ' change per workbook
  's1 stands for the firstSheetName sheet
  Const s1FirstDataRow = 2
  Const s1CommaGroup1 = "E:G"
  Const s1FirstSortCol = "F"
  Const s1FirstColToSort = "A"
  Const s1LastColToSort = "AD"
  Const s1SecondSortCol = "E"
  Const s1SeriesCol = "C"
  Const s1FormulaCol = "D"
'
'Info for the Second sheet (was 3610 in original)
  Const secondSheetName = "3610"
  's2 stands for the secondSheetName sheet
  Const s2FirstDataRow = 2
  Const s2FirstSortRow = 3 ' for the first sorting
  Const s2CommaGroup1 = "H:I"
  Const s2CommaGroup2 = "F"
  Const s2FirstSortCol = "H"
  Const s2FirstColToSort = "A"
  Const s2LastColToSort = "AI"
  Const s2SecondSortCol = "E"
  Const s2FirstFormulaCol = "E"
  Const s2SecondFormulaCol = "F"
  Const s2ThirdFormulaCol = "G"
'end of user re-definable constants
  '
  'working values
  Dim anyWS As Worksheet
  Dim lastRow As Long ' will be like 390 and 412 were in the original
  Dim sortRange As Range
  Dim sortKey As Range
  Dim anyFormulaR1C1 As String
 
  'this will help speed things up
  Application.ScreenUpdating = False
  'begin the work on the first sheet
  Set anyWS = ThisWorkbook.Worksheets(firstSheetName)
  anyWS.Columns(s1CommaGroup1).Style = "Comma"
  On Error Resume Next
  lastRow = anyWS.Cells.Find(What:="*", _
   After:=anyWS.Cells(1), _
   Lookat:=xlPart, _
   LookIn:=xlFormulas, _
   SearchOrder:=xlByRows, _
   SearchDirection:=xlPrevious, _
   MatchCase:=False).Row
  On Error GoTo 0
  If lastRow < s1FirstDataRow Then
    lastRow = s1FirstDataRow
  End If
  anyWS.Sort.SortFields.Clear
  Set sortKey = anyWS.Range(s1FirstSortCol & s1FirstDataRow)
  Set sortRange = anyWS.Range(s1FirstColToSort & s1FirstDataRow & _
   ":" & s1LastColToSort & lastRow)
  anyWS.Sort.SortFields.Add Key:=sortKey, _
   SortOn:=xlSortOnValues, Order:=xlDescending, DataOption:=xlSortNormal
  With anyWS.Sort
    .SetRange sortRange
    .Header = xlNo
    .MatchCase = False
    .Orientation = xlTopToBottom
    .SortMethod = xlPinYin
    .Apply
  End With
 
  anyWS.Sort.SortFields.Clear
  Set sortKey = anyWS.Range(s1SecondSortCol & s1FirstDataRow)
  Set sortRange = anyWS.Range(s1FirstColToSort & s1FirstDataRow & _
   ":" & s1LastColToSort & lastRow)
  anyWS.Sort.SortFields.Add Key:=sortKey, _
   SortOn:=xlSortOnValues, Order:=xlDescending, DataOption:=xlSortNormal
  With anyWS.Sort
    .SetRange sortRange
    .Header = xlNo
    .MatchCase = False
    .Orientation = xlTopToBottom
    .SortMethod = xlPinYin
    .Apply
  End With

  anyWS.Range(s1SeriesCol & s1FirstDataRow) = 1
  anyWS.Range(s1SeriesCol & s1FirstDataRow & ":" & s1SeriesCol & lastRow).DataSeries _
   rowcol:=xlColumns, Type:=xlLinear, step:=1, Trend:=False
'moved up to this section as long as we're on this sheet
  anyFormulaR1C1 = "=VLOOKUP(RC[-1],'" & secondSheetName & "'!C[1],1,0)"
  anyWS.Range(s1FormulaCol & s1FirstDataRow & ":" _
   & s1FormulaCol & lastRow).FormulaR1C1 = anyFormulaR1C1
 
'***************************************************************************
  'move on to the second sheet
  Set anyWS = ThisWorkbook.Worksheets(secondSheetName) ' was 3610 in original
  On Error Resume Next
  lastRow = anyWS.Cells.Find(What:="*", _
   After:=anyWS.Cells(1), _
   Lookat:=xlPart, _
   LookIn:=xlFormulas, _
   SearchOrder:=xlByRows, _
   SearchDirection:=xlPrevious, _
   MatchCase:=False).Row
  On Error GoTo 0
  'not less than 3
  If lastRow < s2FirstSortRow Then
    lastRow = s2FirstSortRow
  End If
 
  anyWS.Columns(s2CommaGroup1).Style = "Comma" ' H:I
  Set sortKey = anyWS.Range(s2FirstSortCol & s2FirstSortRow)
  Set sortRange = anyWS.Range(s2FirstColToSort & s2FirstDataRow _
   & ":" & s2LastColToSort & lastRow)
 
  anyWS.Sort.SortFields.Add Key:=sortKey, _
   SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:=xlSortNormal
 
  With anyWS.Sort
      .SetRange sortRange
      .Header = xlNo
      .MatchCase = False
      .Orientation = xlTopToBottom
      .SortMethod = xlPinYin
      .Apply
  End With
  anyWS.Columns(s2FirstSortCol & ":" & s2FirstSortCol).EntireColumn.AutoFit
 
  'fill column E with a VLOOKUP() formula
  anyFormulaR1C1 = "=VLOOKUP(RC[-1],'" & firstSheetName & "'!C[-3]:C[-2],2,0)"
  anyWS.Range(s2FirstFormulaCol & s2FirstDataRow & ":" _
   & s2FirstFormulaCol & lastRow).FormulaR1C1 = anyFormulaR1C1
  'fill column F with a VLOOKUP() formula
  anyFormulaR1C1 = "=VLOOKUP(RC[-1],'" & firstSheetName & "'!C[-2]:C,3,0)*-1"
  anyWS.Range(s2SecondFormulaCol & s2FirstDataRow & ":" _
   & s2SecondFormulaCol & lastRow).FormulaR1C1 = anyFormulaR1C1
  'fill column G with a VLOOKUP() formula
  anyFormulaR1C1 = "=RC[1]/RC[-1]"
  anyWS.Range(s2SecondFormulaCol & s2FirstDataRow & ":" _
   & s2SecondFormulaCol & lastRow).FormulaR1C1 = anyFormulaR1C1
  anyWS.Range(s2CommaGroup2 & ":" & s2CommaGroup2).Style = "Comma"
'
'the work should be completed now, release assigned resources
'back to the system
  Set anyWS = Nothing
  Set sortKey = Nothing
  Set sortRange = Nothing
End Sub
