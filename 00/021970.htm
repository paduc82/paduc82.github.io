raderm say:

bernie deitrick say:
For a list of names starting in A2 extending down column A

Sub TestMacro()
    Dim myF As String
    Dim i As Integer
    Dim c As Integer

'folder where you want to save all but the last one
    ChDir "C:\Documents and Settings\username\My Documents\Excel"
'include the path in the name if it is different from the default
    myF = "Name of the picture.jpg"

    c = Cells(Rows.Count, 1).End(xlUp).Row
'Copy the file with the new names
    For i = 2 To c - 1
        FileCopy myF, Cells(i, 1).Value  ' & ".jpg"  'uncomment if the cell does not have the jpg extension
    Next i
'Add the last file to a specific folder
    FileCopy myF, "C:\Documents and Settings\username\My Documents\Excel\JPG Files\" & Cells(c, 1).Value & ".jpg"

End Sub
