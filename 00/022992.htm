'Version 1.0

Option Explicit



Private Sub Example_GetPageSetupHeader()

  Dim Header As String

  Debug.Print GetPageSetupHeader(ActiveSheet.PageSetup.LeftHeader, Characters:=Header)

  Debug.Print Header

End Sub



Sub Example_LetPageSetupHeader()

  Dim S As String

  S = LetPageSetupHeader(Characters:="Hello World", Bold:=True, FontName:="Arial", FontSize:=24, Color:=vbBlue)

  ActiveSheet.PageSetup.RightHeader = S

End Sub



Function GetPageSetupHeader(Expression As String, _

    Optional ByRef Characters, Optional ByRef Bold, Optional ByRef Italic, _

    Optional ByRef FontName, Optional ByRef FontSize, Optional ByRef Color, _

    Optional ByRef DoubleUnderline, Optional ByRef Superscript, Optional ByRef Subscript, _

    Optional ByRef Underline, Optional ByRef Strikethrough, _

    Optional ByRef LeftAlign, Optional ByRef CenterAlign, Optional ByRef RightAlign) As Boolean

  'https://msdn.microsoft.com/en-us/library/bb225426(v=office.12).aspx

  Dim i As Long, j As Long

  Dim Digit As String

  'Setup defaults

  GetPageSetupHeader = False

  On Error Resume Next

  Characters = Empty

  Bold = Empty

  Italic = Empty

  FontName = Empty

  FontSize = Empty

  Color = Empty

  DoubleUnderline = Empty

  Superscript = Empty

  Subscript = Empty

  Underline = Empty

  Strikethrough = Empty

  LeftAlign = Empty

  CenterAlign = Empty

  RightAlign = Empty

  On Error GoTo 0

  'Parse the expression

  For i = 1 To Len(Expression)

    Digit = Mid$(Expression, i, 1)

    If Digit = "&" Then

      i = i + 1

      Digit = Mid$(Expression, i, 1)

      Select Case Digit

        Case "L"

          'Left aligns the characters that follow.

          LeftAlign = True

        Case "C"

          'Centers the characters that follow.

          CenterAlign = True

        Case "R"

          'Right aligns the characters that follow.

          RightAlign = True

        Case "E"

          'Turns double-underline printing on or off.

          DoubleUnderline = True

        Case "X"

          'Turns superscript printing on or off.

          Superscript = True

        Case "Y"

          'Turns subscript printing on or off.

          Subscript = True

        Case "B"

          'Turns bold printing on or off.

          Bold = True

        Case "I"

          'Turns italic printing on or off.

          Italic = True

        Case "U"

          'Turns underline printing on or off.

          Underline = True

        Case "S"

          'Turns strikethrough printing on or off.

          Strikethrough = True

        Case """"

          '"fontname" Prints the characters that follow in the specified font. Be sure to include the double quotation marks.

          j = InStr(i + 1, Expression, """")

          FontName = Mid$(Expression, i + 1, j - i - 1)

          i = j

        Case "K"

          'Prints the characters in the specified color. User supplies a hexidecimal color value.

          Color = VBA.RGB( _

            CLng("&H" & Mid$(Expression, i + 1, 2)), _

            CLng("&H" & Mid$(Expression, i + 3, 2)), _

            CLng("&H" & Mid$(Expression, i + 5, 2)))

          i = i + 6

        Case "D"

          'Prints the current date.

        Case "T"

          'Prints the current time.

        Case "F"

          'Prints the name of the document.

        Case "A"

          'Prints the name of the workbook tab.

        Case "P"

          'Prints the page number.

          '&P+number Prints the page number plus the specified number.

          '&P-number Prints the page number minus the specified number.

          Select Case Mid$(Expression, i + 1)

            Case "+", "-"

              j = i + 2

              Do While Mid$(Expression, j, 1) Like "#"

                j = j + 1

              Loop

              i = j - 1

          End Select

        Case "&"

          'Prints a single ampersand.

        Case "N"

          'Prints the total number of pages in the document.

        Case "Z"

          'Prints the file path.

        Case "G"

          'Inserts an image.

        Case Else

          If Digit Like "#" Then

            '&nn Prints the characters that follow in the specified font size. Use a two-digit number to specify a size in points.

            j = i + 1

            Do While Mid$(Expression, j, 1) Like "#"

              j = j + 1

            Loop

            FontSize = CLng(Mid$(Expression, i, j - i))

            i = j - 1

          Else

            'Unkown

            Exit Function

          End If

      End Select

    Else

      Characters = Mid$(Expression, i)

      Exit For

    End If

  Next

  GetPageSetupHeader = True

End Function



Function LetPageSetupHeader( _

    Optional ByVal Characters, Optional ByVal Bold, Optional ByVal Italic, _

    Optional ByVal FontName, Optional ByVal FontSize, Optional ByVal Color, _

    Optional ByVal DoubleUnderline, Optional ByVal Superscript, Optional ByVal Subscript, _

    Optional ByVal Underline, Optional ByVal Strikethrough, _

    Optional ByVal LeftAlign, Optional ByVal CenterAlign, Optional ByVal RightAlign) As String

  'https://msdn.microsoft.com/en-us/library/bb225426(v=office.12).aspx
  Dim Args() As String
  Dim i As Integer, Red As Integer, Green As Integer, Blue As Integer
  ReDim Args(1 To 13)
  If Not IsMissing(LeftAlign) Then If LeftAlign Then i = i + 1: Args(i) = "L"
  If Not IsMissing(CenterAlign) Then If CenterAlign Then i = i + 1: Args(i) = "C"
  If Not IsMissing(RightAlign) Then If RightAlign Then i = i + 1: Args(i) = "R"
  If Not IsMissing(DoubleUnderline) Then If DoubleUnderline Then i = i + 1: Args(i) = "E"
  If Not IsMissing(Superscript) Then If Superscript Then i = i + 1: Args(i) = "X"
  If Not IsMissing(Subscript) Then If Subscript Then i = i + 1: Args(i) = "Y"
  If Not IsMissing(Bold) Then If Bold Then i = i + 1: Args(i) = "B"
  If Not IsMissing(Italic) Then If Italic Then i = i + 1: Args(i) = "I"
  If Not IsMissing(Underline) Then If Underline Then i = i + 1: Args(i) = "U"
  If Not IsMissing(Strikethrough) Then If Strikethrough Then i = i + 1: Args(i) = "S"
  If Not IsMissing(FontSize) Then i = i + 1: Args(i) = FontSize
  If Not IsMissing(FontName) Then i = i + 1: Args(i) = """" & FontName & """"
  If Not IsMissing(Color) Then
    i = i + 1:
    Red = Color And 255 'Low byte only
    Green = Color \ 256 And 255 'Shr 8
    Blue = Color \ 65536 And 255 'Shr 16
    Args(i) = "K" & Right("0" & Hex(Red), 2) & Right("0" & Hex(Green), 2) & Right("0" & Hex(Blue), 2)
  End If
  If i > 0 Then
    ReDim Preserve Args(1 To i)
    LetPageSetupHeader = "&" & Join(Args, "&")
  End If
  LetPageSetupHeader = LetPageSetupHeader & Characters
End Function
