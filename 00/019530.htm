Public Sub LoopFolders(Folders As Outlook.Folders, _
  ByVal Recursive As Boolean _
)
  Dim Folder As Outlook.MAPIFolder

  For Each Folder In Folders
    LoopItems Folder.Items

    If Recursive Then
      LoopFolders Folder.Folders, Recursive
    End If
  Next
End Sub

Private Sub LoopItems(Items As Outlook.Items)
  Dim obj As Object

  For Each obj In Items

      DoAnything obj
    End If
  Next

dirk goldgar say:

Here's a VBA (not VBScript) version of the routine you've requested, designed to be run from Access.  I've tested it lightly, and it seems to work.

 

'------ start of code ------
Sub CompactAllDatabases( _
        FolderPath As String, _
        Optional IncludeSubfolders As Boolean = False, _
        Optional objFSO As Scripting.FileSystemObject, _
        Optional objEngine As DAO.DBEngine)
       
    On Error GoTo Err_Handler
   
    Dim objFile As Scripting.File
    Dim objFolder As Scripting.Folder
    Dim objSubFolder As Scripting.Folder
    Dim strFullName As String
    Dim strFileName As String
    Dim strFileExt As String
    Dim strLockExt As String
    Dim strSourceFile As String
    Dim strBackupFile As String
    Dim strCompactFile As String
    Dim I As Long
   
    If objEngine Is Nothing Then
        Set objEngine = Application.DBEngine
    End If
    If objFSO Is Nothing Then
        Set objFSO = CreateObject("Scripting.FileSystemObject")
    End If
   
    Set objFolder = objFSO.GetFolder(FolderPath)
   
    ' Compact all the files in this folder.
   
    For Each objFile In objFolder.Files
   
        If objFile.Name Like "*.accdb" _
        Or objFile.Name Like "*.mdb" _
        Then
       
            strSourceFile = objFile.path
            strFullName = objFile.Name
            I = InStrRev(strFullName, ".")
            strFileName = Left$(strFullName, I - 1)
            strFileExt = Mid$(strFullName, I)
'            Debug.Print strSourceFile, strFullName, strFileName, strFileExt
           
            Select Case strFileExt
                Case ".mdb":  strLockExt = ".ldb"
                Case ".accdb": strLockExt = ".laccdb"
            End Select
           
            If Not objFSO.FileExists(objFolder.path & "\" & strFileName & strLockExt) Then
                strBackupFile = objFolder.path & "\" & strFileName & "_Backup" & Format(Date, "yyyymmdd") & strFileExt
                strCompactFile = objFolder.path & "\" & strFileName & "_COMPACTED" & strFileExt
'                Debug.Print strBackupFile, strCompactFile

                ' Delete the backup file, if it exists.
                If objFSO.FileExists(strBackupFile) Then
                    objFSO.DeleteFile strBackupFile
                End If

                ' Delete the compact file, if it exists.
                If objFSO.FileExists(strCompactFile) Then
                    objFSO.DeleteFile strCompactFile
                End If
       
                ' Make a backup copy of the source file.
                objFSO.CopyFile strSourceFile, strBackupFile
               
                ' Compact the source file into the output compact file.
                objEngine.CompactDatabase strSourceFile, strCompactFile
               
                ' If no error was raised, we'll proceed here.
                ' Delete the source file and rename the compacted file to
                ' the source file name.
                objFSO.DeleteFile strSourceFile
                objFSO.MoveFile strCompactFile, strSourceFile
       
            End If
       
        End If
   
    Next objFile
   
    ' If we are to process subfolders, recursively call this procedure
    ' with each subfolder.
   
    If IncludeSubfolders Then
   
        For Each objSubFolder In objFolder.SubFolders
       
            CompactAllDatabases objSubFolder.path, True, objFSO, objEngine
           
        Next objSubFolder
       
    End If
           
Exit_Point:
    Exit Sub
   
Err_Handler:
    MsgBox Err.Description, vbExclamation, "Error " & Err.Number
    Resume Exit_Point
       
End Sub
'------ end of code ------
