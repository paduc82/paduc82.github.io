Function FindDuplicates(ByVal Where As Range, _
    Optional ByVal Columns, _
    Optional ByVal Header As XlYesNoGuess = xlNo, _
    Optional ByVal Count As Long = 2, _
    Optional ByVal ExcludeFirst As Boolean = True, _
    Optional ByVal Compare As VbCompareMethod = vbBinaryCompare) As Range
  'Return a range with min. count duplicate rows of Where
  'Works similar as Range.RemoveDuplicates
  '  Columns could be a single value or an array of values with:
  '    If Header = xlYes Then
  '      a heading: must be somewhere in the first row
  '    Else
  '      a column number: 1 to Where.Columns.Count
  '      a column name: "A" to "IV" resp. "XFD" in XL2007 and above
  '  If ExcludeFirst then the first occurrence is not stored in the results
  Dim This As Range
  Dim Data, ThisID
  Dim Key As String
  'From FastUnion:
  Dim Stack As Object, TempStack As Object 'Scripting.Dictionary
  Dim Temp() As Variant
  Dim i As Long, j As Long, cl As Long, rw As Long

  'Step 1: Initialize
  'Set FindDuplicates = New Collection
  Set Where = Intersect(Where.Parent.UsedRange, Where)
  If Where Is Nothing Then Exit Function
#If Mac Then
  Set Stack = New Dictionary
  Set TempStack = New Dictionary
#Else
  Set Stack = CreateObject("Scripting.Dictionary")
  Set TempStack = CreateObject("Scripting.Dictionary")
#End If
  Stack.CompareMode = Compare

  If Header = xlGuess Then
    Err.Raise 17, "FindDuplicates", "Header=xlGuess is not supported"
  End If
  If Where Is Nothing Then
    Err.Raise 91, "FindDuplicates", "Where is nothing"
  End If

  If IsMissing(Columns) Then
    'Compare all columns
    ReDim ThisID(1 To Where.Columns.Count)
    For i = 1 To UBound(ThisID)
      ThisID(i) = i
    Next
  Else
    ThisID = Columns
    If Not IsArray(ThisID) Then
      ReDim ThisID(0 To 0)
      ThisID(0) = Columns
    End If
  End If

  'Read in all data
  Data = Where.Value
  If Not IsArray(Data) Then
    ReDim Data(1 To 1, 1 To 1)
    Data(1, 1) = Where.Value
  End If
  If Header = xlYes And UBound(Data) < 2 Then
    Err.Raise 17, "FindDuplicates", "Where has not enough rows"
  End If

  ReDim Index(LBound(ThisID) To UBound(ThisID))
  ReDim DataKey(LBound(ThisID) To UBound(ThisID))
  If Header = xlYes Then
    'Parse the header
    For j = LBound(ThisID) To UBound(ThisID)
      For cl = 1 To UBound(Data, 2)
        If StrComp(ThisID(j), CStr(Data(1, cl)), Compare) = 0 Then
          Index(j) = cl
          Exit For
        End If
      Next
      If Index(j) = 0 Then
        Err.Raise 17, "FindDuplicates", _
          "Columns: Header " & ThisID(j) & " not found"
      End If
    Next
  Else
    'Parse the columns
    On Error Resume Next
    For j = LBound(ThisID) To UBound(ThisID)
      If IsNumeric(ThisID(j)) Then
        Index(j) = ThisID(j)
      Else
        Set This = Intersect(Where.Parent.Columns(ThisID(j)), Where)
        If Not This Is Nothing Then
          Index(j) = This.Column - Where.Column + 1
        End If
      End If
      If Index(j) < 1 Or Index(j) > UBound(Data, 2) Then
        On Error GoTo 0
        Err.Raise 17, "FindDuplicates", _
          "Columns: Column " & ThisID(j) & " not inside Where"
      End If
    Next
    On Error GoTo 0
  End If

  'Step 2: Parse the data
  For rw = 1 To UBound(Data)
    For j = LBound(Index) To UBound(Index)
      DataKey(j) = CStr(Data(rw, Index(j)))
    Next
    Key = Join(DataKey, vbNullChar)
    If Not Stack.Exists(Key) Then
#If Mac Then
      Stack.Add Key, New Dictionary
#Else
      Stack.Add Key, CreateObject("Scripting.Dictionary")
#End If
    End If
    With Stack(Key)
      .Add .Count, Where.Rows(rw)
    End With
  Next

  'Step 3: Build the result
  If Stack.Count = 0 Then Exit Function
  Data = Stack.Items
  For rw = 0 To UBound(Data)
    If Data(rw).Count >= Count Then
      If ExcludeFirst Then Data(rw).Remove 0
      Temp = Data(rw).Items
      'Combine each fragment with the next one
      j = 1
      Do
        For i = 0 To UBound(Temp) - j Step j * 2
          Set Temp(i) = Union(Temp(i), Temp(i + j))
        Next
        j = j * 2
      Loop Until j > UBound(Temp)
      'At this point we have all cells in the first fragment
      TempStack.Add TempStack.Count, Temp(0)
    End If
  Next
  If TempStack.Count = 0 Then Exit Function
  Temp = TempStack.Items
  j = 1
  Do
    For i = 0 To UBound(Temp) - j Step j * 2
      Set Temp(i) = Union(Temp(i), Temp(i + j))
    Next
    j = j * 2
  Loop Until j > UBound(Temp)
  Set FindDuplicates = Temp(0)
End Function
