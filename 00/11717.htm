    Dim MyFiles() As String
    Dim SourceRcount As Long, FNum As Long
    Dim mybook As Workbook, BaseWks As Worksheet
    Dim sourceRange As Range, destrange As Range
    Dim rnum As Long, CalcMode As Long

    ' Change this to the path\folder location of your files.
    MyPath = "\\ABW2\ROOTFS2\MilnerS952\CON-MA EMSG Senior Supervisor\Rejection database new\New Folder"

        MsgBox "No files found"
        Exit Sub
    End If

    ' Fill the myFiles array with the list of Excel files

    Set BaseWks = Workbooks.Add(xlWBATWorksheet).Worksheets(1)
    rnum = 1

    If FNum > 0 Then

            Set mybook = Nothing

            On Error GoTo 0

            If Not mybook Is Nothing Then
                On Error Resume Next

                ' Change this range to fit your own needs.
                With mybook.Worksheets(1)
                    Set sourceRange = .Range("B2:N10000")
                End With

                If Err.Number > 0 Then
                    Err.Clear
                    Set sourceRange = Nothing

                    ' If source range uses all columns then
                    ' skip this file.
                    If sourceRange.Columns.Count >= BaseWks.Columns.Count Then
                        Set sourceRange = Nothing
                    End If
                End If
                On Error GoTo 0

                If Not sourceRange Is Nothing Then

                    SourceRcount = sourceRange.Rows.Count

                    If rnum + SourceRcount >= BaseWks.Rows.Count Then
                        MsgBox "There are not enough rows in the target worksheet."
                        BaseWks.Columns.AutoFit
                        mybook.Close savechanges:=False

                        With sourceRange
                            BaseWks.Cells(rnum, "A"). _

Dim PrintersArray As Variant
Public Const serverPathforPDFs As String = "Your\Path\to\Store\PDFs\Goes\Here"

    initPrintOptionsForm

Public Sub initPrintOptionsForm()
    'find all printers connected to the computer
    Dim objWS As Object
    Dim oPrinters As Object
    Dim i As Long
    Dim tempArray As Variant
    Dim arrayLength As Integer
    Dim printersNames As Variant
    
    Set objWS = CreateObject("WScript.Network")
    Set oPrinters = objWS.EnumPrinterConnections
    ReDim PrintersArray(0 To oPrinters.Count \ 2 - 1)
    For i = 0 To UBound(PrintersArray)
        PrintersArray(i) = oPrinters.Item(i * 2 + 1)
    Next i
    'now create an array where only the names of the printer (without server information) are stored
    ReDim printersNames(0 To UBound(PrintersArray))
    For i = 0 To UBound(PrintersArray)
        tempArray = Split(PrintersArray(i), "\")
        arrayLength = UBound(tempArray)
        printersNames(i) = tempArray(arrayLength)
    Next i
    'now sort the array with all printers
    printersNames = BubbleSort(printersNames)
    'now add the printer name to the dropDown Combobox
    For i = 0 To UBound(PrintersArray)
        PrintOptionsForm.PrinterList.AddItem (printersNames(i))
    Next i
    'set a desired printer as Default Printer
    For i = 0 To (PrintOptionsForm.PrinterList.ListCount - 1)
        If PrintOptionsForm.PrinterList.List(i, 0) = "DESIRED-PRINTER-NAME" Then
            PrintOptionsForm.PrinterList.Value = "DESIRED-PRINTER-NAME"
            Exit For
        End If
    Next i
    'now show the PrintOptionsForm
    PrintOptionsForm.Show vbModal
End Sub

'function Bubble Sort to sort an array, returns the sorted array
Function BubbleSort(ByRef strArray As Variant) As Variant()
    Dim z       As Long
    Dim i       As Long
    Dim strWert As Variant
    
    For z = UBound(strArray) - 1 To LBound(strArray) Step -1
        For i = LBound(strArray) To z
            If LCase(strArray(i)) > LCase(strArray(i + 1)) Then
                strWert = strArray(i)
                strArray(i) = strArray(i + 1)
                strArray(i + 1) = strWert
            End If
        Next i
    Next z
    BubbleSort = strArray
End Function

Public Sub printPresentation(NumberCopies As Integer, PrintHidden As Boolean, PrinterName As String, openPDFPath As Boolean)
    Dim PrinterFound As Boolean
    Dim PrinterPath As String
    Dim i As Integer
    Dim FindPrinterName As String
   
    'create the pattern to search for the printer that has been selected
    FindPrinterName = "*" + PrinterName + "*"
    For i = 0 To UBound(PrintersArray)
        'search for the printer in the original printer array where also the server information for the printer is stored
        PrinterFound = PrintersArray(i) Like FindPrinterName
        If PrinterFound = True Then
            'if the printer was found save the printer path
            PrinterPath = PrintersArray(i)
            Exit For
        End If
    Next
    'now set all printOptions
    With ActivePresentation.PrintOptions
        .OutputType = ppPrintOutputNotesPages 'always as notespages
        .NumberOfCopies = NumberCopies 'number of copies set by user
        .Collate = msoTrue
        .FrameSlides = msoFalse
        .PrintColorType = ppPrintColor
        .PrintComments = msoFalse
        .printHiddenSlides = Switch(PrintHidden = True, msoTrue, PrintHidden = False, msoFalse) 'hidden slides set by user
        .PrintInBackground = msoTrue
        .RangeType = ppPrintAll
        .ActivePrinter = PrinterPath 'printer selected by user
    End With
    'now start the printing
    ActivePresentation.PrintOut
    createPDF (openPDFPath)
End Sub

'this function prepares the pdf so that the slides on the notes pages are substituted by SlideObjects so they are searchable in the PDF
Public Sub preparePDF()
    Dim lOriginalView As Long
    Dim oSl As Slide
    Dim oSh As Shape
    Dim old_placeholder As Shape
    Dim oNewSh As Shape
    Dim slideAlreadyLinked As Boolean
    Dim OriginalSlide As Integer
    Dim startTime As Long
   
    ' Store user's original view
    lOriginalView = ActiveWindow.ViewType
    OriginalSlide = ActiveWindow.View.Slide.SlideNumber
   
    'show a status message
    StatusMessage.Show vbModeless
    'status message is without content if there is no short break
    startTime = Timer
    Do While Timer < startTime + 0.01 'break is 0.01 seconds, which is enough
        DoEvents    'give control to other processes
    Loop
   
    ' Change to notespage view
    ActiveWindow.ViewType = ppViewNotesPage
   
    'for each slide do the following
    For Each oSl In ActivePresentation.Slides
        'check whether the slide image has already been substituted by a slide link
        For Each oSh In oSl.NotesPage(1).Shapes
            If Len(oSh.Tags("NEWNOTESPLACEHOLDER")) > 0 Then
                Set old_placeholder = oSh
                slideAlreadyLinked = True
                Exit For
            End If
        Next
       
        'if the slides have not yet been substituted, search for the slide image on the notes page
        If old_placeholder Is Nothing Then
            slideAlreadyLinked = False
            For Each oSh In oSl.NotesPage(1).Shapes
                If oSh.Type = msoPlaceholder Then
                    If oSh.PlaceholderFormat.Type = ppPlaceholderTitle Then
                        Set old_placeholder = oSh
                        Exit For
                    End If
                End If
            Next
        End If
       
        'if there is still nothing in the placeholder then there is no slide image on the page
        'in that case do nothing
        If old_placeholder Is Nothing Then
            'do nothing
        'in all other cases, substitute the image or update the slide link
        Else
            'if the slide was already substituted, it should only be updated
            If slideAlreadyLinked = True Then
                old_placeholder.LinkFormat.Update
            'else the slide needs to be copied and the old image deleted
            Else
                'copy the slide
                oSl.Copy
                With ActiveWindow
                    'go to the slide
                    .View.GotoSlide (oSl.SlideIndex)
                    'paste the copied version as linked slide
                    .View.PasteSpecial ppPasteOLEObject, msoFalse, "null", 0, "null", msoTrue
                    With .Selection.ShapeRange(1)
                        'set the x and y position
                        .Left = old_placeholder.Left
                        .Top = old_placeholder.Top
                        'set width and height
                        .Width = old_placeholder.Width
                        .Height = old_placeholder.Height
                        'set a tag to identify the linked slide for the next run
                        .Tags.Add "NEWNOTESPLACEHOLDER", "YES"
                        'draw a line around the slide

'this function saves the presentation as PDF after the notes pages have been updated
'the pdf uses the same name as the name of the current ppt file
'the folder in which the PDF is stored is generated from the folder the ppt is currently saved
'the path is split from the folder "YOURFOLDERHERE"
Public Sub saveAsPDF(openPDFPath As Boolean)
    Dim PresentationName As String
    Dim PresentationNameArray As Variant
    Dim SaveName As String
    Dim PresentationPath As String
    Dim PresentationPathArray As Variant
    Dim SavePath As String
    Dim DirectoryExists As String
    Dim subfolders As Variant
    Dim tempDir As String
    Dim i As Integer
   
    'get the name of the ppt
    PresentationName = ActivePresentation.Name
    PresentationNameArray = Split(PresentationName, ".ppt")
    'create the pdf name
    SaveName = PresentationNameArray(0) + ".pdf"
    'get the save path of the ppt
    PresentationPath = ActivePresentation.Path
    'split the path from the YOURFOLDERHERE directory
    PresentationPathArray = Split(PresentationPath, "\YOURFOLDERHERE\")
    'create the save path for the pdf file
    If PresentationPath = "" Then
        PresentationPathArray = Array(0, "NotSavedDocument") 'For presentations that have not been saved yet
    End If
    SavePath = serverPathforPDFs + "\PDFs\" + PresentationPathArray(1) + "\" + SaveName
    'check whether the directory "PDFs" already exists
    DirectoryExists = Dir(serverPathforPDFs + "\PDFs", vbDirectory)
    'if not create the directory
    If DirectoryExists = "" Then
        MkDir (serverPathforPDFs + "\PDFs")
    End If
    'now check whether the subfolder exists and create them if needed
    subfolders = Split(PresentationPathArray(1), "\")
    tempDir = serverPathforPDFs + "\PDFs"
    For i = 0 To UBound(subfolders)
        tempDir = tempDir + "\"
        tempDir = tempDir & subfolders(i)
        DirectoryExists = Dir(tempDir, vbDirectory)
        If DirectoryExists = "" Then
            MkDir (tempDir)
        End If
    Next
    'now check if the save path exists (just to make sure no error occurs)
    DirectoryExists = Dir(serverPathforPDFs + "\PDFs\" + PresentationPathArray(1), vbDirectory)
    If DirectoryExists <> "" Then
        'if the path exists save the file as pdf
        ActivePresentation.ExportAsFixedFormat SavePath, ppFixedFormatTypePDF, ppFixedFormatIntentPrint, msoFalse, ppPrintHandoutHorizontalFirst, ppPrintOutputNotesPages, msoFalse, , ppPrintAll, , False, False, True, True, False
    End If
    'last open the path to the PDFs if the user clicked the checkbox in the form
    If openPDFPath = True Then
        openHyperlinktoPDFs
    End If
End Sub
