The Intersect is the area that is to be considered for the X's From what you describe above, this would be N42:S245 with Qs in J42:J245 and Ss in N1:S1. Try,

Private Sub Worksheet_BeforeDoubleClick(ByVal Target As Range, Cancel As Boolean)
    If Not Intersect(Target, Range("N42:S245")) Is Nothing Then
        If Cells(Target.Row, "J").Value = "Q" And Cells(1, Target.Column).Value = "C" Then
            If Target = "X" Then
                Target.ClearContents
            Else

            End If
        End If
        Cancel = True
        Target.Activate
    End If
End Sub

Private Sub Worksheet_BeforeDoubleClick(ByVal Target As Range, Cancel As Boolean)
    If Not Intersect(Target, Range("N42:S245")) Is Nothing Then
        If Cells(Target.Row, "J").Value = "Q" And Cells(1, Target.Column).Value = "C" _
          And Target.Font.ColorIndex <> 15 Then
            If Target = "X" Then
                Target.ClearContents
            Else
                Target.Value = "X"
            End If
        End If
        Cancel = True
        Target.Activate
    End If
End Sub

Tap Alt+Q to return to your worksheet. Remember to double-click the cell to get an X, not single-click it.
