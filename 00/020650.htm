Sub Selection_Range_As_Pic_On_Desktop()
Dim ws As Worksheet
Dim sPath As String
sPath = Environ("USERPROFILE") & "\Desktop\"
Dim rng As Range
Set rng = Selection
If MsgBox("select á range ??", vbOKCancel) = vbCancel Then Exit Sub

If sName = "" Then Exit Sub '<< new line

sPath = sPath & sName & ".jpg"
Application.ScreenUpdating = False
Set ws = Worksheets.Add
Charts.Add
ActiveChart.Location Where:=xlLocationAsObject, Name:=ws.Name
With ActiveChart.Parent
.Height = rng.Height
.Width = rng.Width
End With
rng.CopyPicture xlScreen, xlPicture
ActiveChart.Paste
ActiveChart.Export Filename:=sPath, FilterName:="jpg"
Application.DisplayAlerts = False
ws.Delete
Application.DisplayAlerts = True
Application.ScreenUpdating = True
End Sub
