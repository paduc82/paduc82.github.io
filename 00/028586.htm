Sub ExportNotesToText()

   Const outFile As String = "C:\Temp\Outlook notes.csv"

  

   Dim notesFolder As Folder

   Dim itemNo As Long

   Dim thisNote As NoteItem

   Dim fileNo As Integer

   Dim nSubject As String

   Dim nText As String

  

   fileNo = freefile

   Open outFile For Output As #fileNo

   Print #fileNo, "Subject,Text"

  

   Set notesFolder = Application.GetNamespace("MAPI").PickFolder

   For itemNo = 1 To notesFolder.Items.Count

       Set thisNote = notesFolder.Items(itemNo)

       nSubject = """" & Replace(thisNote.Subject, """", "'") & """"

       nText = Replace(thisNote.Body, """", "'")

       If InStr(1, nText, vbNewLine) > 0 Then nText = Mid(nText, InStr(1, nText, vbNewLine) + 2)

       nText = """" & nText & """"

       Print #fileNo, nSubject & "," & nText

   Next

  

   Set thisNote = Nothing

   Set notesFolder = Nothing

   Close #fileNo

  

   MsgBox "Notes have been saved in" & vbNewLine & outFile, vbOKOnly + vbInformation, "Export Notes"

End Sub

questions911 say:

How to convert data from a row to one cell

How can I convert a row of data to one cell and add "or" between the data. This is so I can add it to a query in Access.

Example:

AAER1235

CCRT0365

DFYT9872

MFLO2089

 

I want it to show as:

AAER1235 or CCRT0365 or DFYT9872 or MFLO2089

jeeped say:
To answer your original question, you can string the values together with an ampersand or use the CONCATENATE() function,

=A1&" OR "&A2&" OR "&A3&" OR "&A4
... OR,
=CONCATENATE(A1," OR ",A2," OR ",A3," OR ",A4)
... which produces,
AAER1235 OR CCRT0365 OR DFYT9872 OR MFLO2089

If this is intended for an SQL statement (Access or otherwise) you would be better served to use the IN operator as in,

WHERE myfield IN ("AAER1235","CCRT0365 ","CCRT0365","MFLO2089")

To accomplish that the formula changes to something like,

="IN ('"&A1&"','"&A2&"','"&A3&"','"&A4&"')"

... for a result like,

IN ('AAER1235','CCRT0365','DFYT9872','MFLO2089')
