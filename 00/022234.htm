hansv mvp say:

A modified version:

Sub Copy()
    Dim i As Long
    Dim wshS As Worksheet
    Dim s As Long
    Dim m As Long
    Dim wshT As Worksheet
    Dim t As Long
    Application.ScreenUpdating = False
    Set wshT = Worksheets("TEST SHEET")
    t = wshT.Range("B" & wshT.Rows.Count).End(xlUp).Row
    For i = 1 To 3
        Set wshS = Worksheets(i)
        m = wshS.Range("B" & wshS.Rows.Count).End(xlUp).Row
        For s = 1 To m
            If IsNumeric(wshS.Range("B" & s).Value) And _
                    wshS.Range("B" & s).Value >= 1 Then
                t = t + 1
                wshS.Range("B" & s & ":Q" & s).Copy _
                    Destination:=wshT.Range("B" & t)
            End If
        Next s
    Next i
    Application.ScreenUpdating = True
End Sub
