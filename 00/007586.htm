mike h.. say:

Hi,

 

here's another. It will loop through every sheet in the workbook and if the sheet name is in the string variable S then the data will be copied to the next available row in sheet 1. If you want to copy every sheet to sheet 1 then use the second bit of code

 

Sub Copy_Over()
Dim v As Variant, dst As Worksheet
Dim s As String, LastRow As Long
Dim Sh As Worksheet, lr As Long
'Change to suit, this is the sheet we're copying to
Set dst = Sheets("Sheet1")
'These are the sheets we're copying from so rename/add as required
s = "Sheet2,Sheet3,Sheet4"
v = Split(s, ",")
For Each Sh In ThisWorkbook.Worksheets
    If Not IsError(Application.Match(CStr(Sh.Name), v, 0)) Then
    LastRow = dst.Cells(Cells.Rows.Count, "A").End(xlUp).Row
    lr = Sh.Cells(Cells.Rows.Count, "A").End(xlUp).Row
    Sh.Rows("1:" & lr).Copy dst.Cells(LastRow + 1, 1)
    End If
Next Sh
End Sub

 

Sub Copy_Over1()
Dim dst As Worksheet
Dim LastRow As Long
Dim Sh As Worksheet, lr As Long
'Change to suit, this is the sheet we're copying to
Set dst = Sheets("Sheet1")
For Each Sh In ThisWorkbook.Worksheets
    If Sh.Name <> dst.Name Then
    LastRow = dst.Cells(Cells.Rows.Count, "A").End(xlUp).Row
    lr = Sh.Cells(Cells.Rows.Count, "A").End(xlUp).Row
    Sh.Rows("1:" & lr).Copy dst.Cells(LastRow + 1, 1)
    End If
Next Sh
End Sub
