   Dim vaRecipient As Variant
   Dim rnBody As Range
   Dim Data As DataObject
 
   Const stSubject As String = "YOUR ACTION ITEMS - UPDATE REQUIRED"
   Const stMsg As String = "Data as part of the e-mail's body."
   Const stPrompt As String = "Please select the range:"
 
   'This is one technique to send an e-mail to many recipients but for larger
   'number of recipients it's more convenient to read the recipient-list from
   'a range in the workbook.
   vaRecipient = VBA.Array("email@email")
 
   On Error Resume Next
   Set rnBody = Application.InputBox(Prompt:=stPrompt, _
         Default:=Selection.Address, Type:=8)
 
   'The user canceled the operation.
   If rnBody Is Nothing Then Exit Sub
 
   On Error GoTo 0

   Set noSession = CreateObject("Notes.NotesSession")
   Set noDatabase = noSession.GETDATABASE("", "")
 
   'Make sure Lotus Notes is open and available.
   If noDatabase.IsOpen = False Then noDatabase.OPENMAIL
 
   'Create the document for the e-mail.
   Set noDocument = noDatabase.CreateDocument
 
   'Copy the selected range into memory.
   rnBody.Copy
 
   'Retrieve the data from then copied range.
   Set Data = New DataObject
   Data.GetFromClipboard
 
   'Add data to the mainproperties of the e-mail's document.
   With noDocument
      .Form = "Memo"
      .SendTo = vaRecipient
      .Subject = stSubject
      'Retrieve the data from the clipboard.
      .Body = Data.GetText & " " & stMsg
      .SaveMessageOnSend = True
   End With
 
   'Send the e-mail.
   With noDocument
      .PostedDate = Now()
      .Send 0, vaRecipient
   End With
 
   'Release objects from memory.
   Set noDocument = Nothing
   Set noDatabase = Nothing
   Set noSession = Nothing
 
 
   'Activate Excel for the user.
   AppActivate "Microsoft Excel"
 
   'Empty the clipboard.
   Application.CutCopyMode = False
 
   MsgBox "The e-mail has successfully been created and distributed.", vbInformation
 
End Sub

Sub Send_Unformatted_Rangedata()
   Dim noSession As Object, noDatabase As Object, noDocument As Object
   Dim rnBody As Range, rw As Range
   Dim c As Range
   Dim strAddrList As String
   Dim strMsg As String, strSig As String
  
   Const stSubject As String = "YOUR ACTION ITEMS - UPDATE REQUIRED"
   Const stPrompt As String = "Please select the range:"
   strMsg = "Here is the data you requested: " & Chr(10) & Chr(10)
   strSig = "Best regards, " & Chr(10) & "Your name"
  
   'Load addresses from column L of the selected rows, separated by a comma.
   For Each rw In Selection.Rows
        strAddrList = strAddrList & Range("L" & rw.Row).Value & ","
   Next rw

    'Remove any extra commas:
   If Right(strAddrList, 1) = "," Then strAddrList = Left(strAddrList, Len(strAddrList) - 1)
   While InStr(1, strAddrList, ",,", vbTextCompare) > 0
      strAddrList = Replace(strAddrList, ",,", ",", , , vbTextCompare)
   Wend
   Debug.Print strAddrList

   On Error Resume Next
   Set rnBody = Application.InputBox(Prompt:=stPrompt, _
         Default:=Selection.Address, Type:=8)
   On Error GoTo 0
   If rnBody Is Nothing Then Exit Sub 'The user canceled the operation.
  
   For Each c In rnBody
        strMsg = strMsg & c.Value & Chr(10)
   Next c
   Debug.Print strMsg

    'Instantiate Lotus Notes COM's objects.
   Set noSession = CreateObject("Notes.NotesSession")
   Set noDatabase = noSession.GETDATABASE("", "")

   'Make sure Lotus Notes is open and available.
   If noDatabase.IsOpen = False Then noDatabase.OPENMAIL

   'Create the document for the e-mail.
   Set noDocument = noDatabase.CreateDocument
  
   'Add data to the mainproperties of the e-mail's document.
   With noDocument
      .Form = "Memo"
      .SendTo = strAddrList
      .Subject = stSubject
      .Body = strMsg & Chr(10) & Chr(10) & strSig
      .SaveMessageOnSend = True
   End With

   'Send the e-mail.
   With noDocument
      .PostedDate = Now()
      .Send 0, strAddrList
   End With

   'Release objects from memory.
   Set noDocument = Nothing
   Set noDatabase = Nothing
   Set noSession = Nothing

   'Activate Excel for the user.
   AppActivate "Microsoft Excel"

   MsgBox "The e-mail has successfully been created and distributed.", vbInformation
End Sub
