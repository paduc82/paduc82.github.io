ronrosenfeld say: If that's the only change, it should just be a matter of adding Months instead of Years to GivenDate:

===========================================
Option Explicit
Function Dt(CompDate As Date, GivenDate As Date, _
            NumMonths As Long) As Date
Dim GivenDatePlusMonths As Date
Dim CompDatePlusYearsPlus1 As Date
Dim CompDatePlusYearsPlus7 As Date
Dim CompDatePlusYearsPlus13 As Date
Dim CompDateBase As Date

GivenDatePlusMonths = DateAdd("m", NumMonths, GivenDate)
CompDateBase = IIf(GivenDatePlusMonths >= CompDate, _
    GivenDatePlusMonths, CompDate)

CompDatePlusYearsPlus1 = DateSerial(Year(CompDateBase), _
    Month(CompDate) + IIf(Day(CompDate) = 1, 0, 1), 1)
CompDatePlusYearsPlus7 = DateAdd("m", 6, CompDatePlusYearsPlus1)
CompDatePlusYearsPlus13 = DateAdd("yyyy", 1, CompDatePlusYearsPlus1)

Select Case GivenDatePlusMonths
    Case Is <= CompDatePlusYearsPlus1
        Dt = CompDatePlusYearsPlus1
    Case Is <= CompDatePlusYearsPlus7
        Dt = CompDatePlusYearsPlus7
    Case Else
        Dt = CompDatePlusYearsPlus13
End Select

End Function
