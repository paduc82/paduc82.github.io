Here's some sample code I scavanged from the web that goes through all the ".docm" files in a specific folder and lists all the VB components in each file.  It might provide a start for what you want to do.  You have to set references to:

 

Microsoft Scripting Runtime

Microsoft Visual Basic for Applications Extensibility 5.x

 

You also have to go to Word Options, Trust Center, Trust Center Settings..., Macro Settings and check the "Trust access to the VBA project object model" checkbox.

 

HTH,

 

Eric

 

'===== BEGIN CODE =====

Option Explicit

Sub Test()
  ProcessDirectory "C:\Users\gundeej\Desktop\VBA Bits"
End Sub

Function GetWordDocument(FileName As String) As Word.Document
    On Error Resume Next

    Set GetWordDocument = Application.Documents.Open(FileName)
    If Err.Number = &H80010105 Then
      Application.Documents(FileName).Close SaveChanges:=False
      On Error GoTo 0
      Set GetWordDocument = Application.Documents.Open(FileName)
    End If
End Function

Sub ProcessDirectory(PathName As String)
  Dim fso As New FileSystemObject
  Dim objFile As File
  Dim objFolder As Folder
  Dim objWordDoc As Object

  On Error GoTo Err_Handler

  Set objFolder = fso.GetFolder(PathName)
  For Each objFile In objFolder.Files
    If (Left(objFile.Name, 1) <> "~" And _
        objFile.Name <> ThisDocument.Name And _
        StrComp(Right(objFile.Name, 5), ".docm", vbTextCompare) = 0) Then
      Set objWordDoc = GetWordDocument(objFile.Path)
      ProcessDocument objWordDoc
      objWordDoc.Close 0, 1
      Set objWordDoc = Nothing
    End If
  Next

Exit_Handler:
  Exit Sub

Err_Handler:
  MsgBox "Error " & Err.Number & ": " & Err.Description
  Resume Exit_Handler
  'Resume Next ' or as above
End Sub

Sub ProcessDocument(objWordDoc As Document)
    Dim i As Long
    Dim theVBP As VBProject
'
    Set theVBP = objWordDoc.VBProject
    If (theVBP Is Nothing) Then Exit Sub
    With theVBP
Debug.Print "Processing File " & objWordDoc.FullName
Debug.Print "    This file has " & .VBComponents.Count & " components"
Debug.Print "    Their names are:"
        For i = 1 To .VBComponents.Count
            Debug.Print "        " & .VBComponents(i).Name
        Next i
    End With
End Sub
