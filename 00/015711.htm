Private Sub cboMyCombo_DblClick(Cancel As Integer)

Dim db As Database

Dim strSQL As String

Dim strDetail As String

strDetail = Me.cboMyCombo.Column(1)

If MsgBox("Are you sure your want to delete " & strDetail, vbQuestion + vbYesNo, "R U Possitve") = vbYes Then

Set db = CurrentDb()

db.Execute strSQL, dbFailOnError

Set db = Nothing

End If

End Sub

 

Private Sub cboMyCombo_NotInList(NewData As String, Response As Integer)

Dim strAdding As String

strAdding = "Do you want to add '" & NewData & "' as a new record?"

If MsgBox(strAdding, vbYesNo + vbDefaultButton2 + vbQuestion, "This record is not on file") = vbYes Then

strAdding = "INSERT INTO MyTable ( TableFieldName ) " & "SELECT """ & NewData & """ AS TableFieldName;"

DBEngine(0)(0).Execute strAdding, dbFailOnError

Response = acDataErrAdded

End If

End Sub
