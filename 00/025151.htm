I am assuming that you will have a hidden worksheet with a list something like the following with the User Names and Passwords and that you want the last login date and time as shown. (Depending on your regional location, you might need to edit the date format in the code.)

Next I assume that you will have a Userform something like the following. Note that if you use the PasswordChar property then the dot is created by positioning the curser in the property field and then hold the Alt key and type 0149 on the numeric keypad. (Must be numeric keypad; does not work on the other numeric keys)

The code for the command button is as follows. Note the comments and where you might need to edit the code.

Private Sub CommandButton1_Click()
    Dim wsUserIds As Worksheet
    Dim rngUser As Range
   
    If Me.txtUserId.Value = "" Or Me.txtPassword.Value = "" Then
        MsgBox "Enter User Name and Password"
        Exit Sub
    End If
   
    Set wsUserIds = Worksheets("Sheet4")    'Edit "Sheet4" to your worksheet name
   
    'wsUserIds.Unprotect Password:="ossie"   'Optional. Use if worksheet protected and edit password.
    
    With wsUserIds.Columns("A:A")
        Set rngUser = .Find(What:=Me.txtUserId.Value, _
                    LookIn:=xlFormulas, _
                    LookAt:=xlWhole, _
                    SearchOrder:=xlByRows, _
                    SearchDirection:=xlNext, _
                    MatchCase:=True)
                   
        If Not rngUser Is Nothing Then
            If rngUser.Offset(0, 1) = Me.txtPassword.Value Then
                rngUser.Offset(0, 2).Value = Now()
                rngUser.Offset(0, 2).NumberFormat = "dd mmmm yyyy hh:mm AM/PM"   'Edit date format as required
            Else
                MsgBox "User name or password incorrect."
                Me.txtPassword.Value = ""   'Only need to delete password

                Exit Sub      'Added code

            End If
        Else
            MsgBox "User name or password incorrect."
            Me.txtPassword.Value = ""   'Only need to delete password

            Exit Sub          'Added code

        End If
    End With
   
    'wsUserIds.Protect Password:="ossie"   'Optional
   

     Me.Hide        'Do not use Unload or the next line might not run
    UserForm2.Show      'Edit Userform2 to your Userform name

End Sub
