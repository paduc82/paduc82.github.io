Sub ReplaceHeaders()

    Dim ws As Worksheet
   
    For Each ws In Worksheets
        If ws.PageSetup.CenterHeader <> "" Then

        End If
    Next ws

 ActiveSheet.PageSetup.CenterHeader = Replace(ActiveSheet.PageSetup.CenterHeader, "Q3 2013", "Q4 2013")
