Fiinds one instance, PER YOUR EXAMPLE, of each match and colors yellow.

Option Explicit
Sub HighlightIfCompanyandAmountAgree()
Dim lr As Long
Dim c As Range
Dim mf As Range
lr = Cells(Rows.Count, 1).End(xlUp).Row
For Each c In Range("a2:a" & lr)

Set mf = Columns("C").find(What:=c, LookIn:=xlValues, _
        LookAt:=xlPart, SearchOrder:=xlByRows, _
        SearchDirection:=xlNext, _
        MatchCase:=False, SearchFormat:=False)
    If Not mf Is Nothing Then
    If c.Offset(, 1) = mf.Offset(, 1) Then
    c.Resize(, 2).Interior.ColorIndex = 6
    mf.Resize(, 2).Interior.ColorIndex = 6
    End If
    End If
    Next c
  
End Sub

hansv mvp say:

Try this:

Sub ExportPdf()
    Dim wb As Workbook
    Dim wsI As Worksheet
    Dim ws As Worksheet
    Dim r As Long
    Dim m As Long
    Dim ruta As String, ruta2 As String, ruta3 As String, ruta4 As String
    Dim nFile As String, sFile As String
 
    nFile = Format(Date, "dd-mmm-yyyy")
    ruta = wb.Path & "\"
    ruta2 = ruta & "export\"
    If Dir(ruta2, vbDirectory) = Empty Then
        MkDir ruta2
    End If
    ruta3 = ruta2 & "PDF\"
    If Dir(ruta3, vbDirectory) = Empty Then
        MkDir ruta3
    End If
    ruta4 = ruta3 & nFile & "\"
        If Dir(ruta4, vbDirectory) = Empty Then
    MkDir ruta4
    End If
 
    ' Use other date format in filename
    nFile = Format(Date, "yyyy-mm-dd")
 
    Set wb = ThisWorkbook
    Set wsI = wb.Worksheets("Index")
    m = wsI.Range("A" & wsI.Rows.Count).End(xlUp).Row
 
    For r = 1 To m
        Set ws = wb.Worksheets(wsI.Range("A" & r).Value)
        sFile = nFile & "_" & ws.Name & ".pdf"
        ws.ExportAsFixedFormat Type:=xlTypePDF, Filename:=ruta4 & sFile
    Next r
End Sub
