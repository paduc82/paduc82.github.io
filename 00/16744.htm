john sr wilson say:

I think the code is trying to apply the plot area setting BEFORE the size has actually changed and the settings are not possible at the smaller size

Try adding a DoEvents loop

Sub chartboardandplotarea_Dimensions()

'PURPOSE: Return the dimensions of a selected shape in PowerPoint and apply to existing charts in deck
'SOURCE: www.TheSpreadsheetGuru.com
'https://www.snip2code.com/Snippet/5174/VBA-Excel-Charts-to-PowerPoint
'link is code to transfer execl charts to ppt deck
Dim l As Single
Dim t As Single
Dim h As Single
Dim w As Single
Dim ll As Single
Dim tt As Single
Dim hh As Single
Dim ww As Single
Dim safety As Long
    With ActiveWindow.Selection.ShapeRange(1)
    If .HasChart = True Then
            
            l = .Left
            t = .Top
            h = .Height
            w = .Width
            ll = .Chart.PlotArea.Left
            tt = .Chart.PlotArea.Top
            hh = .Chart.PlotArea.Height
            ww = .Chart.PlotArea.Width
                     
    Else
            MsgBox "You have not selected an OBJECT in PowerPoint to dimension."
            Exit Sub
    End If
    End With

Dim objSlide As Slide
Dim objShape As Shape
  For Each objSlide In ActivePresentation.Slides
    For Each objShape In objSlide.Shapes
      If objShape.HasChart = True Then
        With objShape
          
          .Left = l
          .Top = t
          .Height = h
          .Width = w
          safety = 0
          Do
          safety = safety + 1
          DoEvents
          If safety = 100 Then ' just in case loop is endless
          Exit Do
          End If
          Loop Until objShape.Width = w
          .Chart.PlotArea.Left = ll
          .Chart.PlotArea.Top = tt
          .Chart.PlotArea.Height = hh
          .Chart.PlotArea.Width = ww

         End With
      End If
    Next objShape
  Next objSlide
End Sub
