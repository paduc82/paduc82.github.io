Option Explicit

#Const HideRows = True
#Const HideColumns = True
#Const UseFormButtonOnly = True

Sub Test_KB2503335()
  'See http://support.microsoft.com/kb/2503335
  Const LAR = "HKEY_CURRENT_USER\Software\Microsoft\Office\14.0\Excel\Options\" & _
    "LegacyAnchorResize"
  Const Space = 2
  Dim FName As String
  Dim R1 As Range, R2 As Range
  Dim O As Object

  'Did the hotfix exist?
  Select Case RegKeyRead(LAR)
    Case ""
      MsgBox "Hotfix not found. Aborted"
      Exit Sub
    Case "1"
      If MsgBox("Hotfix enabled, click ok to continue", vbOKCancel) = vbCancel Then
        Exit Sub
      End If
    Case "0"
      If MsgBox("Hotfix disabled, click ok to continue", vbOKCancel) = vbCancel Then
        Exit Sub
      End If
  End Select

  'Create a new file
  Workbooks.Add

  'A place for the controls
  Set R1 = Range("B2").Resize(5, 2)
  Set R2 = R1(1, 1).Offset(R1.Rows.Count + 2).Resize(2, 2)

  'Create form controls
  With R1
    .BorderAround xlContinuous, xlMedium, 5
#If UseFormButtonOnly Then
    ActiveSheet.Buttons.Add _
      Left:=.Left + Space, Top:=.Top + Space, _
      Width:=.Width - 2 * Space, Height:=.Height - 2 * Space
#Else
    Set O = ActiveSheet.GroupBoxes.Add( _
      Left:=.Left + Space, Top:=.Top + Space, _
      Width:=.Width - 2 * Space, Height:=.Height - 2 * Space)
    Set O = ActiveSheet.OptionButtons.Add( _
      Left:=.Left + 2 * Space, Top:=.Top + 2 * Space, _
      Width:=.Width - 4 * Space, Height:=.Height / 2 - 2 * Space)
    O.Interior.ColorIndex = 3
    Set O = ActiveSheet.OptionButtons.Add( _
      Left:=.Left + 2 * Space, Top:=.Top + .Height / 2 + 2 * Space, _
      Width:=.Width - 4 * Space, Height:=.Height / 2 - 4 * Space)
    O.Interior.ColorIndex = 4
#End If
  End With
  'Create ActiveX controls
  With R2
    .BorderAround xlContinuous, xlMedium, 5
    ActiveSheet.OLEObjects.Add ClassType:="Forms.CommandButton.1", _
      Left:=.Left + Space, Top:=.Top + Space, _
      Width:=.Width - 2 * Space, Height:=.Height - 2 * Space
  End With

  'Hide the columns and rows
#If HideColumns Then
  R1.EntireColumn.Hidden = True
  R2.EntireColumn.Hidden = True
#End If
#If HideRows Then
  R1.EntireRow.Hidden = True
  R2.EntireRow.Hidden = True
#End If

  'Save the file
  Application.DisplayAlerts = False
  ActiveWorkbook.Save
  'Close and reopen the file
  FName = ActiveWorkbook.FullName
  ActiveWorkbook.Close
  Workbooks.Open FName

  If MsgBox("Click ok to unhide rows & columns", vbOKCancel) = vbCancel Then
    Exit Sub
  End If

  'Unhide all columns and rows
  Cells.EntireColumn.Hidden = False
  Cells.EntireRow.Hidden = False
End Sub

Private Function RegKeyRead(RegKey As String) As String
  'Reads the value for the registry key RegKey
  On Error Resume Next
  RegKeyRead = CreateObject("WScript.Shell").RegRead(RegKey)
End Function
