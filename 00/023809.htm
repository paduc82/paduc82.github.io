mike h.. say:

Hi,

 

A non-Looping method that will be very fast

 

Sub Delete_Rows()
Dim LastColumn As Long, LastRow As Long
LastColumn = Cells.Find(What:="*", SearchOrder:=xlByColumns, _
               SearchDirection:=xlPrevious, LookIn:=xlFormulas).Column
LastRow = Cells(Rows.Count, "B").End(xlUp).Row
  r = 1
Application.ScreenUpdating = False
 
  Cells(1, LastColumn + 1).Resize(LastRow).Value = Range("B1:B" & LastRow).Value
With Columns(LastColumn + 1)
        On Error Resume Next
        .SpecialCells(xlCellTypeConstants, xlErrors).ClearContents
        .Replace r, "#N/A", xlWhole
       
        .SpecialCells(xlCellTypeConstants, xlNumbers).EntireRow.Delete
        .Delete
        On Error GoTo 0
    End With
  Application.ScreenUpdating = True
End Sub
