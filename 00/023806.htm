vba macro

Sub Download_Pic()

'### 17-03-2019 ###

On Error Resume Next

Dim sPath

sPath = Environ("USERPROFILE") & "\Desktop\web pictures\"

If Dir(sPath, vbDirectory) = Empty Then MkDir sPath

Dim obj As Object

Dim obj1 As Object

Dim fl() As Byte

Set obj = CreateObject("microsoft.xmlhttp")

Set obj1 = CreateObject("ADODB.Stream")

Dim sURL As String

Dim v As Variant

For Each r In Range("A1:A6")

sURL = r.Value

v = Split(sURL, "/")(UBound(Split(sURL, "/")))

With obj

.Open "GET", sURL, False

.Send

fl = .Responsebody

End With

With obj1

.Open

.Type = 1

.Write fl

.SaveToFile sPath & v, 2

.Close

End With

Next

Set obj = Nothing

Set obj1 = Nothing

MsgBox "done"

On Error GoTo 0

End Sub

myall_blues say: One way:
Put the numbers 1,2,3 in cells D2,D3,D4.
Then in E2, enter the following array formula (enter it using CTRL+SHIFT+ENTER).
=INDEX($B$2:$B$21,SMALL(IF($A$2:$A$21="Seattle",ROW($A$2:$A$21)-ROW(INDEX($A$2:$A$21,1,1))+1),D2))
If you have done it correctly the formula will appear in the formula bar with braces { and } around it.
Then copy the formula into cells D3 and D4.
