bill manville say:

Two things occur to me:

a) that you may have more than 2 cells to merge in which case doing as many as possible in one operation would be faster

b) you are probably triggering a message that you are merging where there is content in the second cell; this message is suppressed by displayalerts being false but you could avoid it by clearing the subsequent cells.

So try this variant:

Sub MergeAnimalType()
 'This macro will scan Animal Type column
 'and merge the types that are the same
 'Animal Type column needs to be sorted
 'before running this macro
 '

 Dim lLast As Long
 Dim lRows As Long
 Dim lStart As Long
 lLast = Cells(Rows.Count, "A").End(xlUp).Row
 lStart = 1
 Do While lStart < lLast
   lRows = 1
   Do While Cells(lStart, 1) = Cells(lStart + lRows, 1)
     lRows = lRows + 1
   Loop
   If lRows > 1 Then
    Cells(lStart + 1, 1).Resize(lRows - 1).ClearContents
    Cells(lStart, 1).Resize(lRows).Merge
    Cells(lStart, 1).VerticalAlignment = xlCenter
   End If
   lStart = lStart + lRows
 Loop
End Sub
