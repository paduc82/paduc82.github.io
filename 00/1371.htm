gord dibben say:

Couple of macros.

One to do page numbers in J3 in a single sheet.

You can change to do all sheets.  I'm off to lunch.

One to set rows to repeat on all sheets.

Sub PrintWithPgNumInTitleRow()
Dim NumPages As Long, pg As Long
    NumPages = ExecuteExcel4Macro("Get.document(50)")
    For pg = 1 To NumPages
        With ActiveSheet
            .Range("J3").Value = pg & " of " & NumPages
            .PrintOut from:=pg, To:=pg
        End With
    Next pg
End Sub

Sub Repeat_Rows_All_Sheets()
    Dim wkSht As Worksheet
    For Each wkSht In ActiveWorkbook.Worksheets
         With wkSht.PageSetup
          .PrintTitleRows = "$1:$7"
          .PrintTitleColumns = ""
         End With
    Next wkSht
End Sub
