andreas killer say:
To create the checkboxes on the chart itself will cause several problems, checkboxes on a userform is a better way.

I did not look at the files that Andy Pope provides in the thread you mention, I guess that solution did not work for you, because you've created this thread, right?

My basic idea is to use hide or show the series itself, each series has a Format object which has a Line object with a Visible property.

But as we did not know how many series are in the chart, we must create the checkboxes on the fly on the userform and use a class module for the predefinied event routines.

Sounds compilcated, but it isn't. Add a class module and name it "MyEventHandler" and paste in this code:

--- schnipp ---
Option Explicit

Dim WithEvents CHB As MSForms.CheckBox
Dim WithEvents COB As MSForms.CommandButton
Dim S As Excel.Series

Property Set Series(Value As Object)
  Set S = Value
  'Set the checkbox if we have both objects
  If Not CHB Is Nothing Then CHB.Value = CBool(S.Format.Line.Visible)
End Property

Property Set CheckBox(Value As Object)
  Set CHB = Value
  If Not S Is Nothing Then CHB.Value = CBool(S.Format.Line.Visible)
End Property

Property Set CommandButton(Value As Object)
  Set COB = Value
End Property

Private Sub CHB_Change()
  'Hide or show the line
  S.Format.Line.Visible = CHB.Value
End Sub

Private Sub COB_Click()
  'Close the userform
  Unload COB.Parent
End Sub
--- schnapp ---

Then create an empty userform and add this code:

--- schnipp ---
Option Explicit

Dim MCBs As New Collection

Private Sub UserForm_Activate()
  Dim C As MSForms.Control
  Dim MCB As MyEventHandler
  Dim S As Series
  Dim Y As Single
  Dim W As Single
 
  If ActiveChart Is Nothing Then
    MsgBox "Select a chart and try again"
    Unload Me
    Exit Sub
  End If
 
  'For each series
  For Each S In ActiveChart.SeriesCollection
    'Add a checkbox to the form
    Set C = Me.Controls.Add("Forms.CheckBox.1")
    'Get the name from the series
    C.Caption = S.Name
    'Let it shrink
    C.AutoSize = True
    'Move it below the last checkbox
    C.Top = Y
    'Calculate the position for the next checkbox
    Y = Y + C.Height
    'Get the max. width
    If C.Width > W Then W = C.Width
    'Create a new handler for the event routines of the checkbox
    Set MCB = New MyEventHandler
    'Store the objects
    Set MCB.CheckBox = C
    Set MCB.Series = S
    'Keep it alive in our collection
    MCBs.Add MCB
  Next
 
  'Add a commandbutton below
  Set C = Me.Controls.Add("Forms.CommandButton.1")
  C.Caption = "Done"
  C.AutoSize = True
  'Move it below the last checkbox
  C.Top = Y
  Y = Y + C.Height
  If C.Width > W Then W = C.Width
  'ENTER and ESC call this button
  C.Default = True
  C.Cancel = True
  'Create a new handler for the event routines of the commandbutton
  Set MCB = New MyEventHandler
  Set MCB.CommandButton = C
  MCBs.Add MCB
 
  'Resize the form to show all controls
  Me.Height = Y + Me.Height - Me.InsideHeight
  Me.Width = W + Me.Width - Me.InsideWidth
End Sub
--- schnapp ---

Create a normal module and write a sub to show the form as you like e.g.

Sub Test()
  UserForm1.Show
End Sub

Close the VBA editor, click on a chart, press ALT-F8 and run the macro.

I use the ACWZMAIN.mde for creating labels using VB code.

Public Function LabelWizard()
' launch label wizard
' needs reference to ( Access 97 used wzmain80) now use ACWZMAIN.mde

    Dim intReturn As Integer
    Dim fReturn As Boolean
    Dim qdf As QueryDef
   
    ' Check to make sure AddressList still exists
    On Error Resume Next
    Set qdf = CurrentDb.QueryDefs("AddressList")
    If Err > 0 Then
        ' AddressList is missing, so recreate it
        fReturn = RecreateAddressList()
        If Not fReturn Then GoTo ExitHere
    End If
   
    On Error GoTo HandleErr
    intReturn = mlbl_Entry("AddressList")
   
ExitHere:
    Exit Function

HandleErr:
    Select Case Err
        Case Else
            MsgBox Err & ": " & Err.Description, _
              vbOKOnly, "Error in LabelWizard"
    End Select
    Resume ExitHere
    Resume
End Function

Private Function RecreateAddressList() As Boolean
    ' recreates the AddressList query
   
    Dim db As Database
    Dim qdf As QueryDef
    Dim strSQL As String
   
    On Error GoTo HandleErr
   
    strSQL = "SELECT Household.HouseholdName, Members.Salutation, Members.FirstName, " _
    & "Members.LastName, Household.Address, Household.City, " _
    & "Household.StateOrProvince, Household.PostalCode, " _
    & "Household.Country, Members.MakeLabels FROM Household " _
    & "INNER JOIN Members ON Household.AddressID = Members.AddressID WHERE (((Members.MakeLabels)=-1));"

    Set db = CurrentDb
    Set qdf = db.CreateQueryDef("AddressList", strSQL)
   
    RecreateAddressList = True
   
ExitHere:
    Exit Function

HandleErr:
    RecreateAddressList = False
    Select Case Err
        Case Else
            MsgBox Err & ": " & Err.Description, _
              vbOKOnly, "Error in RecreateAddressList"
    End Select
    Resume ExitHere
End Function
