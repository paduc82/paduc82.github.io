rich007a say:

Try this:

 

Sub UpdateSummaryWorkbook()
    Dim FileSys As Object, objFile As Object
    Dim myFolder As Variant
    Dim dtFile As Date, dtLastDate As Date
    Dim wbDaily As Workbook
    Dim ws As Worksheet, wsDaily As Worksheet
    Dim rngNames As Range, cNames As Range, rngTarg As Range, rngFound As Range
    Dim lnLastRow As Long
   
    Set ws = Workbooks("Summary.xlsm").Sheets("Sheet1")
    dtLastDate = CDate(ws.Cells(1, ws.Columns.Count).End(xlToLeft).Value)  'last column in row 1?
   
     'Set path for files - change for your folder:
    Const myDir As String = "C:\Temp\" 'include trailing backslash
   
     'Set up filesys objects:
    Set FileSys = CreateObject("Scripting.FileSystemObject")
    Set myFolder = FileSys.GetFolder(myDir)
   
     'Loop through each file and get date last modified:
    For Each objFile In myFolder.Files
        Debug.Print objFile.Name, objFile.DateLastModified
        If InStr(1, objFile.Name, ".xlsx") > 0 Then 'file is an Excel file...
            dtFile = objFile.DateLastModified
            If dtFile > dtLastDate Then
                Set wbDaily = Workbooks.Open(myDir & objFile.Name)
                Set wsDaily = wbDaily.Worksheets("Sheet1")
               
                 'Check all names in Daily sheet exist in Summary sheet:
                Set rngNames = wsDaily.Range(wsDaily.Range("A2"), wsDaily.Range("A" & wsDaily.Rows.Count).End(xlUp))
                For Each cNames In rngNames
                    Set rngFound = Nothing
                    Set rngFound = ws.Range("A:A").Find( _
                        What:=cNames.Value, _
                        After:=ws.Range("A1"), _
                        LookIn:=xlValues, _
                        LookAt:=xlWhole, _
                        MatchCase:=False)
                   
                    If rngFound Is Nothing Then 'add name to bottom of Summary sheet column:
                        ws.Range("A" & ws.Rows.Count).End(xlUp).Offset(1, 0).Value = cNames.Value
                    End If
                Next cNames
               
                 'Use Vlookup to get data from Daily sheet to Summary sheet:
                Set rngTarg = ws.Cells(1, ws.Columns.Count).End(xlToLeft).Offset(0, 1)
                rngTarg.Value = dtFile
'                rngTarg.NumberFormat = "dd/mm/yyyy"
                rngTarg.Offset(1, 0).Formula = "=VLOOKUP(A2," & rngNames.Resize(, 2).Address(External:=True) & ",2,FALSE)"
                lnLastRow = ws.Range("A" & ws.Rows.Count).End(xlUp).Row
                Set rngTarg = ws.Range(rngTarg.Offset(1, 0), ws.Cells(lnLastRow, rngTarg.Column))
                rngTarg.FillDown
               
                 'Copy and paste as values:
                rngTarg.Value = rngTarg.Value
               
                wbDaily.Close SaveChanges:=False
            End If
        End If
    Next objFile
           
    Set FileSys = Nothing
    Set myFolder = Nothing
End Sub

 

 

Does that do what you need?
