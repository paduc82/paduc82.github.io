Sub UniqueRandomLetters2()
Const mmin As Long = 2 '<<< start row
Const mmax As Long = 27 '<<< last row
Const myCol As String = "C" '<<< extract Random  in col C
Const L As Long = 10 '  <<< Number of random
Dim x As Double, i As Long
Application.ScreenUpdating = False
Randomize
myNext:
Range(myCol & mmin & ":" & myCol & L & mmin - 1).ClearContents
x = Int((mmax - mmin) * Rnd + mmin)
Cells(mmin, myCol) = Cells(x, "A")
For i = mmin + 1 To L + mmin - 1
x = Int((mmax - mmin) * Rnd + mmin)
Cells(i, myCol) = Cells(x, "A")
If Not IsError(Application.Match(Cells(i, myCol), Range(Cells(mmin, myCol), Cells(i - 1, myCol)), 0)) Then GoTo myNext
Next i
Range(myCol & mmin & ":" & myCol & L + mmin - 1).Select
Selection.Sort Key1:=Range(myCol & mmin), Order1:=xlAscending, Header:=xlNo, _
OrderCustom:=1, MatchCase:=False, Orientation:=xlTopToBottom, _
DataOption1:=xlSortNormal
Range(myCol & mmin).Select
Application.ScreenUpdating = True
End Sub
