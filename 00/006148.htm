Private Sub Document_ContentControlOnEnter(ByVal ContentControl As ContentControl)
  If (ContentControl.Title = "CC1" And ContentControl.Checked = True) Then

  If (ContentControl.Title = "CC1" And ContentControl.Checked = False) Then
