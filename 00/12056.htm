Try something like:

  • Right-click the sheet tab of the main data sheet
  • Select the View Code option from the resultant context menu
  • Paste the following event code:

'=========>>
Option Explicit

'--------->>
Private Sub Worksheet_Change(ByVal Target As Range)
    Dim WB As Workbook
    Dim newSH As Worksheet
    Dim Rng As Range, rCell As Range
    Dim sStr As String

    Const sTypeColumn As String = "A"                           '<<=== Modify

    Set Rng = Intersect(Me.Columns(sTypeColumn), Target)

    If Not Rng Is Nothing Then
        Set WB = ThisWorkbook
        For Each rCell In Rng.Cells
            sStr = rCell.Value
            If Not SheetExists(sStr) Then
                With WB
                    Set newSH = .Sheets.Add(after:=.Sheets(.Sheets.Count))
                    newSH.Name = sStr
                    Me.Activate
                End With
            End If
        Next rCell
    End If
End Sub
'<<=========

     
  • Alt+IM to insert a new      code module
  • In the new module, paste the following code: 

'=========>>
Option Explicit

'--------->>
Public Sub UpdateSheet(aSH As Worksheet)
    Dim WB As Workbook
    Dim srcSH As Worksheet, destSH As Worksheet
    Dim Rng As Range, rCell As Range
    Dim ArrIn As Variant, ArrOut() As Variant, ArrHeaders As Variant
    Dim sStr As String
    Dim i As Long, j As Long, iCtr As Long
    Dim iRow As Long, iCols As Long
    Dim UB As Long, UB2 As Long

    Const sMainSheet As String = "Sheet1"                     '<=== Modify

    Set WB = ThisWorkbook
    Set srcSH = WB.Sheets(sMainSheet)
    Set destSH = aSH
    sStr = destSH.Name

    With srcSH
        iRow = LastRow(srcSH, .Columns("A:A"))
        Set Rng = .Range("A2:D" & iRow)
    End With

    ArrIn = Rng.Value
    ArrHeaders = Rng.Rows(0).Value
    UB = UBound(ArrIn)
    UB2 = UBound(ArrIn, 2)

    For i = 1 To UB
        If ArrIn(i, 1) = sStr Then
            iCtr = iCtr + 1
            ReDim Preserve ArrOut(1 To UB2, 1 To iCtr)
            For j = 1 To UB2
                ArrOut(j, iCtr) = ArrIn(i, j)
            Next j
        End If
    Next i

    If CBool(iCtr) Then
        With destSH
            .Range("A1").Resize(1, UB2).Value = ArrHeaders
            .UsedRange.Offset(1).ClearContents
            .Range("A2").Resize(iCtr, UB2).Value = Application.Transpose(ArrOut)
        End With
    End If
XIT:

End Sub

'--------->>
Public Function LastRow(SH As Worksheet, _
                        Optional Rng As Range, _
                        Optional minRow As Long = 1)
    If Rng Is Nothing Then
        Set Rng = SH.Cells
    End If

    On Error Resume Next
    LastRow = Rng.Find(What:="*", _
                       after:=Rng.Cells(1), _
                       Lookat:=xlPart, _
                       LookIn:=xlFormulas, _
                       SearchOrder:=xlByRows, _
                       SearchDirection:=xlPrevious, _
                       MatchCase:=False).Row
    On Error GoTo 0
    If LastRow < minRow Then
        LastRow = minRow
    End If
End Function

'--------->>
Public Function SheetExists(sSheetName As String, _
                            Optional ByVal WB As Workbook) As Boolean
    On Error Resume Next
    If WB Is Nothing Then
        Set WB = ThisWorkbook
    End If
    SheetExists = CBool(Len(WB.Sheets(sSheetName).Name))
End Function
'<<=========

  • Ctrl+R to access the Project Explorer window
  • Double-click the ThisWorkbook module in your workbook and paste the following event code:

'=========>>
Option Explicit

'--------->>
Private Sub Workbook_SheetActivate(ByVal SH As Object)
    Call UpdateSheet(SH)
End Sub
'<<=========

  • Save the file as an xlsm macro-enabled file

I pasted a long string, with no spaces (a URL) into a cell in a table. Word reformatted the table by widening that column and narrowing all the other columns, in order to try to fit the long string into the cell. I had to manually restore the cell widths.

pamcaswell say: The Automatically resize to fit contents feature is greyed out in the table style options dialog and so cannot be set in a table style.  But if you have a particular structure that you often use, you can format a blank one to your requirements and save it as AutoText or as a Quick Table.

Pam

hansv mvp say:

Click anywhere in the table.

Activate the Layout tab of the ribbon (under Table Tools).

In the Table group, click Properties.

If necessary, activate the Table tab.

Click Options...

Clear the check box "Automatically resize to fit contents".

Click OK twice.
