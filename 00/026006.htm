Function OpenFile()
'Dim the File Dialog object and string
Dim objFileDialog As FileDialog
Dim strFile As String

'Set the objFileDialog to an instance of the FileDialog object
Set objFileDialog = Application.FileDialog(msoFileDialogFilePicker)

'Set the Properties of the objFileDialog object
objFileDialog.AllowMultiSelect = False
objFileDialog.ButtonName = "Select"
objFileDialog.InitialView = msoFileDialogViewDetails
objFileDialog.Title = "Select Excel File"
objFileDialog.InitialFileName = "C:\"
objFileDialog.Filters.Clear
objFileDialog.Filters.Add "Excel", "*.xls; *.xlsx", 1
objFileDialog.FilterIndex = 1

'Show the FileDialog box
objFileDialog.Show

'Set strFile to the first record of the SelectedItems property of our FileDialog
strFile = objFileDialog.SelectedItems(1)

'Return the File Path string
OpenFile = strFile
End Function

Sub BoldSomeWords(shp As Object, str As String, boldWords As String)

    Dim word As Variant
    Dim iStart As Integer, iEnd As Integer

    'Convert the list of words in to an iterable array, and
    ' iterate it.
    For Each word In Split(boldWords, ",")
        'Loop just in case there are duplicates
        Do Until InStr(iEnd + 1, str, word) = 0
            iStart = InStr(iStart + 1, str, word)
            iEnd = iStart + Len(word)
            shp.TextFrame.TextRange.Characters(iStart, Len(word)).Characters.Font.Bold = msoTrue
        Loop
    Next

End Sub

ismet kocaman say:

John, I apologize for jumping into the discussion.

Oldhasbeen, MS Project versions 2013 and later feed those dates to the filters automatically in order for the related graphical reports to run without interruption. Dates are determined based on the current date and the Week starts on setting. If you use the same filters in versions earlier than Project 2013, they will ask for input. 

If you want use these filters interactively in versions 2013 and later, then you can change the prompt strings so that MS Project cannot recognize them. Thus the related graphical reports can be created based on any week that you want such as next week. You can even enter a two week range.

There are probably two things at play here:

  • it's usually best to use HTML entities for reserved characters like &, <, > and so on, precisely to avoid this sort of error. Your URL would then look like this:

https://qa.tbe.taleocloud.net/qa4/ats/careers/cwsAccessViaEmail.jsp?org=ZENIL3&amp;cws=1&amp;currentTime=1427232350841&amp;reqId=275&amp;candId=353&amp;cws=1&amp;WebPage=requisition.jsp&amp;act=actionAccessInvite&amp;org=ZENIL3&amp;signature=8qfXXqRZ7McxqMMhSEQ4MniOe2LLu%2F3zBIq4IsRDgS8%3D
  • Your mail may be being wrapped at 132 characters; I reckon that the '1' after cws= is the 132nd character on that line. Encoding the HTML, e.g. as B64, should get around that.
