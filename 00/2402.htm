Open a new wbk and try the code below:


Sub Subtotal_Sum()

Dim ws1, ws2 As Worksheet

Dim r, LRow As Long

Application.ScreenUpdating = False

Set ws1 = Sheets("Sheet1")

Set ws2 = Sheets("Sheet2")

LRow = ws1.Cells(Rows.Count, 1).End(xlUp).Row

ws1.Range("C" & LRow).Offset(1).FormulaR1C1 = _

"=Subtotal(9,R[-" & LRow - 1 & "]C:R[-1]C)"

ws2.Columns("A:C").ClearContents

ws1.Range("D1").EntireColumn.Insert

ws1.Range("B1:B" & LRow).AdvancedFilter Action:=xlFilterCopy, _

CopyToRange:=ws1.Cells(1, "D"), Unique:=True

r = ws1.Cells(Rows.Count, 4).End(xlUp).Row

For i = 2 To r

ws1.Range("B1").AutoFilter , Field:=2, Criteria1:=ws1.Cells(i, 4)

ws1.Range("A1:C" & LRow + 1).Copy

ws2.Range("A" & Cells(Rows.Count, "C").End(xlUp).Row) _

.Offset(2).PasteSpecial xlValues

ws2.Range("A" & Cells(Rows.Count, "A").End(xlUp).Row) _

.Offset(1) = "TOTAL SUM of " & ws1.Cells(i, 4)

Application.CutCopyMode = False

ws1.Range("B1").AutoFilter

Next i

ws1.Range("D1").EntireColumn.Delete

ws2.Range("A1:C2").Cells.Delete

ws2.Range("A" & ws2.Cells(Rows.Count, "A").End(xlUp).Row) _

.Offset(2) = "GRAND TOTAL SUM = " & ws1.Cells(LRow + 1, 3).Value

For i = 2 To ws2.Cells(Rows.Count, 1).End(xlUp).Row

If IsNumeric(ws2.Cells(i, 1)) Then _

ws2.Cells(i, 1) = Format(ws2.Cells(i, 1), "dd/mm/yyyy")

Next i

ws1.Cells(LRow + 1, 3) = ""

Application.ScreenUpdating = True

End Sub


note: before you run the code, rewrite the dates.
