Sub Add_CheckBox_ActiveX()
 'May 10, 2015
Dim rng As Range
Set rng = [A1:C3]
Dim n As Long: n = 0
For Each s In ActiveSheet.Shapes
If Not Intersect(s.TopLeftCell, rng) Is Nothing Then s.Delete
Next
For Each r In rng
n = n + 1
Set sh = ActiveSheet.OLEObjects.Add(ClassType:="Forms.CheckBox.1", Link:=False, _
DisplayAsIcon:=False, Left:=r.Left, Top:=r.Top, Width:=r.Width, Height:=r.Height)
With sh
.Placement = xlMoveAndSize
.Name = "ChkBx_" & n
.Object.Value = False
.Object.Caption = ""

.Object.BackColor = RGB(153, 204, 255)

End With
Next
End Sub
