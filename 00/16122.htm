jay freedman say:

You're correct that you cannot enter formatted text in the AutoCorrect dialog. It requires a bit of indirection.

First create the replacement string, including all the required formatting, in the body of any document. Select the entire string. Now open the AutoCorrect dialog. The text of the selection will appear in the "With" box as plain text, but the Formatted Text button will not be enabled. Click on that button, and the replacement string will now show the formatting. Enter the trigger text in the Replace box and click OK.

Sub ExportExcelCSV()
  Dim strOut As String
  Dim tbl As AccessObject
  Dim f As Boolean

  With Application.FileDialog(4) ' msoFileDialogFolderPicker
    .Title = "Please select the target folder"
    If .Show Then
      strOut = .SelectedItems(1)
      If Not Right(strOut, 1) = "\" Then
        strOut = strOut & "\"
      End If
    Else
      MsgBox "You didn't select a target folder.", vbExclamation
      Exit Sub
    End If
  End With
  f = (MsgBox("Do you want to export all tables to Excel (No = CSV)?", _
    vbQuestion + vbYesNo) = vbYes)

  For Each tbl In CurrentData.AllTables
    If Not tbl.Name Like "MSys*" And Not tbl.Name Like "~" Then
      If f Then
        DoCmd.TransferSpreadsheet acExport, acSpreadsheetTypeExcel12Xml, _
          tbl.Name, strOut & tbl.Name & ".xlsx", True
      Else
        DoCmd.TransferText acExportDelim, , _
          tbl.Name, strOut & tbl.Name & ".csv", True
      End If
    End If
  Next tbl
End Sub
