The top left cell of your data is A1, from there the CurrentRegion are all adjacent cells that have a content, means all o your data.


The Range object supports up to 3 keys, use the top cell of the column for a key and specify the order. Set the Header property to xlYes if you have headings (the sort starts at the 2nd row) otherwise to xlNo (for the 1st row). Don't use xlGuess, it can fail.


It is superfluous to disable the screen when you execute just one line of code.

andreas killer say:

The default property of the Folder2 object is "Title", not "Path".

To get the path to the selected item you must call the Self property which returns a FolderItem object and this has the path property.

Your code is dangerous, e.g. is the activesheet is empty, your selection is the whole sheet! Below is an example how this process could made a bit safer.

Andreas.

Option Explicit

'Optionflags des ShellGetFolder-Dialogs
Private Enum vbShellGetFolderFlags
  BIF_RETURNONLYFSDIRS = &H1
  BIF_DONTGOBELOWDOMAIN = &H2
  BIF_STATUSTEXT = &H4
  BIF_RETURNFSANCESTORS = &H8
  BIF_EDITBOX = &H10
  BIF_VALIDATE = &H20
  BIF_NEWDIALOGSTYLE = &H40
  BIF_BROWSEINCLUDEURLS = &H80
  BIF_BROWSEFORCOMPUTER = &H1000
  BIF_BROWSEFORPRINTER = &H2000
  BIF_BROWSEINCLUDEFILES = &H4000
  BIF_SHAREABLE = &H8000
  BIF_SHOWALLOBJECTS = &H8
End Enum

Private Const BIF_DefaultOptions = BIF_EDITBOX Or BIF_VALIDATE Or BIF_SHOWALLOBJECTS Or _
  BIF_STATUSTEXT Or BIF_NEWDIALOGSTYLE
Private Const BIF_BrowseFolder = BIF_RETURNONLYFSDIRS Or BIF_DefaultOptions
Private Const ssfDESKTOP = &H0

Sub CopyFolder2_AK()
  Dim fso As Object
  Dim ThisRange As Range, Cell As Range
  Dim FromPath As String, ToPath As String

  Do
    Do
      'Let the user select a path
      FromPath = ShellGetFolder(, "Select a folder you want to copy:", BIF_BrowseFolder)
      'Aborted?
      If FromPath = "" Then Exit Sub
      'Some items return "::" followed by a GUID
      If Left(FromPath, 1) = ":" Then
        MsgBox "Invalid Path selected: " & FromPath
        FromPath = ""
      End If
    Loop Until FromPath <> ""

    Do
      ToPath = ShellGetFolder(, "Select where you want to put them:")
      If ToPath = "" Then Exit Sub
      If Left(ToPath, 1) = ":" Then
        MsgBox "Invalid Path selected: " & ToPath
        ToPath = ""
      End If
    Loop Until ToPath <> ""

    'Make sure the pathes are not a part of each other
    If (InStr(FromPath, ToPath) > 0) Or (InStr(ToPath, FromPath) > 0) Then
      MsgBox "One path is a subpath of the other"
      FromPath = ""
    End If
  Loop Until FromPath <> ""
 
  'Install an error handler to catch unexpected errors
  On Error GoTo ErrorHandler
 
  Set fso = CreateObject("scripting.filesystemobject")
 
  'Get the filled range in column A
  Set ThisRange = Range("A1", Range("A" & Rows.Count).End(xlUp))
  For Each Cell In ThisRange
    'Copy only if we have a value in the cell!
    If Not IsEmpty(Cell) Then fso.CopyFolder FromPath, ToPath & "\" & Cell
  Next

  MsgBox "You can find the files and subfolders from" & vbCrLf & FromPath & vbCrLf & _
    "in the created folders of" & vbCrLf & ToPath
  Exit Sub
 
ErrorHandler:
  MsgBox "Error: " & Err.Number & ": " & Err.Description
End Sub

Private Function ShellGetFolder( _
    Optional Path As Variant = ssfDESKTOP, _
    Optional Caption As String = "", _
    Optional Options As vbShellGetFolderFlags = BIF_DefaultOptions) As String
  'http://msdn.microsoft.com/en-us/library/windows/desktop/bb774065(v=vs.85).aspx
  Dim objShell As Object, objBrowse As Object

  On Error Resume Next
  Set objShell = CreateObject("Shell.Application")
  'Dialog starten und Path zurÃ¼ckgeben
  If IsNumeric(Path) Then
    'Anfangspfad als Konstante
    Set objBrowse = objShell.BrowseForFolder(&H0, Caption, Options, CLng(Path))
  Else
    'Anfangspfad als String
    Set objBrowse = objShell.BrowseForFolder(&H0, Caption, Options, Path & Chr(0))
  End If
  'objBrowse.ParentFolder.ParseName objBrowse.Title
  ShellGetFolder = objBrowse.Self.Path
End Function
