Hi Jack,

Sorry for the delayed response. I have been away on personal business and did not have access to my development machines. iPad's are nice but they don't allow me to write code. :-)

Here is a macro that will convert the individual documents or groups of documents from doc to docx format. To select multiple documents for conversion, when you are prompted to select the file you can hold down the Shift key and select a contiguous list of files or use the Cmd key to select multiple individual files.

Sub ConvertFiles()
    'By Richard V Michaels, October 2013
    'http://www.greatcirclelearning.com
    Dim aScript As String, aPath As String, aList As String, aFile As Variant, N As Long
    Dim doc As Word.Document, nDoc As Word.Document, fName As String
    Set doc = ActiveDocument
    
    On Error GoTo ErrHandler
    aPath = MacScript("return (path to documents folder) as String")
    aScript = "set applescript's text item delimiters to "","" " & vbCr
    aScript = aScript & "set sFiles to (choose file of type "
    aScript = aScript & " {""com.microsoft.Word.doc""} "
    aScript = aScript & "with prompt ""Select the File(s)"" default location alias """
    aScript = aScript & aPath & """ multiple selections allowed true) as string" & vbCr
    aScript = aScript & "set applescript's text item delimiters to """" " & vbCr
    aScript = aScript & "return sFiles"
    aList = MacScript(aScript)
    'aList contains a comma delimited list of the user's selected files (full path names)
    If aList = "" Then
        MsgBox "No Files Selected", vbExclamation, "ConvertFiles Macro"
        Exit Sub
    End If
    Application.ScreenUpdating = False
    aFile = Split(aList, ",")
    For N = LBound(aFile) To UBound(aFile)
        Set nDoc = Documents.Open(aFile(N), AddToRecentFiles:=False)
        fName = nDoc.FullName & "x"
        nDoc.SaveAs fileName:=fName, FileFormat:=Word.wdFormatXMLDocument
        nDoc.Close
    Next N
    Application.ScreenUpdating = True
    doc.Activate
    Set nDoc = Nothing
    Exit Sub
ErrHandler:
    Err.Clear
    aList = ""
    Application.ScreenUpdating = True
    Set nDoc = Nothing
    doc.Activate
End Sub
