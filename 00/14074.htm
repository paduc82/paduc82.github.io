    RE: all dates in a month for a particular weekday

    VBA solution...

    'Jim Cone - 02/14/2005
    'Finds all particular weekdays in a month, given the first day.

    Sub TestWeekdays()
    Dim F         As Long
    Dim lngN      As Long
    Dim dteFirst  As Date
    Dim varAnswer As Variant
    Dim arrWeekdays() As Variant

    'First day of any month
    dteFirst = Range("A1").Value
    'A weekday in any month is limited to max of 5
    ReDim arrWeekdays(0 To 4, 1 To 1)

    For F = 0 To 31
        varAnswer = Weekday(dteFirst + F)
       'vbSunday is 1, vbSaturday is 7
        If varAnswer = vbSunday Then          '<<<<<<<<<
           If Month(dteFirst) = Month(dteFirst + F) Then
              arrWeekdays(lngN, 1) = dteFirst + F
              lngN = lngN + 1
           Else
               arrWeekdays(lngN, 1) = "None"
           End If
        End If
    Next 'F
    Range("B1:B5").Value = arrWeekdays()
    End Sub
