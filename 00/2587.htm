Hi,

Please copy and paste this macro in place of yours within the "Main File" and clear all data currently in your main file.  Make sure only the files you want data copied from are in the folder "Files."

Sub GetData()

Dim wkb As Workbook, myfiles, mypath, numnames() As String, wks As Worksheet

Set wks = ThisWorkbook.ActiveSheet

mypath = "C:\Documents and Settings\Desktop\Files\"

myfiles = Dir(mypath & "*.xl*")

If myfiles = "" Then Exit Sub

Do While myfiles <> ""
    Set wkb = Workbooks.Open(mypath & myfiles)
    lastCRow = wkb.ActiveSheet.Cells(Rows.Count, "B").End(xlUp).Row - 8
    lastProw = wks.Cells(Rows.Count, "B").End(xlUp).Row
    wkb.ActiveSheet.Range("B9:CL9" & lastCRow).Copy Destination:=wks.Range("B1").Offset(lastProw, 0)
    numnames = Split(myfiles, "-")
    wks.Range("A:A").SpecialCells(xlCellTypeBlanks).Value = numnames(0)
    myfiles = Dir()
    wkb.Close SaveChanges:=False
Loop


End Sub
