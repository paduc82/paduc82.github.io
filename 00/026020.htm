andreas killer say:
In VBA we have a split function that makes it very easy for us. A simple UDF makes life a lot easier.

B1:F1: {=MySplit(A1,"x"," ")} 

Function MySplit(ByVal Expression, ParamArray Delimiters()) As Variant
  Dim i As Integer
  For i = 0 To UBound(Delimiters)
    Expression = Replace(Expression, Delimiters(i), vbNullChar, Compare:=vbTextCompare)
  Next
  Do While InStr(Expression, vbNullChar & vbNullChar) > 0
    Expression = Replace(Expression, vbNullChar & vbNullChar, vbNullChar)
  Loop
  Do While Left(Expression, 1) = vbNullChar
    Expression = Mid(Expression, 2)
  Loop
  Do While Right(Expression, 1) = vbNullChar
    Expression = Left(Expression, Len(Expression) - 1)
  Loop
  MySplit = VBA.Split(Expression, vbNullChar, Compare:=vbTextCompare)
End Function
