doug robbins - mvp office apps & services (word) say:

Use a macro containing the following code:

Dim i As Long

Dim rngcaption As Range

Dim rngTable As Range

On Error GoTo ExitSub

With ActiveDocument

    For i = 1 To .Fields.Count

        If .Fields(i).Type = wdFieldSequence And InStr(.Fields(i).Code, "Table") > 1 Then

            Set rngcaption = .Fields(i).Code.Paragraphs(1).Range

            Set rngTable = .Range

            rngTable.End = rngcaption.Start - 1

            rngTable.Collapse wdCollapseEnd

            Set rngTable = rngTable.Tables(1).Range

            rngTable.Cut

            rngcaption.ParagraphFormat.KeepWithNext = True

            rngcaption.Collapse wdCollapseEnd

            rngcaption.Paste

        End If

    Next i

End With

ExitSub:
