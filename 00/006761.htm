andreas killer say:

To add to the confusion a bit, some of the plain text values have 2 rows of title and some just have 1 row.

Also at the bottom of all the tables there is a blank row and then a "total" row. This does not need included within the table but it would be nice if it can be formatted as a solid colour. I could possibly modify your original code to search for "total" and then apply basic formatting to it too.

Okay, that's easy, try the code below.

Andreas.

Sub Test()
  Dim Where As Range, Here As Range, This As Range
  Dim FirstAddress As String
  Dim Diff As Long

  'Step 1: Create the tables

  'Where to search?
  Set Where = Range("A:A")
  'Find the 1st occurence
  Set Here = Where.Find("Business", LookIn:=xlValues, LookAt:=xlWhole)
  'Found?
  If Not Here Is Nothing Then
    'Remember the position
    FirstAddress = Here.Address
    Do
      'Remove rows above Here if any
      Set This = Here.CurrentRegion
      Diff = Here.Row - This.Row
      If Diff > 0 Then
        Set This = This.Offset(Diff).Resize(This.Rows.Count - Diff)
      End If
      'Format as table
      ActiveSheet.ListObjects.Add xlSrcRange, This
      'Find the next occurence
      Set Here = Where.FindNext(Here)
      'Loop until 1st position
    Loop Until FirstAddress = Here.Address
  End If

  'Step 2: Color the Total rows

  'Find the 1st occurence
  Set Here = Where.Find("Total", LookIn:=xlValues, LookAt:=xlWhole)
  'Found?
  If Not Here Is Nothing Then
    'Remember the position
    FirstAddress = Here.Address
    Do
      'Apply a color
      With Here.CurrentRegion.Interior
        .ThemeColor = xlThemeColorAccent2
        .TintAndShade = 0.4
      End With
      'Find the next occurence
      Set Here = Where.FindNext(Here)
      'Loop until 1st position
    Loop Until FirstAddress = Here.Address
  End If
End Sub
