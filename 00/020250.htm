ken sheridan say: I assume all the dates are Tuesdays, and you wish to correlate the previous, following and week-after-that values on Market, CouponName, Tactic, Type and Segment.  Try this:

SELECT Market, CouponName, [Date], Tactic, [Type], Segment,
    (SELECT TotalRedeemed
      FROM TheTable As T2
      WHERE T2.Market = T1.Market
      AND T2.CouponName = T1.CouponName
      AND T2.Tactic = T1.Tactic
      AND T2.[Type] = T1.[Type]
      AND T2.Segment = T1.Segment
      AND T2.[Date] = T1.[Date]-7) AS WeekPrior,
TotalRedeemed AS CurrentWeek,
    (SELECT TotalRedeemed
      FROM TheTable As T2
      WHERE T2.Market = T1.Market
      AND T2.CouponName = T1.CouponName
      AND T2.Tactic = T1.Tactic
      AND T2.[Type] = T1.[Type]
      AND T2.Segment = T1.Segment
      AND T2.[Date] = T1.[Date]+7) AS FollowingWeek,
    (SELECT TotalRedeemed
      FROM TheTable As T3
      WHERE T3.Market = T1.Market
      AND T3.CouponName = T1.CouponName
      AND T3.Tactic = T1.Tactic
      AND T3.[Type] = T1.[Type]
      AND T3.Segment = T1.Segment
      AND T3.[Date] = T1.[Date]+14) AS WeekAfterThat
FROM TheTable As T2;

If the dates are not all Tuesdays, then you can return a week-starting date for any Date with the following function:

Public Function WeekStart(intStartDay As Integer, Optional varDate As Variant)

    ' Returns 'week starting' Date for any Date
    
    ' Arguments:
    ' 1. intStartDay - weekday on which week starts, 1-7 (Sun - Sat)
    ' 2. varDate - optional Date value for which week starting
    '   Date to be returned.  Defaults to current date
    
    If IsMissing(varDate) Then varDate = VBA.Date
    
    If Not IsNull(varDate) Then
        WeekStart = DateValue(varDate) - Weekday(varDate, intStartDay) + 1
    End If
    
End Function

and correlate the subqueries with the outer query on the return value of the function, e.g.

      AND WeekStart(1,T3.[Date]) = WeekStart(1,T1.[Date]+14)) AS WeekAfterThat

I would recommend against using Date and Type as column names.  These are the names of a function and statement respectively in Access and as such, are 'reserved words'.  This is why I've enclosed them in square brackets in the SQL statement.  However they are best avoided as object names.

rich007a say:

To test this, on Sheet1 I had:
 names in cells B4:B10
 in C3:G3 I had: "Poles", "Barrels", "Flag", "Polo" and "Arena"
 in C4:G10 I had either blank or "x"
 and the sheets named: "Poles", "Barrels", "Flag", "Polo" and "Arena"

The first procedure randomises the names, keeping their "x"s with them.
The second procedure copies from Sheet1 to the event sheets, currently to A2-down.

 

Sub Randomise()
    Dim lnTopRow As Long, lnRows As Long, lnLeftCol As Long, lnCols As Long
    Dim rng As Range
   
    Set rng = Range("B4:G10")  'Set range here
   
    Application.ScreenUpdating = False
    Application.Calculation = xlCalculationManual
   
    lnTopRow = rng(1, 1).Row
    lnRows = rng.Rows.Count
    lnLeftCol = rng(1, 1).Column
    lnCols = rng.Columns.Count
   
     'Insert two extra columns to the left of the range:
    rng.Resize(, 2).EntireColumn.Insert
   
     'Enter RAND and RANK formulae in the new columns:
    Cells(lnTopRow, lnLeftCol + 1).Formula = "=RAND()"
    Cells(lnTopRow, lnLeftCol).FormulaR1C1 = _
        "=RANK(RC[1],R" & lnTopRow & "C[1]:R" & lnTopRow - 1 + lnRows & "C[1])"
       
     'Fill down and calculate:
    Range(Cells(lnTopRow, lnLeftCol), Cells(lnTopRow - 1 + lnRows, lnLeftCol + 1)).FillDown
    Application.Calculate
   
     'Equivalent of copy and paste as values:
    Cells(lnTopRow, lnLeftCol).Resize(lnRows, 2).Value _
        = Cells(lnTopRow, lnLeftCol).Resize(lnRows, 2).Value
   
     'Sort:
    ActiveSheet.Sort.SortFields.Clear
    ActiveSheet.Sort.SortFields.Add Key:=Cells(lnTopRow, lnLeftCol).Resize(lnRows, 1), _
        SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:=xlSortNormal
    With ActiveWorkbook.Worksheets("Sheet1").Sort
        .SetRange Cells(lnTopRow, lnLeftCol).Resize(lnRows, lnCols + 2)
        .Header = xlNo
        .MatchCase = False
        .Orientation = xlTopToBottom
        .SortMethod = xlPinYin
        .Apply
    End With
   
     'Delete helper columns:
    Cells(lnTopRow, lnLeftCol).Resize(, 2).EntireColumn.Delete
   
    ActiveSheet.UsedRange 'Reset the used range
    Application.ScreenUpdating = True
    Application.Calculation = xlCalculationAutomatic
End Sub

Sub CopyToSheets()
    Dim rng As Range, rngCol As Range, rngCopy As Range
    Dim i As Long
    Dim wsNames As Worksheet, wsEvent As Worksheet
   
    Set wsNames = Worksheets("Sheet1") 'Update as required
    Set rng = wsNames.Range("C4:G10") 'Range containing blank cells or "x"
   
    For i = 1 To rng.Columns.Count
        Set rngCol = rng.Columns(i)
        Set wsEvent = Worksheets(rngCol.Cells(1, 1).Offset(-1, 0).Value)
        
         'Clear existing names from event sheets:
         '(Note: Cells in row 3 must contain event names that match sheet names)
        wsEvent.Range(wsEvent.Range("A2"), wsEvent.Range("A2").End(xlDown)).Clear
       
         'Copy randomised names to event sheets:
        On Error Resume Next
        Set rngCopy = Intersect(rngCol.SpecialCells(xlCellTypeConstants).EntireRow _
            , wsNames.Columns("B"))
        On Error GoTo 0
        If Not rngCopy Is Nothing Then
            rngCopy.Copy Destination:=wsEvent.Range("A2") 'Destination range on event sheets
        End If
        Set rngCopy = Nothing
    Next i
End Sub
