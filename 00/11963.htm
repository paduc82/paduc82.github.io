Sub AddFormatConditions()
    With Range(Range("S2"), Range("S" & Rows.Count).End(xlUp)).FormatConditions
        .Delete
        .Add(Type:=xlExpression, Formula1:="=AND(S2<>"""",ABS(S2)<=0.1)") _

If you have to use legacy text form fields (which you do have to use if the form must be usable on Word 2003 or earlier), then it is not possible. If you must use legacy form fields and forms protection, the only thing users can do is to draw in some other program such as Paint, save the drawing as a file, and use the Picture button on Word's Insert ribbon while the cursor is in the unprotected area between the section breaks.

If everyone uses Word 2007 or 2010, there's a much better alternative. Instead of legacy form fields, use text content controls. To protect the text that isn't inside content controls, select text plus content controls, and click Group > Group on the Developer ribbon. You can make one or several groups, and leave other areas not grouped. Any text area that isn't included within any grouping remains editable, including all the drawing tools.

that is exactly what a mail merge is for.

ossiemac say:

I didn't bother waiting for a reply to my last message to you. I already had an example that I have modified for you. Copy the code into a standard module and run the sub SendEmail.

See the comments throughout the code re the options I have given you. Currently saves as xl97-2003 format but you can commment out that line and uncommment the line to save as later versions (xl2007-2010). Also the options to Display so the email is opened and the user sends manually or Send and they automatically send.

Note:  It is worth reviewing the solution by teylyn and see if you like it better.

 

Sub SendEmail()
    Dim objOutlook As Object
    Dim objNameSpace As Object
    Dim objMailItem As Object
    Dim ws As Worksheet
    Dim rngEmailAddress As Range
    Dim c As Range
    Dim rngAttach As Range
    Dim strBody As String
    Dim strAttachment As String
   
    Set ws = Sheets("ASPAC Members 120912")   'Edit the sheet name if required.
   
    With ws
        Set rngEmailAddress = .Range(.Cells(2, "R"), .Cells(.Rows.Count, "R").End(xlUp))
    End With
   
    On Error Resume Next
    Set objOutlook = GetObject(, "Outlook.Application")
    On Error GoTo 0
   
    If objOutlook Is Nothing Then
        Set objOutlook = CreateObject("Outlook.Application")
    End If
   
    Set objNameSpace = objOutlook.GetNamespace("MAPI")
   
    strBody = "Please review the data in the attachment"     'Edit as required.
   
    For Each c In rngEmailAddress
        With ws
            Set rngAttach = Union(.Range(.Cells(1, "A"), .Cells(1, "BD")), _
                        .Range(.Cells(c.Row, "A"), .Cells(c.Row, "BD")))
        End With
       
        Set objMailItem = objOutlook.CreateItem(0)
       
        strAttachment = FileToAttach(rngAttach)
        With objMailItem
            .to = c.Value
            '.CC = "email address here"        'Optional
            '.BCC = "email address here"       'Optional
            .Subject = "This is sample subject"      'Edit as required.
            .Body = strBody
            .Attachments.Add (strAttachment)
            .Display    'Displays message and waits for user to send
            '.Send       'Auto Send email (only use Display or Send; NOT both.)
        End With
        Kill strAttachment
        DoEvents
    Next c
   
    Set c = Nothing
    Set rngEmailAddress = Nothing
    Set ws = Nothing
    Set objMailItem = Nothing
    Set objNameSpace = Nothing
    Set objOutlook = Nothing

End Sub


Function FileToAttach(rng As Range) As String

    Dim wbTemp As Workbook
 
    rng.Copy
    Set wbTemp = Workbooks.Add(1)
   
    With wbTemp.Sheets(1)
        rng.Copy
        .Cells(1, 1).PasteSpecial Paste:=8
        .Cells(1, 1).PasteSpecial xlPasteValues, , False, False
        .Cells(1, 1).PasteSpecial xlPasteFormats, , False, False
        .UsedRange.Columns.AutoFit
        .Cells(1, 1).Select
    End With
    Application.CutCopyMode = False
   
    Application.DisplayAlerts = False
    'Following line saves as xl2007-xl2010 file format
    'wbTemp.SaveAs Filename:=ThisWorkbook.Path & "\TempWorkbook.xlsx", _
                FileFormat:=xlOpenXMLWorkbook, CreateBackup:=False
               
   
    'Following line saves as xl97-2003 file format
    wbTemp.SaveAs Filename:= _
        ThisWorkbook.Path & "\TempWorkbook.xls", FileFormat:=xlExcel8
       
    Application.DisplayAlerts = True
   
    FileToAttach = wbTemp.FullName
    wbTemp.Close
    Set wbTemp = Nothing
End Function
