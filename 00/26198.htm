macropod say:

Hi Orcas,

You don't really need a macro for this - you can do it with Find/Replace after selecting the table, where:
Find = ^+
Replace = -

Nevertheless, here's a macro that uses Find/Replace to process all tables in the document:

Sub TableFix()
Dim oTbl As Table
For Each oTbl In ActiveDocument.Tables
  With oTbl.Range.Find
    .ClearFormatting
    .Text = "^+"
    With .Replacement
      .ClearFormatting
      .Text = "-"
    End With
    .Forward = True
    .Wrap = wdFindStop
    .Format = False
    .MatchCase = False
    .MatchWholeWord = False
    .MatchWildcards = False
    .MatchSoundsLike = False
    .MatchAllWordForms = False
    .Execute Replace:=wdReplaceAll
 End With
Next
End Sub
