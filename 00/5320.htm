Sub Email_QC_Form()
  Dim xRng As Range
  Dim OutApp As Object
  Dim OutMail As Object
  Dim wDoc As Object
  Dim wRng As Object
 
  'Get access to Outlook first
  On Error Resume Next
  Set OutApp = GetObject(, "Outlook.Application")
  If OutApp Is Nothing Then
    Set OutApp = CreateObject("Outlook.Application")
    If OutApp Is Nothing Then
      MsgBox "Outlook is not accessible"
      Exit Sub
    End If
  End If
 
  'Get the contents
  On Error GoTo Errorhandler
  Sheets("QC Form").Unprotect "bac_cet"
  Set xRng = Sheets("QC Form").Range("A1:O60").SpecialCells(xlCellTypeVisible)
 
  'Now create a mail
  Set OutMail = OutApp.CreateItem(0)
  OutMail.Subject = "QC Feedback_CET"
  Set wDoc = OutMail.GetInspector.WordEditor
  Set wRng = wDoc.Range
 
  'Copy and paste the contents
  xRng.Copy
  wRng.Paste
 
  OutMail.Display
 
Errorhandler:
  Select Case Err
    Case 0
      'Success
    Case 1004
      MsgBox "The selection is not a range or the sheet is protected" & _
        vbNewLine & "please correct and try again."
    Case Else
      MsgBox "Error " & Err.Number & ": " & Err.Description
  End Select

ExitPoint:
  Sheets("QC Form").Protect "bac_cet"
End Sub
