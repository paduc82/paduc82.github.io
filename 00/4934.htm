hansv mvp say:

Why do you use

 

    GCriteria = GCriteria & "..."

 

If you want to apply one condition at a time, remove GCriteria & throughout the code, so that you have

 

    GCriteria = "..."

 

If you want to apply multiple criteria:

 

Private Sub cmdsearch_Click()
    Const conJetDate = "\#mm\/dd\/yyyy\#"
    Me.Requery
    If Len(cbosearchField) = 0 Or IsNull(cbosearchField) = True Then
        MsgBox "You must select a field to search."
    ElseIf Len(txtSearchString) = 0 Or IsNull(txtSearchString) Then
        MsgBox "You must enter a search string."
    Else
        Select Case cbosearchField.Value
            Case "Bank"
                GCriteria = GCriteria & " AND ([Bank] Like ""*" & Me.txtSearchString & "*"")"
            Case "ID"
                GCriteria = GCriteria & " AND ([ID] = " & Me.txtSearchString & ")"
            Case "Amount"
                GCriteria = GCriteria & " AND ([Amount] = " & Me.txtSearchString & ")"
            Case "Cheque No"
                GCriteria = GCriteria & " AND ([ChequeNo] Like ""*" & Me.txtSearchString & "*"")"
            Case "Moveing Date"
                GCriteria = GCriteria & " AND ([movedate] = " & Format(Me.txtSearchString, conJetDate) & ")"
            Case "Due Date"
                GCriteria = GCriteria & " AND ([duedate] = " & Format(Me.txtSearchString, conJetDate) & ")"
            Case "Date Enter"
                GCriteria = GCriteria & " AND ([todayenter] = " & Format(Me.txtSearchString, conJetDate) & ")"
        End Select
        If Left(GCriteria, 5) = " AND " Then
            GCriteria = Mid(GCriteria, 6)
        End If
        Form_InfoMoveSearch.RecordSource = "SELECT * FROM BankInfoMoveSearchQry WHERE " & GCriteria
        MsgBox "Results have been filtered."
    End If
End Sub
