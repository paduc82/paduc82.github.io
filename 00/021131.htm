Public Function RegExp_Replace(Expression As Variant, pattern As String, ReplaceStr As String, Optional ByVal Index, Optional ByVal IgnoreCase, Optional ByVal FullMatch) As Variant
   Dim objRegEx            As New RegExp
   Dim objMatch            As MatchCollection
   Dim vbeCompareMethod    As VbCompareMethod
   Dim strResult           As String

   On Error GoTo Error_RegExp_Replace
   
   If IsMissing(IgnoreCase) Then IgnoreCase = True
   If IsMissing(FullMatch) Then FullMatch = True
   If IsMissing(Index) Then Index = 0
   
   pattern = Replace(pattern, ";", ",")
   
   If IgnoreCase Then
      vbeCompareMethod = vbTextCompare
   Else
      vbeCompareMethod = vbBinaryCompare
   End If
   
   objRegEx.IgnoreCase = IgnoreCase
   objRegEx.Global = FullMatch
   objRegEx.pattern = pattern
   
   If objRegEx.test(Expression) Then
      Set objMatch = objRegEx.Execute(Expression)
      
      If Index = 0 Then
         strResult = objRegEx.Replace(Expression, ReplaceStr)
      Else
         If Index - 1 >= objMatch.Count Then Index = objMatch.Count
         strResult = Left(Expression, objMatch(Index - 1).FirstIndex) & Replace(Expression, objMatch(Index - 1), ReplaceStr, objMatch(Index - 1).FirstIndex + 1, 1, vbeCompareMethod)
      End If
   Else
      ''' modified 12/09/2015, return the expression iso empty string.
      strResult = Expression
   End If
   
   
   RegExp_Replace = strResult


Exit_Function:
   ''' reset changed applications settings here
   Set objMatch = Nothing
   Set objRegEx = Nothing
   Exit Function

Yes,

First, add some folder browser code to your code module.  This is from Chip Pearson's site and I use it often.  You could simply copy and paste it below the End Sub of your current process:

Function BrowseFolder(Title As String, _
    Optional InitialFolder As String = vbNullString, _
    Optional InitialView As Office.MsoFileDialogView = _
    msoFileDialogViewList) As String
'Source: http://www.cpearson.com/Excel/BrowseFolder.aspx
  Dim V As Variant
  Dim InitFolder As String
  With Application.FileDialog(msoFileDialogFolderPicker)
    .Title = Title
    .InitialView = InitialView
    If Len(InitialFolder) > 0 Then
      If Dir(InitialFolder, vbDirectory) <> vbNullString Then
        InitFolder = InitialFolder
        If Right(InitFolder, 1) <> "\" Then
          InitFolder = InitFolder & "\"
        End If
        .InitialFileName = InitFolder
      End If
    End If
    .Show
    On Error Resume Next
    Err.Clear
    V = .SelectedItems(1)
    If Err.Number <> 0 Then
      V = vbNullString
    End If
  End With
  BrowseFolder = CStr(V)
End Function

Let's say you have a birth date in cell A1 (as a date value, not as 19540301 - Excel does not recognize that as a date).

The eligibility date is given by the formula

Format the cell with the formula as a date.
