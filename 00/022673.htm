This kind of thing happens some of the times when Office updates get installed. To make Office 2010 the default for presentation files, do the following:

1. Open command prompt as Administrator:

1.1 In Windows 7, click Start button and search for "cmd".

1.2 Right-click on the cmd program found in the search results and click "Run as administrator" menu item.

2. Navigate to Office 2010 folder in program files. You can do that through the following command in command prompt:

cd "\Program Files\Microsoft Office\Office14"

3. Run the following in the command prompt:

powerpnt /regserver

4. Office 2010 should reconfigure itself and make it the default for presentations.

Here's the code:

 

Dim bSmartArtWarn As Boolean

Sub SetEnglish()
    changeLanguage (msoLanguageIDEnglishUK)
End Sub

Private Sub changeLanguage(lang As MsoLanguageID)
    Dim shp As Shape
    Dim sld As Slide
    Dim des As Design
    Dim clo As CustomLayout
    Dim strLang As String, strCurDefLang As String
   
    bSmartArtWarn = False
   
    strCurDefLang = "."
     'Get current default language:
    Select Case ActivePresentation.DefaultLanguageID
        Case 2057: strCurDefLang = ", which is: English (UK)"
        Case 1033: strCurDefLang = ", which is: English (US)"
        Case 1029: strCurDefLang = ", which is: Czech"
        Case 1031: strCurDefLang = ", which is: German"
        Case 1036: strCurDefLang = ", which is: French"
        Case 1040: strCurDefLang = ", which is: Italian"
        Case Else 'Do nothing
    End Select
   
     'Set default language in application
    If MsgBox("Do you want to change the defualt language for the active presentation" & vbCr _
        & "and all future presentations?" & vbCr & vbCr _
        & "If you press No, new content in the active presentation will have the current" & vbCr _
        & "default language setting" & strCurDefLang, vbYesNo, "Set Default Language?") = vbYes Then
   
        ActivePresentation.DefaultLanguageID = lang
    End If
   
     'Loop through each SLIDE MASTER (design)
    For Each des In ActivePresentation.Designs
        For Each shp In des.SlideMaster.Shapes
            Call SetShpLang(shp, lang)
        Next shp
       
         'Now loop through each CUSTOM LAYOUT in the Design/SlideMaster
        For Each clo In des.SlideMaster.CustomLayouts
            For Each shp In clo.Shapes
                Call SetShpLang(shp, lang)
            Next shp
        Next clo
    Next des
   
     'Now we check the NOTES MASTER
    For Each shp In ActivePresentation.NotesMaster.Shapes
        Call SetShpLang(shp, lang)
    Next shp
    
     'Now we check the HANDOUT MASTER
    For Each shp In ActivePresentation.HandoutMaster.Shapes
        Call SetShpLang(shp, lang)
    Next shp
   
     'Now we check each SLIDE
    For Each sld In ActivePresentation.Slides
        For Each shp In sld.Shapes
            Call SetShpLang(shp, lang)
        Next shp
       
         'Check the slide's NOTES PAGE
        For Each shp In sld.NotesPage.Shapes
            Call SetShpLang(shp, lang)
        Next shp
    Next sld
   
    Select Case lang
        Case 2057: strLang = "set to English (UK)"
        Case 1033: strLang = "set to English (US)"
        Case 1029: strLang = "set to Czech"
        Case 1031: strLang = "set to German"
        Case 1036: strLang = "set to French"
        Case 1040: strLang = "set to Italian"
        Case Else: strLang = "changed"
    End Select
   
    MsgBox "Language has been " & strLang & " throughout the presentation.", vbOKOnly + vbInformation, "Set Language"
   
    If bSmartArtWarn Then MsgBox "SmartArt objects were detected in this presentation." & vbCr & vbCr & _
        "Please set the language manually for all SmartArt objects." & vbCr & vbCr & _
        "To do this, select a shape inside each SmartArt collection, press Ctrl+A and then use Review tab > Language..." _
        , vbOKOnly + vbInformation, "Set Language"
End Sub

Private Sub SetShpLang(ByRef shp As Shape, lang As MsoLanguageID)
    Dim gi As GroupShapes
    Dim chtShp As Shape
    Dim R As Long, C As Long, i As Long
   
     'Check first if shp is a table:
    If shp.HasTable Then
        For R = 1 To shp.Table.Rows.Count
            For C = 1 To shp.Table.Columns.Count
                shp.Table.Cell(R, C).Shape.TextFrame.TextRange.LanguageID = lang
            Next C
        Next R
   
     'Check if shp has a chart:
    ElseIf shp.HasChart Then
         'Call this same procedure for each shape in the chart object:
        For Each chtShp In shp.Chart.Shapes
            Call SetShpLang(chtShp, lang)
        Next chtShp
   
    Else
        Set gi = Nothing
        On Error Resume Next
        Set gi = shp.GroupItems
        On Error GoTo 0
         'Check if shp is a group of shapes
        If Not gi Is Nothing Then
            On Error Resume Next
            If shp.GroupItems.Count > 0 Then
                 'Handle error -2147024891 (no group items despite shp.GroupItems being an object)
                If Err.Number = -2147024891 Then
                    shp.Parent.Select  'Select the slide
                    If Not Application.CommandBars.GetPressedMso("SelectionPane") Then
                        Application.CommandBars.ExecuteMso ("SelectionPane")
                    End If
                    shp.Select
                    MsgBox "Error setting language with shape """ & shp.Name _
                        & """ on slide " & shp.Parent.SlideIndex & "." & vbCr _
                        & "This shape may be corrupted. Click OK to continue." _
                        , vbExclamation + vbOKOnly, "Error"
                    Exit Sub
                End If
                On Error GoTo 0
                If shp.Type = msoSmartArt Then  'Can't do SmartArt in VBA 2007
                    bSmartArtWarn = True
                ElseIf shp.Type = msoPlaceholder Then
                    If shp.PlaceholderFormat.ContainedType = msoSmartArt Then
                        bSmartArtWarn = True
                    Else
                        GoTo Jump
                    End If
                Else
Jump:
                    For i = 1 To shp.GroupItems.Count
                         'If a SmartArt is grouped with other shapes, it doesn't have a textframe
                         'and its groupitems aren't checked so the next line is ok.
                        If shp.GroupItems(i).HasTextFrame Then
                            shp.GroupItems(i).TextFrame.TextRange.LanguageID = lang
                        End If
                    Next i
                End If
            End If
       
        Else 'shp is none of the above, it's just a simple shape, change the language ID
            If shp.HasTextFrame Then shp.TextFrame.TextRange.LanguageID = lang
        End If
    End If
End Sub
