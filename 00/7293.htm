Sub MoveActiveCell()

   'cut the contents of an active cell,
   'move up 1 cell,
   'then over 1 cell to the right & paste.
   'move 1 cell to the left,
   'down 2 cells,
   'cut the contents,
   'move up 2 cells
   'then over 2 cells to the right & paste.

   With ActiveCell
      If .Row > 1 And .Column < Columns.Count - 1 Then
         .Offset(-1, 1) = .Value
         .Offset(-1, 2) = .Offset(1)
         .Resize(2).ClearContents
         .Offset(-1, 2).Select
      End If
   End With
End Sub

ronrosenfeld say: Here is a macro solution.

To enter this Macro (Sub), <alt-F11> opens the Visual Basic Editor.
Ensure your project is highlighted in the Project Explorer window.
Then, from the top menu, select Insert/Module and
paste the code below into the window that opens.

To use this Macro (Sub), <alt-F8> opens the macro dialog box. Select the macro by name, and <RUN>.

Please read the notes within the macro.  As written, it will select the entries in column A beginning with A2 (to allow for a title or label row) and write the results in the adjacent column.  This can be easily changed within the macro.

========================================
Option Explicit
Sub SortByLastChar()
    Dim r As Range
    Dim v As Variant, v2() As Variant
    Dim i As Long
   
'Many ways to set up range.  This selects all the
'cells in column A beginning with A2

Set r = Range("A2", Cells(Cells.Rows.Count, "A").End(xlUp))

v = WorksheetFunction.Transpose(r)
ReDim v2(1 To 2, LBound(v) To UBound(v))
For i = LBound(v) To UBound(v)
    v2(1, i) = v(i)
    v2(2, i) = Right(v(i), 1)
Next i

MyQuickSort_Single v2, LBound(v2, 2), UBound(v2, 2), 2, True

'remove offset to overwrite the original
r.Offset(columnoffset:=1) = WorksheetFunction.Transpose(v2)

End Sub

' ***********************************************
'               Multidimensional Array sorted on a single dimensions
'  Note: May need to transpose before sorting
' ***********************************************
Sub MyQuickSort_Single(ByRef SortArray As Variant, ByVal First As Long, ByVal Last As Long, _
                        ByVal PrimeSort As Integer, ByVal Ascending As Boolean)
Dim Low As Long, High As Long
Dim Temp As Variant, List_Separator1 As Variant
Dim i As Long
Dim TempArray() As Variant
ReDim TempArray(UBound(SortArray, 1))
Low = First
High = Last
List_Separator1 = SortArray(PrimeSort, (First + Last) / 2)
Do
    If Ascending = True Then
        Do While (SortArray(PrimeSort, Low) < List_Separator1)
            Low = Low + 1
        Loop
        Do While (SortArray(PrimeSort, High) > List_Separator1)
            High = High - 1
        Loop
    Else
        Do While (SortArray(PrimeSort, Low) > List_Separator1)
            Low = Low + 1
        Loop
        Do While (SortArray(PrimeSort, High) < List_Separator1)
            High = High - 1
        Loop
    End If
    If (Low <= High) Then
        For i = LBound(SortArray, 1) To UBound(SortArray, 1)
            TempArray(i) = SortArray(i, Low)
        Next
        For i = LBound(SortArray, 1) To UBound(SortArray, 1)
            SortArray(i, Low) = SortArray(i, High)
        Next
        For i = LBound(SortArray, 1) To UBound(SortArray, 1)
            SortArray(i, High) = TempArray(i)
        Next
        Low = Low + 1
        High = High - 1
    End If
Loop While (Low <= High)
If (First < High) Then MyQuickSort_Single SortArray, First, High, PrimeSort, Ascending
If (Low < Last) Then MyQuickSort_Single SortArray, Low, Last, PrimeSort, Ascending
End Sub

Sub Reverse_Sort()
Dim LastColumn As Long, Lastrow As Long, i As Long
Dim reverse As String, MyRange As Range, c As Range
Dim MyCol As String
MyCol = "A" 'Change to suit
LastColumn = Cells.Find(What:="*", SearchOrder:=xlByColumns, _
               SearchDirection:=xlPrevious, LookIn:=xlFormulas).Column
Lastrow = Cells(Rows.Count, MyCol).End(xlUp).Row
Cells(1, LastColumn + 1).Resize(Lastrow).Value = Range(MyCol & "1:" & MyCol & Lastrow).Value
Set MyRange = Cells(1, LastColumn + 1).Resize(Lastrow)
For Each c In MyRange
    reverse = ""
    For i = Len(c.Value) To 1 Step -1
        reverse = reverse + Mid(c.Value, i, 1)
    Next
c.Value = reverse
Next
Cells(1, LastColumn + 2).Resize(Lastrow).Value = Range(MyCol & "1:" & MyCol & Lastrow).Value
ActiveSheet.Sort.SortFields.Clear
ActiveSheet.Sort.SortFields.Add Key:=MyRange, _
        SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:=xlSortNormal
    With ActiveSheet.Sort
        .SetRange MyRange.Resize(, 2)
        .Header = xlGuess
        .MatchCase = False
        .Orientation = xlTopToBottom
        .SortMethod = xlPinYin
        .Apply
End With
Range(MyCol & "1:" & MyCol & Lastrow).Value = Cells(1, LastColumn + 2).Resize(Lastrow).Value
MyRange.Resize(, 2).ClearContents
End Sub

neo yu msft say:

Hi John,

The email address with contoso.mail.onmicrosoft.com as suffix is not a UPN and is used a SMTP address which can be used to forward the Autodiscover record (by using the TargetAddress attribute). The error message is actually applied to the original test user credential, which is used when logging into the Autodiscover service. 

I suggest you check if the credential used for the user *** Email address is removed for privacy *** is correct. Also, please use the Office 365 global admin account to perform a password reset for the problematic user (or if you have password sync enabled, please reset the password in the on-premises AD server, then force a sync), then check if the issue persists.
