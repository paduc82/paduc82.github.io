    1. <spotlight_importer_path> should be similar to ‚Äò‚Äò/Applications/Microsoft Outlook.app/Contents/Library/Spotlight/Microsoft Outlook Spotlight Importer.mdimporter‚Äù unless you installed Outlook at a different location.

      • Main Profile is the default and could have a different name

Go to Finder, hold down the option key and click on the "Go" menu in the menu bar at top. Private Function BuildFilter() As Variant
    Dim varWhere As Variant
    Dim lngLen As Long
   
    varWhere = Null  ' Main filter
     
    ' Check for Category
    If Me.cboVWICategories > 0 Then
        varWhere = varWhere & "[CategoryID] = " & Me.cboVWICategories & " AND "
    End If
   
    ' Check for Department
    If Me.cboDepartment > 0 Then
        varWhere = varWhere & "[DepID] = " & Me.cboDepartment & " AND "
    End If
          
     'Check for LIKE Keyword Search
     If Me.txtSearchKeyword > "" Then
         varWhere = varWhere & "[Title] LIKE ""*" & Me.txtSearchKeyword.Text & "*"" AND "
     End If
     
    ' Check if there is a filter to return...
    If IsNull(varWhere) Then
        varWhere = ""
        ' msg if no data
        MsgBox "No criteria", vbInformation, "Nothing to do."
        Me.SearchSubform.Form.FilterOn = True
    Else
        varWhere = "WHERE " & varWhere
       
        ' strip off last "AND" in the filter
        If Right(varWhere, 5) = " AND " Then
            varWhere = Left(varWhere, Len(varWhere) - 5)
        End If
    End If
   
    BuildFilter = varWhere
   
End Function
