gord dibben say:

List of addresses and links on a new sheet.

 

Sub List_Hyperlinks()
'Rick Rothstein with edits by Gord Dibben
  Dim h As Hyperlink
  Dim Sh As Worksheet
  Dim CSh As Worksheet
  Set CSh = ActiveWorkbook.Worksheets.Add
  CSh.Name = "Hyperlinks"
  With CSh
     .Range("A1").Value = "Sheet Address"
     .Range("B1").Value = "Link Address"
  End With
  For Each Sh In ActiveWorkbook.Worksheets
    If Sh.Name <> CSh.Name Then
      For Each h In Sh.Hyperlinks
        With CSh.Cells(CSh.Rows.Count, "A").End(xlUp).Offset(1, 0)
          .Value = Sh.Name & " - " & h.Parent.Address
          .Offset(0, 1).Value = h.SubAddress
        End With
      Next
    End If
  Next
  Columns("A:B").AutoFit
End Sub

    Const SEARCH_RANGE = "A1:Z65000"
    
    MsgBox Range(SEARCH_RANGE).Hyperlinks.Count
    
    For Each strCELL In Range(SEARCH_RANGE).Hyperlinks
        MsgBox strCELL.SubAddress
    Next

For information on the actual cell, it is the parent so you can reference it like so:

     MsgBox strCELL.Parent.Row & "," & strCELL.Parent.Column

You can also activate it by doing a: strCELL.Parent.Activate  if you wanted to, and of course once you've gotten the row and column like shown above, you can replace it with something else.  If you write this out as a function, you can send call it with an argument that is the new value to reference so it can update the hyperlink, or in the case where you want to remove it you can just code a blank argument will remove the hyperlink.
