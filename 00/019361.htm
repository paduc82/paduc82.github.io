Use this macro (see http://www.gmayor.com/installing_macro.htm if needed):

You can't add unique RGB colors to the palette, but you can include them in
a table, paragraph, or character style. To create a new table style, go to
the Style dialog, click New, select Table as the style type (or you can
choose to modify an existing table style), and use Format | Stripes to
define the row banding (number of rows to alternate) and Format | Borders
and Shading to specify the color for the shaded rows (select "Odd row
stripes" or "Even row stripes" under "Apply formatting to").

hansv mvp say:

The export is problematic - Access requires an export specification for anything else than the default (comma separated), but this is dependent on the table structure. Try this version, which uses the default export settings:

Sub ImExp()
  Dim strIn As String
  Dim strOut As String
  Dim strFile As String
  Dim intPos As String
  Dim strExt As String
  Dim intType As Integer
  Dim strTable As String
  Dim strCSV As String

  With Application.FileDialog(4) ' msoFileDialogFolderPicker
    .Title = "Please select the source folder"
    If .Show Then
      strIn = .SelectedItems(1)
      If Not Right(strIn, 1) = "\" Then
        strIn = strIn & "\"
      End If
    Else
      MsgBox "You didn't select a source folder.", vbExclamation
      Exit Sub
    End If
    .Title = "Please select the target folder"
    If .Show Then
      strOut = .SelectedItems(1)
      If Not Right(strOut, 1) = "\" Then
        strOut = strOut & "\"
      End If
    Else
      MsgBox "You didn't select a target folder.", vbExclamation
      Exit Sub
    End If
  End With
  strFile = Dir(strIn & "*.xls*")
  Do While Not strFile = ""
    intPos = InStrRev(strFile, ".")
    strTable = Left(strFile, intPos - 1)
    strExt = Mid(strFile, intPos + 1)
    Select Case strExt
      Case "xlsx", "xlsm"
        intType = acSpreadsheetTypeExcel12Xml
      Case "xlsb"
        intType = acSpreadsheetTypeExcel12
      Case "xls"
        intType = acSpreadsheetTypeExcel8
      Case Else
        intType = False
    End Select
    If intType Then
      DoCmd.TransferSpreadsheet TransferType:=acImport, _
        SpreadsheetType:=intType, TableName:=strTable, _
        filename:=strIn & strFile, HasFieldNames:=True
      strCSV = strOut & strTable & ".csv"
      DoCmd.TransferText TransferType:=acExportDelim, _
        TableName:=strTable, filename:=strCSV, _
        HasFieldNames:=True
    End If
    strFile = Dir
  Loop
End Sub

You cannot have two different Skype accounts linked with the same Hotmail account.
