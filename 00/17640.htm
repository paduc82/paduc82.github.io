Nearly everthing you add to a master will appear behind "normal" objects.

Possible workarounds:

Can your employee pics be added as a backgound? Right click off slide > Format Backgrounfd > Picture Fill. They will be full screen if you do this though.

I would add the company info to slide one not the master and then copy it and PageDown followed by Ctrl V to paste on all other slides. Even with 150 this is fairly quick or you could do it with a macro.

Sub paster()
Dim osld As Slide
If ActiveWindow.Selection.Type = ppSelectionShapes Then
ActiveWindow.Selection.ShapeRange(1).Cut
For Each osld In ActivePresentation.Slides
ActiveWindow.View.GotoSlide (osld.SlideIndex)
osld.Shapes.Paste
Next osld
Else
MsgBox "Select the shape!"
End If
End Sub

How to use

If you really need the company image on the master the only way I know is to resize the footer placeholder > fill with a picture and in edit view INSERT > Headers and Footers > Footer> Apply to All. You will need to add text but a single period will probably look OK.

Option Explicit

Const lGoal As Long = 99

Sub AddRows()
    Dim oTbl As Table
    Set oTbl = ActiveWindow.Selection.ShapeRange(1).Table
    With oTbl
        Debug.Print .Rows.Count
        While .Rows.Count < lGoal
            .Rows.Add
        Wend
    End With
    
End Sub

Function GetRowOverFlowIndex(oShape As Shape, oPres As Presentation) As Long

    Dim Index As Long

    Dim sngSldHeight As Single

    Dim sngCurrHeight As Single

   

    sngSldHeight = oPres.PageSetup.SlideHeight

    'Get the top position of the shape on the slide

    sngCurrHeight = oShape.Top

   

    For Index = 1 To oShape.Table.Rows.count

        'Check if the current height exceeds that of the slide height

        If sngCurrHeight + oShape.Table.Rows(Index).Height > sngSldHeight Then

            'We have found the row at which the table moves off the slide.

            GetRowOverFlowIndex = Index

            Exit Function

        Else

            'Increment the current height

            sngCurrHeight = sngCurrHeight + oShape.Table.Rows(Index).Height

        End If

    Next

End Function

 

Sub HandleSplit(count As Integer, limit As Integer, oSlide As Slide, oSourceShape As Shape, RowIndex As Long)

            oSourceShape.Top, _

Sub SplitTable()

    Dim RowIndex As Long

    Dim oShp As Shape

    Dim oSld As Slide

    Dim I As Long

   

    Set oShp = ActiveWindow.Selection.ShapeRange(1)

    'Check if the selected shape is a table.

    If Not oShp.HasTable Then

        MsgBox "This is not a table.", vbExclamation

        Exit Sub

    End If

   

    'Get the row at which table moves off the slide

    RowIndex = GetRowOverFlowIndex(oShp, ActivePresentation)

   

    'If no rows are out of slide, just get out otherwise process it

    If RowIndex > 0 Then

       

            'Add a new slide for the a new table

        Set oSld = ActivePresentation.Slides.Add(oShp.Parent.SlideIndex + 1, oShp.Parent.Layout)

   

            'Now copy the rows to the new table.

        Call CopyToNewTable(oSld, oShp, RowIndex)

   

   

                'Delete the rows from the original table

        For I = oShp.Table.Rows.count To RowIndex Step -1

           

            oShp.Table.Rows(I).Delete

        Next

    End If

End Sub

Sub split_Table_across_slides()
    Dim oShp As Shape
    Dim copyToTable As Table
    Dim otbl As Table
    Dim tableSlideIndex As Integer
    Dim maxRow As Long
    Dim I As Long

    Set oShp = ActiveWindow.Selection.ShapeRange(1)
    If oShp.HasTable Then
        Set otbl = oShp.Table
    Else
        Exit Sub
    End If
    maxRow = getmaxRow(otbl)
    For I = 1 To oShp.Table.Rows.Count

        If I Mod maxRow = 1 Then
            Set copyToTable = createTable(tableSlideIndex + I / maxRow + 1, oShp.Table.Rows(I).Cells.Count, oShp.Table.Style)
        End If
        copyCell oShp.Table.Rows(I), copyToTable.Rows(copyToTable.Rows.Count)
        copyToTable.Rows.Add
        If copyToTable.Rows.Count > maxRow Then
            copyToTable.Rows(copyToTable.Rows.Count).Delete
        End If
    Next I
    oShp.Parent.Delete
End Sub

Function getmaxRow(otbl As Table) As Long
Dim irow As Integer
For irow = 1 To otbl.Rows.Count
If otbl.Cell(irow, 1).Shape.Top + otbl.Cell(irow, 1).Shape.Height > _
ActivePresentation.PageSetup.SlideHeight Then
getmaxRow = irow - 1
Exit Function
End If
Next irow
End Function

Function createTable(slideIndex As Integer, columnCount As Integer, sourceTableStyle As TableStyle) As Table
Dim newSlide As Slide
Dim newTable As Table
Set newSlide = ActivePresentation.Slides.AddSlide(slideIndex, ActivePresentation.Slides(1).CustomLayout)
For I = newSlide.Shapes.Count To 1 Step -1
    newSlide.Shapes(I).Delete
Next I
Set newTable = newSlide.Shapes.AddTable(1, columnCount).Table
newTable.ApplyStyle sourceTableStyle.Id
Set createTable = newTable
End Function

Sub copyCell(sourceRow As Row, targetRow As Row)
For I = 1 To sourceRow.Cells.Count
    targetRow.Cells(I).Shape.TextFrame.TextRange.Text = sourceRow.Cells(I).Shape.TextFrame.TextRange.Text
    targetRow.Cells(I).Shape.TextFrame.TextRange.Font.Name = sourceRow.Cells(I).Shape.TextFrame.TextRange.Font.Name
Next I
End Sub

How can you change the keyboard shortcuts for existing macros in Word - Microsoft Community

sbrookes say:

How can you change the keyboard shortcuts for existing macros in Word 2010?

You can see the keyboard shortcut easily when making a new macro, but when you try to edit an existing one, all you get is the Visual Basic screen.

Is there a way to do it without knowing visual Basic?

hansv mvp say:

Select File > Options.

Click Customize Ribbon.

Click Customize... at the bottom of the dialog, next to Keyboard Shortcuts.

Scroll down the list of Categories and select Macros.

You'll see a list of available macros in the list on the right.

Select a macro.

You'll see the keyboard shortcuts assigned to this macro (if any) in the Current keys box.

You can select a keyboard shortcut, then click Remove.

To create a new one, click in the Press new shortcut key box.

Press the desired new shortcut. You'll see if it has already been assigned to something else, and if so, to what.

To assign the shortcut to the macro, click Assign.

Click Close when you're done.
