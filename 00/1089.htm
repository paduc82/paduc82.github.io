The short answer to this is to remove the field from the header and replace it with a docvariable field
{ DocVariable varName } then populate and update the field with your code e.g

Change your code as follows

'Declare the variables used

Dim oSection As Section
Dim oHeader As HeaderFooter
Dim oFld As Field
Dim strAge as String
Dim strGender as String
Dim strName as String



strAge = ActiveDocument.Fields(5).Result

strGender = Left(ActiveDocument.Fields(4).Result, 1)

strName = Gender & "_" & Age

'MsgBox strName


ActiveDocument.Variables("varName").Value = strName

For Each oSection In ActiveDocument.Sections
    For Each oHeader In oSection.Headers
        If oHeader.Exists Then
            For Each oFld In oHeader.Range.Fields
                If oFld.Type = wdFieldDocVariable Then oFld.Update
            Next oFld
        End If
    Next oHeader
Next oSection
