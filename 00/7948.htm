    Try this

    Sub RangeFooter()
    'Define first row
    i = 74
    'Set Range. Note that you could callout
    'a named range if so desired
    For Each c In Range("C74:F81")
    If xFooter = "" Then
    'First cell
    xFooter = c.Text
    ElseIf c.Row > i Then
    'New row
    xFooter = xFooter & c.Text
    i = i + 1
    ElseIf c.Column = 6 Then
    'last column
    xFooter = xFooter & "  " & c.Text & vbCr
    Else
    xFooter = xFooter & "  " & c.Text
    End If
    Next
    'Now set the footer
    ActiveSheet.PageSetup.CenterFooter = xFooter
    End Sub
