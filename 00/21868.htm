I don't think this can be done in queries. It would work if you were searching for a Single keyword, but not for multiples. You might be able to generate a UNION query for each individual keyword.

Dim db As Database, rs As Recordset

Dim strSQL as String

Set db = CurrentDB()

Set rs = db.OpenRecordset("SELECT keywords FROM tblKeywords WHERE Group = " & Me.Group &";")

Do While NOT rs.EOF

     strSQL = "INSERT INTO tmpMatches (DescriptionID) " & _

                    "SELECT DescriptionID FROM tblDescriptions WHERE Description Like *" &  _

                     rs.Fields("Keyword") & "*;"

    CurrentDB..Execute strSQL

    rs.MoveNext

Loop
