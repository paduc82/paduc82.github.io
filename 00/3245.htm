andrewstegmaier say:

Richard,

This is Andrew Stegmaier, and I'm the Program Manager in charge of Surveys.

You're correct to note that the limit to the number of questions in a choices question comes from the number of characters in the "Choices" field, not the number of questions itself. 

When you hit "save" Excel Surveys is parsing what you've typed in the Choices field and adding a character to delimit separate questions. This parsed field can't exceed 1024 characters. So when you had a list of 283 choices that had 741 total (pre-parsing) characters, the parsing operation added another 283 characters, which took you to the 1024 limit (283 + 741 =1024).

-Andrew

Hi,

Please include the following code and follow the instructions to create the waterfall chart.

'Steps to create the Waterfall Chart:
'1.Copy the Range of cells for which the chart is to be created to Sheet 1
'2.Select the copied range. For example, A4:E5
'3.Press Ctrl+Q or Run Macro AddWaterfallChart
'4.The data required to build the Waterfall chart will be created in Sheet 2
'5. Row 2 in Sheet2 will serve as input data for the water fall chart. The rest of the cells are calculated automatically.
'6.The cells(upto row 7) in sheet 2 can be copied to other files and chart can be created based on the data
Sub AddWaterfallChart()
   Dim chtChart As Chart
   Dim sel1 As Range
   Set sel1 = Application.ActiveWindow.RangeSelection
   Dim Sum As String
   Sum = ""
   Sheet2.Cells.Clear
   numcolumns = sel1.Columns.Count
   Numrows = sel1.Rows.Count
   'Check for valid input (selection)
   If Numrows = 2 And numcolumns >= 2 Then
   'Create data for Waterfall Chart
   'Row 4 Initial and Final Values
   'Row 5 Hidden Series
   'Row 6 Negative increment
   'Row 7 Positive increment
  
   For X = 1 To numcolumns
    Sheet2.Cells(1, X).Formula = sel1(1, X).Formula
    Sheet2.Cells(2, X).Formula = sel1(2, X).Formula
    Sheet2.Cells(3, X).Formula = "=" + Sheet2.Cells(1, X).AddressLocal(RowAbsolute:=False, ColumnAbsolute:=False)
   Next
   Sheet2.Cells(4, 1).Formula = "=" + Sheet2.Cells(2, 1).AddressLocal(RowAbsolute:=False, ColumnAbsolute:=False)
   Sheet2.Cells(4, numcolumns).Formula = "=" + Sheet2.Cells(2, numcolumns).AddressLocal(RowAbsolute:=False, ColumnAbsolute:=False)
   For X = 2 To numcolumns - 1
    For y = 1 To X - 1
        Sum = Sum + "+" + Sheet2.Cells(2, y).AddressLocal(RowAbsolute:=False, ColumnAbsolute:=False)
    Next
    Sheet2.Cells(5, X).Formula = "=IF(" + Sheet2.Cells(2, X).AddressLocal(RowAbsolute:=False, ColumnAbsolute:=False) + "<0," + Sum + "+" + Sheet2.Cells(2, X).AddressLocal(RowAbsolute:=False, ColumnAbsolute:=False) + "," + Sum + ")"
    Sheet2.Cells(6, X).Formula = "=IF(" + Sheet2.Cells(2, X).AddressLocal(RowAbsolute:=False, ColumnAbsolute:=False) + "<0,Abs(" + Sheet2.Cells(2, X).AddressLocal(RowAbsolute:=False, ColumnAbsolute:=False) + "),0)"
    Sheet2.Cells(7, X).Formula = "=IF(" + Sheet2.Cells(2, X).AddressLocal(RowAbsolute:=False, ColumnAbsolute:=False) + "<0,0," + Sheet2.Cells(2, X).AddressLocal(RowAbsolute:=False, ColumnAbsolute:=False) + ")"
    Sum = ""
   Next
   If Sheet2.ChartObjects.Count = 0 Then
    Sheet2.ChartObjects.Add Left:=200, Top:=50, Width:=500, Height:=350
   End If
   Set chtChart = Sheet2.ChartObjects(1).Chart
   Set sel1 = Sheet2.Range(Sheet2.Cells(3, 1), Sheet2.Cells(7, numcolumns))
   With chtChart      'Set chart properties
    .ChartType = xlColumnStacked
    .SetSourceData Source:=sel1, PlotBy:=Excel.XlRowCol.xlRows
    .HasLegend = False
    .SeriesCollection(2).Fill.Visible = False
    .SeriesCollection(1).ApplyDataLabels Type:=xlDataLabelsShowValue
    .SeriesCollection(1).DataLabels.NumberFormat = "#,##0.0;-#,##0.0;"
    .SeriesCollection(4).ApplyDataLabels Type:=xlDataLabelsShowValue
    .SeriesCollection(4).DataLabels.NumberFormat = "#,##0.0;-#,##0.0;"
    .SeriesCollection(3).ApplyDataLabels Type:=xlDataLabelsShowValue
    .SeriesCollection(3).DataLabels.NumberFormat = "-#,##0.0;-#,##0.0;"
    .SeriesCollection(2).Border.LineStyle = xlNone
    .SeriesCollection(3).Interior.Color = RGB(255, 0, 0)
    .SeriesCollection(4).Interior.Color = RGB(0, 255, 0)
    MsgBox ("Please refer to output in Sheet2")
   End With
   Else
   MsgBox ("Selection should have two rows and atleast two columns")
   End If
  

End Sub
