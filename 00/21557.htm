Sub Formulate_All()
    Dim lngCol As Long
    Dim lngLastCol As Long
    Dim lngRow1 As Long
    Dim lngRow2 As Long
    Application.Calculation = xlCalculationAutomatic
    ' Speed up the macro by hiding what we do
    Application.ScreenUpdating = False
    ' Get the last used row in column A
    lngRow1 = Cells(Rows.Count, 1).End(xlUp).Row
    ' Get the last used column in row 2
    lngLastCol = Cells(2, Columns.Count).End(xlToLeft).Column
    ' Loop from column D to the last used column
    For lngCol = 4 To lngLastCol
        ' Get the first unused row in the column
        lngRow2 = Cells(Rows.Count, lngCol).End(xlUp).Row + 1
        With Range(Cells(lngRow2, lngCol), Cells(lngRow1, lngCol))
            ' Copy formula from row 1
            Cells(1, lngCol).Copy .Cells
            ' Replace formulas with values
            .Value = .Value
         End With
    Next lngCol
    ' Show the result
    Application.ScreenUpdating = True
End Sub
