The AppData Folder is a hidden one. You could make it visible by changing the settings in Explorer. But there is a more simple way to clear the OneNote cache:

Start OneNote with the parameter /safeboot. you can do that using WIN+R, then type OneNote /safeboot and hit return. A dialog come up showing you some options. Use the clear cache option and restart OneNote the normal way after that. Hope that will help.

Usually the cache of any notebook you actually close in OneNote is flushed. So closing all notebooks *should* have the same effect. But I recommend the method above.

bernie deitrick say:

1) Copy this code.
2) Open the VBE (press Alt-F11)
3) Select your workbook in the project explorer
4) Double-click the ThisWorkbook object
5) Paste the code into the window that appears.
6) Save the file as a macro-enabled .xlsm file.

Change the "A2" to the address of the cell that you use to store the date

Private Sub Workbook_Open()
    Dim shFirst As Worksheet
    Dim strAdd As String
    Dim dteV As Date
    
    strAdd = "A2"
    
    Set shFirst = Me.Worksheets(1)
    dteV = shFirst.Range(strAdd).Value
    If dteV < Date Then
        If Application.Weekday(dteV) > Application.Weekday(Date) Then
            Application.EnableEvents = False
            Application.DisplayAlerts = False
            Me.SaveCopyAs ThisWorkbook.Path & "\Workbook " & _
                Format(Me.Worksheets(Me.Worksheets.Count).Range(strAdd).Value, "yyyy-mm-dd") & _
                " to " & Format(dteV, "yyyy-mm-dd") & ".xlsm"
            For i = Me.Worksheets.Count To 2 Step -1
                Me.Worksheets(i).Delete
            Next i
            shFirst.Range(strAdd).Value = Date
            Me.Save
            Application.EnableEvents = False
            Application.DisplayAlerts = False
        Else
            shFirst.Copy Before:=shFirst
            With Me.Worksheets(1)
                .Range(strAdd).Value = Date
                .Name = Format(Date, "yyyy-mm-dd")
            End With
            Me.Save
        End If
    End If
    
End Sub
