By default, email messages that are sent from Exchange Online in Office 365 use the TNEF (Transport Neutral Encapsulation Format). 3. Run the command below:

Try the following and see if it is doing what you required. Note that there are other methods of finding a match but I used the method that you were initially attempting to use in the formula method.

Private Sub Worksheet_Activate()
    Dim Sht As Worksheet
    Dim ws As Worksheet
    Dim rngColA As Range
    Dim rngCel As Range
    Dim bFound As Boolean
  
    Set Sht = ActiveSheet
    
    With Sht
        Set rngColA = .Range(.Cells(2, "A"), .Cells(.Rows.Count, "A").End(xlUp))
    End With
   
    rngColA.Select
   
    For Each rngCel In rngColA
        bFound = False  'Initialize to not found
        For Each ws In Worksheets
            Select Case ws.Name
                'Edit following line and add/delete sheet names to be checked.
                Case "1", "2", "3", "4", "5A", "5", "6A", "6", "7", "8A", "8", "9"
                    With ws
                        If WorksheetFunction.CountIf(.Range(.Cells(2, "A"), _
                                .Cells(.Rows.Count, "A")), "=" & rngCel.Value) > 0 Then
                            bFound = True   'Set to True if match found
                            GoTo SetDate    'Match found so no further tests required
                        End If
                    End With
            End Select
        Next ws
SetDate:
        If bFound = True Then
            If rngCel.Offset(0, 10) = "" Then   'Enter Date if found and if Col Kx is blank
                rngCel.Offset(0, 10).Value = Date
            End If
        Else
            rngCel.Offset(0, 10).ClearContents   'ClearContents of Col Kx cell if no match found.
        End If
    Next rngCel

End Sub

.Range()
