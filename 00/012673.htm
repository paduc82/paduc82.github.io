    .Shapes(212).Fill.ForeColor.RGB = RGB(167, 0, 80)

Cell A | Cell B

greg maxey say:

Ah, moving the goal post after the start of the game ;-)

 

This is clunky but seems to work:

Sub ScratchMacro()
'A basic Word macro coded by Greg Maxey
Dim oTbl As Word.Table, oTblTemp As Word.Table
Dim lngIndex As Long
Dim oRng As Word.Range
Set oRng = ActiveDocument.Range
oRng.Collapse wdCollapseEnd
oRng.InsertBefore vbCr
Set oTblTemp = ActiveDocument.Tables.Add(oRng, 1, 2)
For lngIndex = 1 To ActiveDocument.Tables.Count - 1
  Set oTbl = ActiveDocument.Tables(lngIndex)
  oTbl.Cell(1, 2).Range.Cut
  oTblTemp.Cell(1, 1).Range.Paste
  oTbl.Cell(2, 1).Range.Cut
  oTbl.Cell(1, 2).Range.Paste
  oTblTemp.Cell(1, 1).Range.Copy 'Cut
  oTbl.Cell(2, 1).Range.Paste
Next lngIndex
oTblTemp.Delete
End Sub
