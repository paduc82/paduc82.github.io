This little demo file illustrates three ways to insert data into related tables via forms.  One uses subforms on separate pages of a tab control, the other two use separate 'linked' forms like yours.  In one case the 'linked' form is opened in dialogue mode, in the other case in synchronized mode.  In your case the former is probably more appropriate.

When the related form is opened it is filtered to the current ClientID by means of the WhereCondition argument of the OpenForm method.  The current ClientID value is also passed to the related form as the OpenArgs argument of the OpenForm method, e.g.

    Const FORMNAME = "frmEvents_2"
    Const MESSAGETEXT = "No current client record."
    Dim strCriteria As String
    
    strCriteria = "ClientID = " & Me.ClientID
    
    ' be sure a record has been entered
    If Not IsNull(Me.ClientID) Then
        ' save current client record
        Me.Dirty = False
    
        ' open linked form in dialogue mode
        DoCmd.OpenForm FORMNAME, _
            WhereCondition:=strCriteria, _
            WindowMode:=acDialog, _
            OpenArgs:=(Me.ClientID)
    Else
         MsgBox MESSAGETEXT, vbExclamation, "Invalid Operation"
    End If

If you wanted the related form to be opened empty for the insertion of new records only then, rather than filtering it by means of the WhereCondtion argument it would be opened at an empty new record by means of the DataMode argument:

        DoCmd.OpenForm FORMNAME, _
            DataMode:=acFormAdd, _
            WindowMode:=acDialog, _
            OpenArgs:=(Me.ClientID)

In the related form the value of the ClientID passed to the form is assigned to the form's cboClient combo box's DefaultValue property in the form's Open event procedure.  Note that this property is always a string expression, regardless of the data type of the column in question, so should be wrapped in literal quotes characters as below:

    If Not IsNull(Me.OpenArgs) Then
        Me.Event.SetFocus
        Me.cboClient.DefaultValue = """" & Me.OpenArgs & """"
    End If

The following Outlook macro will copy the sender name from a selected message to the clipboard.

Option Explicit

Sub GetSender()
Dim olMsg As MailItem
    On Error GoTo err_Handler
    Set olMsg = ActiveExplorer.Selection.Item(1)
    CopyAsText olMsg.SenderName
    MsgBox olMsg.SenderName & vbCr & "copied to clipboard"
lbl_Exit:
    Exit Sub
err_Handler:
    MsgBox "Sender name not copied. Did you select a message first?"
    Err.Clear
    GoTo lbl_Exit
End Sub

Private Sub CopyAsText(strText)
Dim dFname As DataObject
Dim fFname As String
    Set dFname = New DataObject
    fFname = strText
    dFname.SetText fFname
    dFname.PutInClipboard
lbl_Exit:
    Exit Sub
End Sub

    On Error GoTo Err_Handler
    Const MESSAGETEXT_1 = _
        "Please select a patient and treatment, and enter a start date first."
    Const MESSAGETEXT_2 = _
        "A treatment programme already exists.  Do you want to replace this?"
    Dim strCriteria As String
    Dim ctrl As Control
    Dim strSQL As String

    ' ensure all necessary data has been entered
    For Each ctrl In Me.Controls
        If ctrl.Tag = "LockMe" And IsNull(ctrl) Then
            MsgBox MESSAGETEXT_1, vbExclamation, "Invalid Operation"
            Exit Sub
        End If
    Next ctrl

    ' does a treatment programme for this patient/treatment/start date
    ' already exist?
    strCriteria = "PatientID = " & Me.PatientID & _
        " And TreatmentID = " & Me.TreatmentID & _
        " And StartDate = #" & Format(Me.StartDate, "yyyy-mm-dd") & "#"

    If Not IsNull(DLookup("PatientID", "PatientTreatmentProgramme", strCriteria)) Then
        If MsgBox(MESSAGETEXT_2, vbQuestion + vbYesNo, "Confirm") = vbNo Then
            Exit Sub
        Else
            strSQL = "DELETE * FROM PatientTreatmentProgramme WHERE " & strCriteria
            CurrentDb.Execute strSQL, dbFailOnError
        End If
    End If

    ' ensure current record is saved, then execute 'append'
    ' query to insert rows into PatientTreatmentProgramme table
    Me.Dirty = False
    DoCmd.SetWarnings False
    DoCmd.OpenQuery "qryAppendPatientProgramme"

    ' requery subform to show new rows
    Me.sfcPatientTreatmentProgramme.Requery

Exit_Here:
    DoCmd.SetWarnings True
    Exit Sub

Err_Handler:
    MsgBox Err.Description, vbExclamation, "Error"
    Resume Exit_Here

PARAMETERS [Forms]![frmPatientTreatments]![PatientID] LONG,

[StartDate]+[Interval], PatientTreatments.StartDate
