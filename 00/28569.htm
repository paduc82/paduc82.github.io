Sub mcr_Collect_Unique()
    Dim ws As Worksheet, wsu As Worksheet
    Set ws = ActiveSheet
    Set wsu = Sheets.Add(after:=Sheets(Sheets.Count))
    ws.Cells(1, 1).CurrentRegion.Copy Destination:=wsu.Cells(1, 1)
    With wsu.Cells(1, 1).CurrentRegion
