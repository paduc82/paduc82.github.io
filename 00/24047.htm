rich007a say:

You could use the macro code below.

 

In your target PowerPoint file, press Alt+F11 to open the VBE.
Do Insert > Module, then on the right, paste the code below. 

You'll need to edit the code where highlighted to include the path of the folder from where you want to import all the slides (include the trailing backshlash).


Close the VBE and go back to PowerPoint. 

Save your file as Macro enabled *.pptm

 

Now do View tab > Macros... > select ImportFiles > Run.

 

Hope that helps.

Cheers
Rich

 

 

Sub ImportFiles()
    Dim FileSys As Object, objFile As Object
    Dim myFolder As Variant
    Dim presSource As Presentation, presTarg As Presentation
    Dim sld As Slide
   
    Set presTarg = ActivePresentation
   
     'Set path for files - change for your folder:
    Const myDir As String = "C:\Temp\"               'include trailing backslash
   
     'Set up filesys objects:
    Set FileSys = CreateObject("Scripting.FileSystemObject")
    Set myFolder = FileSys.GetFolder(myDir)
   
     'Loop through each file and get date last modified:
    For Each objFile In myFolder.Files
'        Debug.Print objFile.Name, objFile.DateLastModified
        If InStr(1, objFile.Name, ".pptx") > 0 Then 'file is an PowerPoint file...
            presTarg.Slides.InsertFromFile objFile.Path, presTarg.Slides.Count
        End If
    Next objFile
   
    Set FileSys = Nothing
    Set myFolder = Nothing
End Sub
