graham mayor say:
The following macro should do the job to replace the appropriate levels with heading styles "Heading 1", "Heading 2" and "Heading 3". Modify those styles to include the required list numbering style and then run the macro.

Sub AddHeadingStyles()
Dim oPara As Paragraph
Dim oRng As Range
    For Each oPara In ActiveDocument.Paragraphs
        Set oRng = oPara.Range
        Select Case oRng.Words(1)
            Case Is = "I", "II", "III", "IV", "V", "VI", "VII", "VIII", "IX", "X"
                oRng.Collapse wdCollapseStart
                oRng.MoveEndUntil Chr(46)
                oRng.End = oRng.End + 3
                oRng.Select
                If MsgBox("Replace with Heading 1?", vbYesNo) = vbYes Then
                    oRng.Style = "Heading 1"
                    oRng.Delete
                End If
            Case Is = "  "
                oRng.Collapse wdCollapseStart
                oRng.MoveEndUntil Chr(46)
                oRng.End = oRng.End + 3
                oRng.Select
                If MsgBox("Replace with Heading 2?", vbYesNo) = vbYes Then
                    oRng.Style = "Heading 2"
                    oRng.Delete
                End If
            Case Is = "    "
                oRng.Collapse wdCollapseStart
                oRng.MoveEndUntil Chr(46)
                oRng.End = oRng.End + 3
                oRng.Select
                If MsgBox("Replace with Heading 3?", vbYesNo) = vbYes Then
                    oRng.Style = "Heading 3"
                    oRng.Delete
                End If
        End Select
    Next oPara
End Sub
