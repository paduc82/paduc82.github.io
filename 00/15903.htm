daniel pineault say:

You could do something along the lines of

Sub z()
    On Error GoTo Error_Handler
    Dim db                    As DAO.Database
    Dim rs                    As DAO.Recordset
    Dim dtQualityDate         As Date

    Set db = CurrentDb

    Set rst = db.OpenRecordset("Query1", dbOpenSnapshot)

    With rs
        If .RecordCount = 0 Then
            MsgBox "Houston we have a problem the query didn't return any records?!", vbCritical
            GoTo Error_Handler_Exit
        Else
            dtQualityDate = ![QualityDate]
        End If
    End With

    If dtQualityDate <= "ww/ww/wwww" Then
        DoCmd.OpenForm "FormA"
    ElseIf dtQualityDate Between #xx/xx/xxxx# AND #yy/yy/yyyy# Then
        DoCmd.OpenForm "FormB"
    ElseIf dtQualityDate Between #yy/yy/yyyy# AND #zz/zz/zzzz# Then
        DoCmd.OpenForm "FormC"
    ElseIf dtQualityDate >= #zz/zz/zzzz# Then
        DoCmd.OpenForm "FormD"
    Else
        'What is this for?
        DoCmd.CancelEvent
    End If

Error_Handler_Exit:
    On Error Resume Next
    If Not rs Is Nothing Then
        rs.Close
        Set rs = Nothing
    End If
    If Not db Is Nothing Then Set db = Nothing
    Exit Sub

Error_Handler:
    MsgBox "The following error has occured" & vbCrLf & vbCrLf & _
           "Error Number: " & Err.Number & vbCrLf & _
           "Error Source: z" & vbCrLf & _
           "Error Description: " & Err.Description & _
           Switch(Erl = 0, "", Erl <> 0, vbCrLf & "Line No: " & Erl) _
           , vbOKOnly + vbCritical, "An Error has Occured!"
    Resume Error_Handler_Exit
End Sub

Another alternative to the RecordSet approach would be to use DLookUp() instead.
