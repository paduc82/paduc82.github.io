SELECT HideCharacters([FirstNameField], "B", 2) as TheFirstNameMinus

                            

Public Function HideCharacters(strName As String, strFrom As String, intNum As Integer) As String
    'strName is the first name
    'strFrom is a text value of "B" beginning, "E" end, "R" as random
    'intNum is the number of character to remove

    Dim intLen As Integer  ' for the length of the strName
    Dim intMinChars        ' Minumum number of characters to return
    intMinChars = 1
    Dim strOut As String
    intLen = Len(strName)
    If intLen < intNum + intMinChars Then    ' don't want to remove all characters
        intNum = intLen - intMinChars
    End If
    Select Case strFrom
        Case "B"   'beginning
            strOut = String(intNum, "_") & Mid(strName, intNum + 1)

        Case "E"   'end
            strOut = Left(strName, intLen - intNum) & String(intNum, "_")

        Case "R"   'random
    End Select
    HideCharacters = strOut
End Function
