Private Sub Worksheet_Change(ByVal Target As Range)
    Dim wsLists As Worksheet
   
    Set wsLists = Worksheets("Sheet2")  'Edit "Sheet2" to the sheet name containing the lists
   
    If Target.Column = 1 And Target.Cells.Count = 1 Then        'Column 1 is column A
        On Error Resume Next    'Because following code errors if not part of the validation
        If wsLists.Range(Target.Value).Cells.Count = 1 Then     'If number of elements in the validation list is 1
            Target.Offset(0, 1) = wsLists.Range(Target.Value).Cells(1, 1)
        Else
            Target.Offset(0, 1).ClearContents
        End If
        On Error GoTo 0
    End If
End Sub
