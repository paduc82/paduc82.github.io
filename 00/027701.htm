Oops, sorry, I didn't take into account that you would have £ signs in the cells, or that some of the cells would be empty. Either of those inputs causes the CSng function (which converts a numeric string to the Single data type) to hit that error. Replace the previous code with this:

Sub Totals()
    Dim Total1 As Single
    Dim Total2 As Single
    Dim Total3 As Single
    Dim Total4 As Single
    Dim tbl As Table
    Dim lastrow As Long
    Dim rownum As Long
    Dim cellval As String
    Dim Test As String

        MsgBox "Please click in the table and rerun this macro."
        Exit Sub
    End If

    lastrow = tbl.Rows.Count

        cellval = tbl.Cell(Row:=rownum, Column:=5).Range.Text
        Total1 = Total1 + CellValue(cellval)

        Total2 = Total2 + CellValue(cellval)

        Total3 = Total3 + CellValue(cellval)

        Total4 = Total4 + CellValue(cellval)
    Next rownum

    ActiveDocument.Variables("Total2").Value = Format(Total2, "£#,##0.00")
    ActiveDocument.Variables("Total3").Value = Format(Total3, "£#,##0.00")
    ActiveDocument.Variables("Total4").Value = Format(Total4, "£#,##0.00")
    ActiveDocument.Sections(1).Footers(wdHeaderFooterPrimary).Range.Fields.Update
End Sub

    Dim strTemp As String  ' defaults to empty string, ""
    Dim i As Long
    Dim c As String

        c = Mid(cv, i, 1)
        If IsNumeric(c) Or InStr(".,", c) > 0 Then _
            strTemp = strTemp & Mid(cv, i, 1)

        CellValue = 0#

End Function
