gord dibben say:

Rows with Merged cells will not Autofit.

You need VBA code to do that.

Here is code from Greg Wilson.

NOTE:  Cells must first be formatted to wrap text and autofit.

Private Sub Worksheet_Change(ByVal Target As Range)
Dim NewRwHt As Single
Dim cWdth As Single, MrgeWdth As Single
Dim c As Range, cc As Range
Dim ma As Range

With Target
If .MergeCells And .WrapText Then
Set c = Target.Cells(1, 1)
cWdth = c.ColumnWidth
Set ma = c.MergeArea
For Each cc In ma.Cells
         MrgeWdth = MrgeWdth + cc.ColumnWidth
Next
Application.ScreenUpdating = False
     ma.MergeCells = False
      c.ColumnWidth = MrgeWdth
       c.EntireRow.AutoFit
        NewRwHt = c.RowHeight
       c.ColumnWidth = cWdth
     ma.MergeCells = True
    ma.RowHeight = NewRwHt
   cWdth = 0: MrgeWdth = 0
Application.ScreenUpdating = True
End If
End With
End Sub

Sub GetComments()

Dim Wb1 As Workbook

Dim Wb2 As Workbook

Dim List1 As Range

Dim List2 As Range

Dim StaffName As Range

Dim Stf1 As Range

Dim Stf2 As Range

Application.ScreenUpdating = False

 ''*************************************************************************

'' This will open WBook1 in case is closed

On Error Resume Next

Workbooks.Open ("C:\Users\User\Documents\WBook1.xlsm")

''*************************************************

''Part 1 Setting the variables

Set Wb1 = Application.Workbooks("WBook1.xlsm")

Set Wb2 = ThisWorkbook

Set List1 = Wb1.Worksheets(1).Range("Table1[[Name]]")

Set List2 = Wb2.Worksheets(1).Range("Table2[[Name]]")

'''**************************************************************

''Part 2  Setting all the cases

For Each StaffName In List2

    Set Stf1 = List1.Find(What:=StaffName)

    Set Stf2 = List2.Find(What:=StaffName)

    '''Case 1 if the Staff Name is not found in workbook1 then

    ''' Add the name to WBook1 and highlight-notify in Wbook2

        If Stf1 Is Nothing Then

            Stf2.Offset(0, 2).Value = "New Staff"

            Stf2.Offset(0, 2).Font.Color = RGB(255, 0, 0)

            List1.End(xlDown).Offset(1, 0).Value = StaffName

            GoTo Xs1

        End If

     '''Case 2 if No Data Inputted in WBook1 then highlight-notify in Wbook2

        If Stf1.Offset(0, 1).Value = "" Then

            Stf2.Offset(0, 1).Value = "No Data Inputted"

            Stf2.Offset(0, 1).Font.Color = RGB(255, 0, 0)

            Stf2.Offset(0, 2).Value = Stf1.Offset(0, 2).Value

            GoTo Xs2

        End If

      ''' Case 3 if Date doesn't Match then highlight-notify in Wbook2

        If Stf1.Offset(0, 1).Value <> Stf2.Offset(0, 1).Value Then

            Stf2.Offset(0, 2).Value = "Date doesn't Match"

            Stf2.Offset(0, 2).Font.Color = RGB(255, 0, 0)

        GoTo Xs3

        End If

        '''' Else transfer the Comments to WBook2

        Stf2.Offset(0, 2).Value = Stf1.Offset(0, 2).Value

Xs1:

Xs2:

Xs3:

Next StaffName

'' Close and save WBook1

'' Save WBook2

'' This is optional. So you can comment this line

Wb1.Close savechanges:=True

Wb2.Save

Application.ScreenUpdating = True

MsgBox "Done"

End Sub
