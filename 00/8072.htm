·         Click Start

·         Type msconfig into the search field

·         Press Enter on your keyboard

·         Click OK

ossiemac say:

Thought that maybe I should post the associated code for my previous post for anyone who is nervous about downloading the example file.

Create a simple Userform1 with a Label called "Label1".

Insert 2 Forms control buttons on the worksheet. The first one assign the macro "CallStartTimer" and the second on assign the macro "StopTimer" and edit the caption to suitable text like "Start Timer" and "Stop Timer"

Copy the following code and paste into a standard module:

'Public variables must be declared at top of a Standard Module before any subs
Public RunWhen As Date
Public UsrFrmVisible As Boolean

Sub CallStartTimer()
   
    '*****************************************************************************************
    'After the first time the code is run, IF THE WORKBOOK IS SAVED then
    'the code between the asterisk lines can be deleted
    'or commented out because no longer required.
    Dim varNameVal As Variant
    If Not NameExists("frmLeft") Then
        ActiveWorkbook.Names.Add Name:="frmLeft", RefersTo:="3"
    End If
  
    If Not NameExists("frmTop") Then
        ActiveWorkbook.Names.Add Name:="frmTop", RefersTo:="3"
    End If
    '******************************************************************************************
   
    Call StartTimer
   
End Sub

Sub StartTimer()
    Dim TimerIntervals
   
    If UsrFrmVisible = False Then
        Beep
        DoEvents
        UserForm1.Show
    Else
        Call StopTimer
        Exit Sub
    End If
   
    TimerIntervals = 3  'In seconds (Variable used in next line of code)
   
    RunWhen = Now + TimeSerial(0, 0, TimerIntervals)
   
    Application.OnTime _
        EarliestTime:=RunWhen, _
        Procedure:="StartTimer", _
        Schedule:=True
End Sub

Sub StopTimer()
'On Error required in case timer already stopped
'RunWhen must be same value that started timer

    On Error Resume Next
   
    Application.OnTime _
        EarliestTime:=RunWhen, _
        Procedure:="StartTimer", _
        Schedule:=False
       
    Unload UserForm1
    On Error GoTo 0
End Sub

Public Function NameExists(RangeName As String) As Boolean
    'This UDF used in Sub CallStartTimer.
    'If the code to test for existance of Defined Names
    'is disabled then this Function is no longer required.
  On Error Resume Next
  NameExists = Len(Names(RangeName).Name) <> 0
End Function

Then copy the following code and paste into the Userform module.


Private Sub UserForm_Initialize()
    Dim rngToTest As Range
    Dim cel As Range
   
    'If StartUpPosition property is set to Manual in Userform Properties
    'then the following line is superfluous.
    'Property must be Manual or userform position cannot be changed with code
    Me.StartUpPosition = 0
   
    With Worksheets("Sheet1")
        'Following 2 lines removes the equals sign from the defined name value
        'and assigns the values to the Userform Left and Right properties
        Me.Left = Replace(ActiveWorkbook.Names("frmLeft"), "=", "")
        Me.Top = Replace(ActiveWorkbook.Names("frmTop"), "=", "")
       
        Set rngToTest = .Range(.Cells(10, "D"), .Cells(13, "D"))
    End With
   
    For Each cel In rngToTest
        Select Case UCase(cel.Value)
            Case "DELAY"
                Me.Label1.Caption = cel.Offset(0, 1).Value
                Me.Label1.ForeColor = &HFF&
                Exit For
            Case "FASTER"
                Me.Label1.Caption = cel.Offset(0, 2).Value
                Me.Label1.ForeColor = &HC000&
                Exit For
        End Select
    Next cel
   
    UsrFrmVisible = True
End Sub

Private Sub UserForm_Terminate()
   
    UsrFrmVisible = False
End Sub

'See following link for more info on UserForm_QueryClose
'https://msdn.microsoft.com/en-us/library/office/gg278635.aspx?f=255&MSPPError=-2147217396
Private Sub UserForm_QueryClose(Cancel As Integer, CloseMode As Integer)
    'This code saves the last Userform position on Close Userform to Defined Names
    ActiveWorkbook.Names("frmLeft").Value = Me.Left
    ActiveWorkbook.Names("frmTop").Value = Me.Top
End Sub
