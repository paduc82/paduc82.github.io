Option Explicit

Sub Test()
  Dim A As String, B As String, S As String
  Dim BasicStart As Date, BasicEnd As Date, ClockStart As Date, ClockEnd As Date
  Dim OverTime As Date, BasicTime As Date
  Dim i As Long

  BasicStart = Range("A1")
  BasicEnd = Range("B1")
  ClockStart = Range("A3")
  ClockEnd = Range("B3")

  GanttString S, ClockStart, ClockEnd, "n"
  GanttString S, BasicStart, BasicEnd, Marker:=" "
  OverTime = Len(Replace(S, " ", "")) / 60 / 24

  With WorksheetFunction
    GanttString S, .Min(BasicStart, ClockStart), .Max(BasicEnd, ClockEnd), "n"
  End With
  A = S
  B = S
  GanttString A, BasicStart, BasicEnd, Marker:=" "
  GanttString B, ClockStart, ClockEnd, Marker:=" "
  For i = 1 To Len(S)
    Mid$(S, i, 1) = IIf(Mid$(A, i, 1) = Mid$(B, i, 1), "x", " ")
  Next
  BasicTime = Len(Replace(S, " ", "")) / 60 / 24
 
  Range("A5") = OverTime
  Range("B5") = BasicTime
End Sub

Private Sub Example_GanttString()
  Dim S As String
  'Create string with overlapping times
  GanttString S, CDate("01.09.2012 12:00"), CDate("01.09.2012 13:00"), "h"
  GanttString S, CDate("01.09.2012 12:00"), CDate("01.09.2012 14:00")
  GanttString S, CDate("01.09.2012 16:00"), CDate("01.09.2012 18:00")
  GanttString S, CDate("01.09.2012 18:00"), CDate("01.09.2012 20:00")
  MsgBox S & vbCrLf & "Elapsed: " & Len(S) - Len(Replace(S, "h", "")) & " Hours"

  S = ""
  'Create string with overlapping times
  GanttString S, CDate("12:00"), CDate("12:30"), "n"
  GanttString S, CDate("13:00"), CDate("13:09")
  GanttString S, CDate("12:20"), CDate("12:51")
  MsgBox S & vbCrLf & "Elapsed: " & Len(S) - Len(Replace(S, "n", "")) & " Minutes"

  S = ""
  'Whole day in minutes
  GanttString S, CDate("06:00"), CDate("14:15"), "n"
  'Subtract break times
  GanttString S, CDate("09:00"), CDate("09:15"), Marker:=" "
  GanttString S, CDate("12:00"), CDate("12:30")
  MsgBox S & vbCrLf & "Elapsed: " & Len(Replace(S, " ", "")) / 60 & " Hours"
End Sub

Sub GanttString(ByRef Elapsed As String, _
    Optional ByVal StartTime As Date, Optional ByVal EndTime As Date, _
    Optional ByVal Interval As String, Optional ByVal Marker)
  'Creates a Gantt string, exceeds an existing one
  'Interval could be
  '  y Year
  '  q Quarter
  '  m Month
  '  w Week
  '  d Day
  '  h Hour
  '  n Minute
  '  s Second
  Static MinTime As Date
  Static Resolution As Double
  Static Fill As String * 1
  Dim Pos As Long, Cnt As Long
  'Resolution given?
  If Len(Interval) > 0 Then
    Select Case LCase$(Left$(Interval, 1))
      Case "y": Resolution = 365
      Case "q": Resolution = 90
      Case "m": Resolution = 30
      Case "w": Resolution = 7
      Case "d": Resolution = 1
      Case "h": Resolution = 1 / 24
      Case "n": Resolution = 1 / 24 / 60
      Case "s": Resolution = 1 / 24 / 60 / 60
      Case Else
        Exit Sub
    End Select
    Fill = LCase$(Left$(Interval, 1))
  End If
  If Not IsMissing(Marker) Then Fill = Left$(Marker, 1)
  'Exit if bad input
  If EndTime - StartTime <= 0 Then Exit Sub
  If Resolution = 0 Then Exit Sub
  'First time called?
  If Len(Elapsed) = 0 Then MinTime = StartTime
  'Calculate the position and length of this time string
  Pos = (StartTime - MinTime) / Resolution
  Cnt = (EndTime - StartTime) / Resolution
  'Start exceeded?
  If Pos < 0 Then
    MinTime = StartTime
    Elapsed = Space(-Pos) & Elapsed
    Pos = 0
  End If
  'End exceeded?
  If Pos + Cnt > Len(Elapsed) Then
    Elapsed = Elapsed & Space((Pos + Cnt) - Len(Elapsed))
  End If
  'Draw the elapsed time string
  If Cnt > 0 Then
    Mid$(Elapsed, Pos + 1, Cnt) = String(Cnt, Fill)
  ElseIf Len(Elapsed) = 0 Then
    Elapsed = Space(1)
  End If
End Sub

The specific formula assumes you are only looking at four entries (1,5,9,13), as you wrote.

In the formula below Intvl2 is the distance between the reference cell and the variable cell:

=SUMPRODUCT(--(N(OFFSET(A1,{0,4,8,12},0))>

 =SUMPRODUCT(--(N(OFFSET(A1,Intvl1*(ROW(
