rich007a say:

Since you have merged cells, that complicates matters, so actually I used a â€œFor i = 1 Toâ€¦â€ Next loop.

First you need to set up your top two rows to have the correct number of days in a month, and the correct day names (Mon, Tue, etc) in row 2.
Doing the following keeps the first of the month in column B.

Put the following formula in cell AD1:
=IF(COLUMN()-1>DAY(DATE(YEAR(TODAY()),MONTH(DATEVALUE("1" & $A$1 & YEAR(TODAY())))+1,0)),"",COLUMN()-1)
and drag it across to cell AF1.

Put the following formula in cell B2:
=IF(B1<>"",CHOOSE(WEEKDAY(DATEVALUE(B1&$A$1&YEAR(TODAY())),2),"MON","TUE","WED","THU","FRI","SAT","SUN"),"")
and drag it across to cell AF2.

Now, the following macro should work:

'====================================================
Sub Roster()
    Dim rng As Range, i As Long, strtDate As Date
    Dim myMonthNum As Long, daysInMonth As Long, myDur As Long
    Dim myMonthName As String, mySubject As String
    Dim olApp As Outlook.Application, olApt As AppointmentItem
   
    Set olApp = New Outlook.Application
   
    myMonthName = Range("A1").Value
    myMonthName = Left(myMonthName, 3) 'Jan, Feb, Mar etc.
    myMonthNum = Month(DateValue("1 " & myMonthName & " " & Year(Date)))
    daysInMonth = Day(DateSerial(Year(Date), myMonthNum + 1, 1) - 1)

     'Find name, ofset by 1 column to first day of month.
    Set rng = Range("A3:A10000").Find(What:="Name15", LookIn:=xlFormulas, LookAt _
        :=xlPart, SearchOrder:=xlByRows, SearchDirection:=xlNext, MatchCase:= _
        False, SearchFormat:=False).Offset(0, 1)

    'rng.Select 'For debugging only
   
    For i = 1 To daysInMonth
         'Get start date
        strtDate = DateSerial(Year(Date), myMonthNum, i)
       
         'Get subject (may be blank)
        mySubject = rng.Value
       
         'Get duration
        If rng.MergeCells Then
             'Merged, so get num of merged columns
            myDur = rng.Offset(0, 1).Column - rng.Column
            i = i + myDur - 1 'Increment i so the loops stops at the right cell.
        Else
             'Not merged, so enter 1-day appointment
            myDur = 1
        End If
       
        If rng.Value <> vbNullString Then
            Set olApt = olApp.CreateItem(olAppointmentItem)
            With olApt
                .Start = strtDate
                .Duration = myDur * 60 * 24 'Convert days to minutes
                .AllDayEvent = True
                .Subject = mySubject
                .ReminderSet = False
                Select Case mySubject
                    Case "A", "B", "C": .Categories = "Blue Category"
                    Case "D", "E":      .Categories = "Red Category"
                    Case "Leave":       .Categories = "Holiday"
                    Case "Course":      .Categories = "Must Attend"
                    Case Else:          'Do nothing
                End Select
                '.Display 'For debugging only
                .Save
            End With
            Set olApt = Nothing
        End If

        Set rng = rng.Offset(0, 1)
        'rng.Select 'For debugging only
    Next i
    Set olApp = Nothing
End Sub
'====================================================

I don't think you can do it with a query alone. This code will get you all the users and stick them into the table.

Public Sub TestCode(strDomain As String, strGroup As String)
    Dim objGroup As Object
    Dim objRootDSE As Object
    Dim objNet As Object
    Dim strDomain As Variant
    Dim objUser As Object
    Dim db As DAO.Database
    Dim rs As DAO.Recordset
    Dim strSQL As String
    Dim strGroup As String
    Dim i As Integer

    Set db = CurrentDb
    'Get one record jus tso we have a record set to add to.
    strSQL = "SELECT TOP 1 * FROM ADUsers"
    Set rs = db.OpenRecordset(strSQL, dbOpenDynaset)

    Set objNet = CreateObject("WScript.Network")
    Set objRootDSE = GetObject("LDAP://rootDSE")
    Set objGroup = GetObject("WinNT://" & strDomain & "/" & strGroup & ",group")
    For Each objUser In objGroup.Members
        i = i + 1
        With rs
            .AddNew
            !UserName = objUser.Name
            .Update
        End With
    Next
    Set rs = Nothing
    Set db = Nothing
    Set objGroup = Nothing
    Set objRootDSE = Nothing
    Set objNet = Nothing
